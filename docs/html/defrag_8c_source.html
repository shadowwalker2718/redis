48 /* Defrag helper for generic allocations.
49  *
50  * returns NULL in case the allocatoin wasn't moved.
51  * when it returns a non-null value, the old pointer was already released
52  * and should NOT be accessed. */
53 void* activeDefragAlloc(void *ptr) {
54  int bin_util, run_util;
55  size_t size;
56  void *newptr;
57  if(!je_get_defrag_hint(ptr, &bin_util, &run_util)) {
58  server.stat_active_defrag_misses++;
59  return NULL;
60  }
61  /* if this run is more utilized than the average utilization in this bin
62  * (or it is full), skip it. This will eventually move all the allocations
63  * from relatively empty runs into relatively full runs. */
64  if (run_util > bin_util || run_util == 1<<16) {
65  server.stat_active_defrag_misses++;
66  return NULL;
67  }
68  /* move this allocation to a new allocation.
69  * make sure not to use the thread cache. so that we don't get back the same
70  * pointers we try to free */
71  size = zmalloc_size(ptr);
72  newptr = zmalloc_no_tcache(size);
73  memcpy(newptr, ptr, size);
74  zfree_no_tcache(ptr);
75  return newptr;
76 }

78 /*Defrag helper for sds strings
79  *
80  * returns NULL in case the allocatoin wasn't moved.
81  * when it returns a non-null value, the old pointer was already released
82  * and should NOT be accessed. */
83 sds activeDefragSds(sds sdsptr) {
84  void* ptr = sdsAllocPtr(sdsptr);
85  void* newptr = activeDefragAlloc(ptr);
86  if (newptr) {
87  size_t offset = sdsptr - (char*)ptr;
88  sdsptr = (char*)newptr + offset;
89  return sdsptr;
90  }
91  return NULL;
92 }

94 /* Defrag helper for robj and/or string objects
95  *
96  * returns NULL in case the allocatoin wasn't moved.
97  * when it returns a non-null value, the old pointer was already released
98  * and should NOT be accessed. */
99 robj *activeDefragStringOb(robj* ob, int *defragged) {
100  robj *ret = NULL;
101  if (ob->refcount!=1)
102  return NULL;
103 
104  /* try to defrag robj (only if not an EMBSTR type (handled below). */
105  if (ob->type!=OBJ_STRING || ob->encoding!=OBJ_ENCODING_EMBSTR) {
106  if ((ret = activeDefragAlloc(ob))) {
107  ob = ret;
108  (*defragged)++;
109  }
110  }
111 
112  /* try to defrag string object */
113  if (ob->type == OBJ_STRING) {
114  if(ob->encoding==OBJ_ENCODING_RAW) {
115  sds newsds = activeDefragSds((sds)ob->ptr);
116  if (newsds) {
117  ob->ptr = newsds;
118  (*defragged)++;
119  }
120  } else if (ob->encoding==OBJ_ENCODING_EMBSTR) {
121  /* The sds is embedded in the object allocation, calculate the
122  * offset and update the pointer in the new allocation. */
123  long ofs = (intptr_t)ob->ptr - (intptr_t)ob;
124  if ((ret = activeDefragAlloc(ob))) {
125  ret->ptr = (void*)((intptr_t)ret + ofs);
126  (*defragged)++;
127  }
128  } else if (ob->encoding!=OBJ_ENCODING_INT) {
129  serverPanic("Unknown string encoding");
130  }
131  }
132  return ret;
133 }

135 /* Defrag helper for dictEntries to be used during dict iteration (called on
136  * each step). Teturns a stat of how many pointers were moved. */
137 int dictIterDefragEntry(dictIterator *iter) {
138  /* This function is a little bit dirty since it messes with the internals
139  * of the dict and it's iterator, but the benefit is that it is very easy
140  * to use, and require no other chagnes in the dict. */
141  int defragged = 0;
142  dictht *ht;
143  /* Handle the next entry (if there is one), and update the pointer in the
144  * current entry. */
145  if (iter->nextEntry) {
146  dictEntry *newde = activeDefragAlloc(iter->nextEntry);
147  if (newde) {
148  defragged++;
149  iter->nextEntry = newde;
150  iter->entry->next = newde;
151  }
152  }
153  /* handle the case of the first entry in the hash bucket. */
154  ht = &iter->d->ht[iter->table];
155  if (ht->table[iter->index] == iter->entry) {
156  dictEntry *newde = activeDefragAlloc(iter->entry);
157  if (newde) {
158  iter->entry = newde;
159  ht->table[iter->index] = newde;
160  defragged++;
161  }
162  }
163  return defragged;
164 }

166 /* Defrag helper for dict main allocations (dict struct, and hash tables).
167  * receives a pointer to the dict* and implicitly updates it when the dict
168  * struct itself was moved. Returns a stat of how many pointers were moved. */
169 int dictDefragTables(dict** dictRef) {
170  dict *d = *dictRef;
171  dictEntry **newtable;
172  int defragged = 0;
173  /* handle the dict struct */
174  dict *newd = activeDefragAlloc(d);
175  if (newd)
176  defragged++, *dictRef = d = newd;
177  /* handle the first hash table */
178  newtable = activeDefragAlloc(d->ht[0].table);
179  if (newtable)
180  defragged++, d->ht[0].table = newtable;
181  /* handle the second hash table */
182  if (d->ht[1].table) {
183  newtable = activeDefragAlloc(d->ht[1].table);
184  if (newtable)
185  defragged++, d->ht[1].table = newtable;
186  }
187  return defragged;
188 }

191 void zslUpdateNode(zskiplist *zsl, zskiplistNode *oldnode, zskiplistNode *newnode, zskiplistNode **update) {
192  int i;
193  for (i = 0; i < zsl->level; i++) {
194  if (update[i]->level[i].forward == oldnode)
195  update[i]->level[i].forward = newnode;
196  }
197  serverAssert(zsl->header!=oldnode);
198  if (newnode->level[0].forward) {
199  serverAssert(newnode->level[0].forward->backward==oldnode);
200  newnode->level[0].forward->backward = newnode;
201  } else {
202  serverAssert(zsl->tail==oldnode);
203  zsl->tail = newnode;
204  }
205 }

207 /* Defrag helper for sorted set.
208  * Update the robj pointer, defrag the skiplist struct and return the new score
209  * reference. We may not access oldele pointer (not even the pointer stored in
210  * the skiplist), as it was already freed. Newele may be null, in which case we
211  * only need to defrag the skiplist, but not update the obj pointer.
212  * When return value is non-NULL, it is the score reference that must be updated
213  * in the dict record. */
214 double *zslDefrag(zskiplist *zsl, double score, sds oldele, sds newele) {
215  zskiplistNode *update[ZSKIPLIST_MAXLEVEL], *x, *newx;
216  int i;
217  sds ele = newele? newele: oldele;
218 
219  /* find the skiplist node referring to the object that was moved,
220  * and all pointers that need to be updated if we'll end up moving the skiplist node. */
221  x = zsl->header;
222  for (i = zsl->level-1; i >= 0; i--) {
223  while (x->level[i].forward &&
224  x->level[i].forward->ele != oldele && /* make sure not to access the
225  ->obj pointer if it matches
226  oldele */
227  (x->level[i].forward->score < score ||
228  (x->level[i].forward->score == score &&
229  sdscmp(x->level[i].forward->ele,ele) < 0)))
230  x = x->level[i].forward;
231  update[i] = x;
232  }
233 
234  /* update the robj pointer inside the skip list record. */
235  x = x->level[0].forward;
236  serverAssert(x && score == x->score && x->ele==oldele);
237  if (newele)
238  x->ele = newele;
239 
240  /* try to defrag the skiplist record itself */
241  newx = activeDefragAlloc(x);
242  if (newx) {
243  zslUpdateNode(zsl, x, newx, update);
244  return &newx->score;
245  }
246  return NULL;
247 }

249 /* Utility function that replaces an old key pointer in the dictionary with a
250  * new pointer. Additionally, we try to defrag the dictEntry in that dict.
251  * Oldkey mey be a dead pointer and should not be accessed (we get a
252  * pre-calculated hash value). Newkey may be null if the key pointer wasn't
253  * moved. Return value is the the dictEntry if found, or NULL if not found.
254  * NOTE: this is very ugly code, but it let's us avoid the complication of
255  * doing a scan on another dict. */
256 dictEntry* replaceSateliteDictKeyPtrAndOrDefragDictEntry(dict *d, sds oldkey, sds newkey, unsigned int hash, int *defragged) {
257  dictEntry **deref = dictFindEntryRefByPtrAndHash(d, oldkey, hash);
258  if (deref) {
259  dictEntry *de = *deref;
260  dictEntry *newde = activeDefragAlloc(de);
261  if (newde) {
262  de = *deref = newde;
263  (*defragged)++;
264  }
265  if (newkey)
266  de->key = newkey;
267  return de;
268  }
269  return NULL;
270 }

272 /* for each key we scan in the main dict, this function will attempt to defrag
273  * all the various pointers it has. Returns a stat of how many pointers were
274  * moved. */
275 int defragKey(redisDb *db, dictEntry *de) {
276  sds keysds = dictGetKey(de);
277  robj *newob, *ob;
278  unsigned char *newzl;
279  dict *d;
280  dictIterator *di;
281  int defragged = 0;
282  sds newsds;
283 
284  /* Try to defrag the key name. */
285  newsds = activeDefragSds(keysds);
286  if (newsds)
287  defragged++, de->key = newsds;
288  if (dictSize(db->expires)) {
289  /* Dirty code:
290  * I can't search in db->expires for that key after i already released
291  * the pointer it holds it won't be able to do the string compare */
292  uint64_t hash = dictGetHash(db->dict, de->key);
293  replaceSateliteDictKeyPtrAndOrDefragDictEntry(db->expires, keysds, newsds, hash, &defragged);
294  }
295 
296  /* Try to defrag robj and / or string value. */
297  ob = dictGetVal(de);
298  if ((newob = activeDefragStringOb(ob, &defragged))) {
299  de->v.val = newob;
300  ob = newob;
301  }
302 
303  if (ob->type == OBJ_STRING) {
304  /* Already handled in activeDefragStringOb. */
305  } else if (ob->type == OBJ_LIST) {
306  if (ob->encoding == OBJ_ENCODING_QUICKLIST) {
307  quicklist *ql = ob->ptr, *newql;
308  quicklistNode *node = ql->head, *newnode;
309  if ((newql = activeDefragAlloc(ql)))
310  defragged++, ob->ptr = ql = newql;
311  while (node) {
312  if ((newnode = activeDefragAlloc(node))) {
313  if (newnode->prev)
314  newnode->prev->next = newnode;
315  else
316  ql->head = newnode;
317  if (newnode->next)
318  newnode->next->prev = newnode;
319  else
320  ql->tail = newnode;
321  node = newnode;
322  defragged++;
323  }
324  if ((newzl = activeDefragAlloc(node->zl)))
325  defragged++, node->zl = newzl;
326  node = node->next;
327  }
328  } else if (ob->encoding == OBJ_ENCODING_ZIPLIST) {
329  if ((newzl = activeDefragAlloc(ob->ptr)))
330  defragged++, ob->ptr = newzl;
331  } else {
332  serverPanic("Unknown list encoding");
333  }
334  } else if (ob->type == OBJ_SET) {
335  if (ob->encoding == OBJ_ENCODING_HT) {
336  d = ob->ptr;
337  di = dictGetIterator(d);
338  while((de = dictNext(di)) != NULL) {
339  sds sdsele = dictGetKey(de);
340  if ((newsds = activeDefragSds(sdsele)))
341  defragged++, de->key = newsds;
342  defragged += dictIterDefragEntry(di);
343  }
344  dictReleaseIterator(di);
345  dictDefragTables((dict**)&ob->ptr);
346  } else if (ob->encoding == OBJ_ENCODING_INTSET) {
347  intset *is = ob->ptr;
348  intset *newis = activeDefragAlloc(is);
349  if (newis)
350  defragged++, ob->ptr = newis;
351  } else {
352  serverPanic("Unknown set encoding");
353  }
354  } else if (ob->type == OBJ_ZSET) {
355  if (ob->encoding == OBJ_ENCODING_ZIPLIST) {
356  if ((newzl = activeDefragAlloc(ob->ptr)))
357  defragged++, ob->ptr = newzl;
358  } else if (ob->encoding == OBJ_ENCODING_SKIPLIST) {
359  zset *zs = (zset*)ob->ptr;
360  zset *newzs;
361  zskiplist *newzsl;
362  struct zskiplistNode *newheader;
363  if ((newzs = activeDefragAlloc(zs)))
364  defragged++, ob->ptr = zs = newzs;
365  if ((newzsl = activeDefragAlloc(zs->zsl)))
366  defragged++, zs->zsl = newzsl;
367  if ((newheader = activeDefragAlloc(zs->zsl->header)))
368  defragged++, zs->zsl->header = newheader;
369  d = zs->dict;
370  di = dictGetIterator(d);
371  while((de = dictNext(di)) != NULL) {
372  double* newscore;
373  sds sdsele = dictGetKey(de);
374  if ((newsds = activeDefragSds(sdsele)))
375  defragged++, de->key = newsds;
376  newscore = zslDefrag(zs->zsl, *(double*)dictGetVal(de), sdsele, newsds);
377  if (newscore) {
378  dictSetVal(d, de, newscore);
379  defragged++;
380  }
381  defragged += dictIterDefragEntry(di);
382  }
383  dictReleaseIterator(di);
384  dictDefragTables(&zs->dict);
385  } else {
386  serverPanic("Unknown sorted set encoding");
387  }
388  } else if (ob->type == OBJ_HASH) {
389  if (ob->encoding == OBJ_ENCODING_ZIPLIST) {
390  if ((newzl = activeDefragAlloc(ob->ptr)))
391  defragged++, ob->ptr = newzl;
392  } else if (ob->encoding == OBJ_ENCODING_HT) {
393  d = ob->ptr;
394  di = dictGetIterator(d);
395  while((de = dictNext(di)) != NULL) {
396  sds sdsele = dictGetKey(de);
397  if ((newsds = activeDefragSds(sdsele)))
398  defragged++, de->key = newsds;
399  sdsele = dictGetVal(de);
400  if ((newsds = activeDefragSds(sdsele)))
401  defragged++, de->v.val = newsds;
402  defragged += dictIterDefragEntry(di);
403  }
404  dictReleaseIterator(di);
405  dictDefragTables((dict**)&ob->ptr);
406  } else {
407  serverPanic("Unknown hash encoding");
408  }
409  } else if (ob->type == OBJ_MODULE) {
410  /* Currently defragmenting modules private data types
411  * is not supported. */
412  } else {
413  serverPanic("Unknown object type");
414  }
415  return defragged;
416 }

441 /* Utility function to get the fragmentation ratio from jemalloc.
442  * It is critical to do that by comparing only heap maps that belown to
443  * jemalloc, and skip ones the jemalloc keeps as spare. Since we use this
444  * fragmentation ratio in order to decide if a defrag action should be taken
445  * or not, a false detection can cause the defragmenter to waste a lot of CPU
446  * without the possibility of getting any results. */
447 float getAllocatorFragmentation(size_t *out_frag_bytes) {
448  size_t epoch = 1, allocated = 0, resident = 0, active = 0, sz = sizeof(size_t);
449  /* Update the statistics cached by mallctl. */
450  je_mallctl("epoch", &epoch, &sz, &epoch, sz);
451  /* Unlike RSS, this does not include RSS from shared libraries and other non
452  * heap mappings. */
453  je_mallctl("stats.resident", &resident, &sz, NULL, 0);
454  /* Unlike resident, this doesn't not include the pages jemalloc reserves
455  * for re-use (purge will clean that). */
456  je_mallctl("stats.active", &active, &sz, NULL, 0);
457  /* Unlike zmalloc_used_memory, this matches the stats.resident by taking
458  * into account all allocations done by this process (not only zmalloc). */
459  je_mallctl("stats.allocated", &allocated, &sz, NULL, 0);
460  float frag_pct = ((float)active / allocated)*100 - 100;
461  size_t frag_bytes = active - allocated;
462  float rss_pct = ((float)resident / allocated)*100 - 100;
463  size_t rss_bytes = resident - allocated;
464  if(out_frag_bytes)
465  *out_frag_bytes = frag_bytes;
466  serverLog(LL_DEBUG,
467  "allocated=%zu, active=%zu, resident=%zu, frag=%.0f%% (%.0f%% rss), frag_bytes=%zu (%zu%% rss)",
468  allocated, active, resident, frag_pct, rss_pct, frag_bytes, rss_bytes);
469  return frag_pct;
470 }

475 /* Perform incremental defragmentation work from the serverCron.
476  * This works in a similar way to activeExpireCycle, in the sense that
477  * we do incremental work across calls. */
478 void activeDefragCycle(void) {
479  static int current_db = -1;
480  static unsigned long cursor = 0;
481  static redisDb *db = NULL;
482  static long long start_scan, start_stat;
483  unsigned int iterations = 0;
484  unsigned long long defragged = server.stat_active_defrag_hits;
485  long long start, timelimit;
486 
487  if (server.aof_child_pid!=-1 || server.rdb_child_pid!=-1)
488  return; /* Defragging memory while there's a fork will just do damage. */
489 
490  /* Once a second, check if we the fragmentation justfies starting a scan
491  * or making it more aggressive. */
492  run_with_period(1000) {
493  size_t frag_bytes;
494  float frag_pct = getAllocatorFragmentation(&frag_bytes);
495  /* If we're not already running, and below the threshold, exit. */
496  if (!server.active_defrag_running) {
497  if(frag_pct < server.active_defrag_threshold_lower || frag_bytes < server.active_defrag_ignore_bytes)
498  return;
499  }
500 
501  /* Calculate the adaptive aggressiveness of the defrag */
502  int cpu_pct = INTERPOLATE(frag_pct,
503  server.active_defrag_threshold_lower,
504  server.active_defrag_threshold_upper,
505  server.active_defrag_cycle_min,
506  server.active_defrag_cycle_max);
507  cpu_pct = LIMIT(cpu_pct,
508  server.active_defrag_cycle_min,
509  server.active_defrag_cycle_max);
510  /* We allow increasing the aggressiveness during a scan, but don't
511  * reduce it. */
512  if (!server.active_defrag_running ||
513  cpu_pct > server.active_defrag_running)
514  {
515  server.active_defrag_running = cpu_pct;
516  serverLog(LL_VERBOSE,
517  "Starting active defrag, frag=%.0f%%, frag_bytes=%zu, cpu=%d%%",
518  frag_pct, frag_bytes, cpu_pct);
519  }
520  }
521  if (!server.active_defrag_running)
522  return;
523 
524  /* See activeExpireCycle for how timelimit is handled. */
525  start = ustime();
526  timelimit = 1000000*server.active_defrag_running/server.hz/100;
527  if (timelimit <= 0) timelimit = 1;
528 
529  do {
530  if (!cursor) {
531  /* Move on to next database, and stop if we reached the last one. */
532  if (++current_db >= server.dbnum) {
533  long long now = ustime();
534  size_t frag_bytes;
535  float frag_pct = getAllocatorFragmentation(&frag_bytes);
536  serverLog(LL_VERBOSE,
537  "Active defrag done in %dms, reallocated=%d, frag=%.0f%%, frag_bytes=%zu",
538  (int)((now - start_scan)/1000), (int)(server.stat_active_defrag_hits - start_stat), frag_pct, frag_bytes);
539 
540  start_scan = now;
541  current_db = -1;
542  cursor = 0;
543  db = NULL;
544  server.active_defrag_running = 0;
545  return;
546  }
547  else if (current_db==0) {
548  /* Start a scan from the first database. */
549  start_scan = ustime();
550  start_stat = server.stat_active_defrag_hits;
551  }
552 
553  db = &server.db[current_db];
554  cursor = 0;
555  }
556 
557  do {
558  cursor = dictScan(db->dict, cursor, defragScanCallback, defragDictBucketCallback, db);
559  /* Once in 16 scan iterations, or 1000 pointer reallocations
560  * (if we have a lot of pointers in one hash bucket), check if we
561  * reached the tiem limit. */
562  if (cursor && (++iterations > 16 || server.stat_active_defrag_hits - defragged > 1000)) {
563  if ((ustime() - start) > timelimit) {
564  return;
565  }
566  iterations = 0;
567  defragged = server.stat_active_defrag_hits;
568  }
569  } while(cursor);
570  } while(1);
571 }
