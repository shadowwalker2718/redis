69  * allocator that will automatically release it all once the callback returns.

272 /* Release the chain of blocks used for pool allocations. */
273 void poolAllocRelease(RedisModuleCtx *ctx) {
274  RedisModulePoolAllocBlock *head = ctx->pa_head, *next;
275 
276  while(head != NULL) {
277  next = head->next;
278  zfree(head);
279  head = next;
280  }
281  ctx->pa_head = NULL;
282 }

284 /* Return heap allocated memory that will be freed automatically when the
285  * module callback function returns. Mostly suitable for small allocations
286  * that are short living and must be released when the callback returns
287  * anyway. The returned memory is aligned to the architecture word size
288  * if at least word size bytes are requested, otherwise it is just
289  * aligned to the next power of two, so for example a 3 bytes request is
290  * 4 bytes aligned while a 2 bytes request is 2 bytes aligned.
291  *
292  * There is no realloc style function since when this is needed to use the
293  * pool allocator is not a good idea.
294  *
295  * The function returns NULL if `bytes` is 0. */
296 void *RM_PoolAlloc(RedisModuleCtx *ctx, size_t bytes) {
297  if (bytes == 0) return NULL;
298  RedisModulePoolAllocBlock *b = ctx->pa_head;
299  size_t left = b ? b->size - b->used : 0;
300 
301  /* Fix alignment. */
302  if (left >= bytes) {
303  size_t alignment = REDISMODULE_POOL_ALLOC_ALIGN;
304  while (bytes < alignment && alignment/2 >= bytes) alignment /= 2;
305  if (b->used % alignment)
306  b->used += alignment - (b->used % alignment);
307  left = (b->used > b->size) ? 0 : b->size - b->used;
308  }
309 
310  /* Create a new block if needed. */
311  if (left < bytes) {
312  size_t blocksize = REDISMODULE_POOL_ALLOC_MIN_SIZE;
313  if (blocksize < bytes) blocksize = bytes;
314  b = zmalloc(sizeof(*b) + blocksize);
315  b->size = blocksize;
316  b->used = 0;
317  b->next = ctx->pa_head;
318  ctx->pa_head = b;
319  }
320 
321  char *retval = b->memory + b->used;
322  b->used += bytes;
323  return retval;
324 }

330 /* Create an empty key of the specified type. 'kp' must point to a key object
331  * opened for writing where the .value member is set to NULL because the
332  * key was found to be non existing.
333  *
334  * On success REDISMODULE_OK is returned and the key is populated with
335  * the value of the specified type. The function fails and returns
336  * REDISMODULE_ERR if:
337  *
338  * 1) The key is not open for writing.
339  * 2) The key is not empty.
340  * 3) The specified type is unknown.
341  */
342 int moduleCreateEmptyKey(RedisModuleKey *key, int type) {
343  robj *obj;
344 
345  /* The key must be open for writing and non existing to proceed. */
346  if (!(key->mode & REDISMODULE_WRITE) || key->value)
347  return REDISMODULE_ERR;
348 
349  switch(type) {
350  case REDISMODULE_KEYTYPE_LIST:
351  obj = createQuicklistObject();
352  quicklistSetOptions(obj->ptr, server.list_max_ziplist_size,
353  server.list_compress_depth);
354  break;
355  case REDISMODULE_KEYTYPE_ZSET:
356  obj = createZsetZiplistObject();
357  break;
358  case REDISMODULE_KEYTYPE_HASH:
359  obj = createHashObject();
360  break;
361  default: return REDISMODULE_ERR;
362  }
363  dbAdd(key->db,key->key,obj);
364  key->value = obj;
365  return REDISMODULE_OK;
366 }

368 /* This function is called in low-level API implementation functions in order
369  * to check if the value associated with the key remained empty after an
370  * operation that removed elements from an aggregate data type.
371  *
372  * If this happens, the key is deleted from the DB and the key object state
373  * is set to the right one in order to be targeted again by write operations
374  * possibly recreating the key if needed.
375  *
376  * The function returns 1 if the key value object is found empty and is
377  * deleted, otherwise 0 is returned. */
378 int moduleDelKeyIfEmpty(RedisModuleKey *key) {
379  if (!(key->mode & REDISMODULE_WRITE) || key->value == NULL) return 0;
380  int isempty;
381  robj *o = key->value;
382 
383  switch(o->type) {
384  case OBJ_LIST: isempty = listTypeLength(o) == 0; break;
385  case OBJ_SET: isempty = setTypeSize(o) == 0; break;
386  case OBJ_ZSET: isempty = zsetLength(o) == 0; break;
387  case OBJ_HASH : isempty = hashTypeLength(o) == 0; break;
388  default: isempty = 0;
389  }
390 
391  if (isempty) {
392  dbDelete(key->db,key->key);
393  key->value = NULL;
394  return 1;
395  } else {
396  return 0;
397  }
398 }

522 /* Helper for RM_CreateCommand(). Truns a string representing command
523  * flags into the command flags used by the Redis core.
524  *
525  * It returns the set of flags, or -1 if unknown flags are found. */
526 int commandFlagsFromString(char *s) {
527  int count, j;
528  int flags = 0;
529  sds *tokens = sdssplitlen(s,strlen(s)," ",1,&count);
530  for (j = 0; j < count; j++) {
531  char *t = tokens[j];
532  if (!strcasecmp(t,"write")) flags |= CMD_WRITE;
533  else if (!strcasecmp(t,"readonly")) flags |= CMD_READONLY;
534  else if (!strcasecmp(t,"admin")) flags |= CMD_ADMIN;
535  else if (!strcasecmp(t,"deny-oom")) flags |= CMD_DENYOOM;
536  else if (!strcasecmp(t,"deny-script")) flags |= CMD_NOSCRIPT;
537  else if (!strcasecmp(t,"allow-loading")) flags |= CMD_LOADING;
538  else if (!strcasecmp(t,"pubsub")) flags |= CMD_PUBSUB;
539  else if (!strcasecmp(t,"random")) flags |= CMD_RANDOM;
540  else if (!strcasecmp(t,"allow-stale")) flags |= CMD_STALE;
541  else if (!strcasecmp(t,"no-monitor")) flags |= CMD_SKIP_MONITOR;
542  else if (!strcasecmp(t,"fast")) flags |= CMD_FAST;
543  else if (!strcasecmp(t,"getkeys-api")) flags |= CMD_MODULE_GETKEYS;
544  else if (!strcasecmp(t,"no-cluster")) flags |= CMD_MODULE_NO_CLUSTER;
545  else break;
546  }
547  sdsfreesplitres(tokens,count);
548  if (j != count) return -1; /* Some token not processed correctly. */
549  return flags;
550 }

552 /* Register a new command in the Redis server, that will be handled by
553  * calling the function pointer 'func' using the RedisModule calling
554  * convention. The function returns REDISMODULE_ERR if the specified command
555  * name is already busy or a set of invalid flags were passed, otherwise
556  * REDISMODULE_OK is returned and the new command is registered.
557  *
558  * This function must be called during the initialization of the module
559  * inside the RedisModule_OnLoad() function. Calling this function outside
560  * of the initialization function is not defined.
561  *
562  * The command function type is the following:
563  *
564  * int MyCommand_RedisCommand(RedisModuleCtx *ctx, RedisModuleString **argv, int argc);
565  *
566  * And is supposed to always return REDISMODULE_OK.
567  *
568  * The set of flags 'strflags' specify the behavior of the command, and should
569  * be passed as a C string compoesd of space separated words, like for
570  * example "write deny-oom". The set of flags are:
571  *
572  * * **"write"**: The command may modify the data set (it may also read
573  * from it).
574  * * **"readonly"**: The command returns data from keys but never writes.
575  * * **"admin"**: The command is an administrative command (may change
576  * replication or perform similar tasks).
577  * * **"deny-oom"**: The command may use additional memory and should be
578  * denied during out of memory conditions.
579  * * **"deny-script"**: Don't allow this command in Lua scripts.
580  * * **"allow-loading"**: Allow this command while the server is loading data.
581  * Only commands not interacting with the data set
582  * should be allowed to run in this mode. If not sure
583  * don't use this flag.
584  * * **"pubsub"**: The command publishes things on Pub/Sub channels.
585  * * **"random"**: The command may have different outputs even starting
586  * from the same input arguments and key values.
587  * * **"allow-stale"**: The command is allowed to run on slaves that don't
588  * serve stale data. Don't use if you don't know what
589  * this means.
590  * * **"no-monitor"**: Don't propoagate the command on monitor. Use this if
591  * the command has sensible data among the arguments.
592  * * **"fast"**: The command time complexity is not greater
593  * than O(log(N)) where N is the size of the collection or
594  * anything else representing the normal scalability
595  * issue with the command.
596  * * **"getkeys-api"**: The command implements the interface to return
597  * the arguments that are keys. Used when start/stop/step
598  * is not enough because of the command syntax.
599  * * **"no-cluster"**: The command should not register in Redis Cluster
600  * since is not designed to work with it because, for
601  * example, is unable to report the position of the
602  * keys, programmatically creates key names, or any
603  * other reason.
604  */
605 int RM_CreateCommand(RedisModuleCtx *ctx, const char *name, RedisModuleCmdFunc cmdfunc, const char *strflags, int firstkey, int lastkey, int keystep) {
606  int flags = strflags ? commandFlagsFromString((char*)strflags) : 0;
607  if (flags == -1) return REDISMODULE_ERR;
608  if ((flags & CMD_MODULE_NO_CLUSTER) && server.cluster_enabled)
609  return REDISMODULE_ERR;
610 
611  struct redisCommand *rediscmd;
612  RedisModuleCommandProxy *cp;
613  sds cmdname = sdsnew(name);
614 
615  /* Check if the command name is busy. */
616  if (lookupCommand(cmdname) != NULL) {
617  sdsfree(cmdname);
618  return REDISMODULE_ERR;
619  }

622  * in the command table, so that the generic command that works as
623  * binding between modules and Redis, can know what function to call
624  * and what the module is.
625  *
626  * Note that we use the Redis command table 'getkeys_proc' in order to
627  * pass a reference to the command proxy structure. */
628  cp = zmalloc(sizeof(*cp));
629  cp->module = ctx->module;
630  cp->func = cmdfunc;
631  cp->rediscmd = zmalloc(sizeof(*rediscmd));

634  cp->rediscmd->arity = -1;

640  cp->rediscmd->microseconds = 0;
641  cp->rediscmd->calls = 0;

643  dictAdd(server.orig_commands,sdsdup(cmdname),cp->rediscmd);

689 /* Add a new object to release automatically when the callback returns. */
690 void autoMemoryAdd(RedisModuleCtx *ctx, int type, void *ptr) {
691  if (!(ctx->flags & REDISMODULE_CTX_AUTO_MEMORY)) return;
692  if (ctx->amqueue_used == ctx->amqueue_len) {
693  ctx->amqueue_len *= 2;
694  if (ctx->amqueue_len < 16) ctx->amqueue_len = 16;
695  ctx->amqueue = zrealloc(ctx->amqueue,sizeof(struct AutoMemEntry)*ctx->amqueue_len);
696  }
697  ctx->amqueue[ctx->amqueue_used].type = type;
698  ctx->amqueue[ctx->amqueue_used].ptr = ptr;
699  ctx->amqueue_used++;
700 }
701 
702 /* Mark an object as freed in the auto release queue, so that users can still
703  * free things manually if they want.
704  *
705  * The function returns 1 if the object was actually found in the auto memory
706  * pool, otherwise 0 is returned. */
707 int autoMemoryFreed(RedisModuleCtx *ctx, int type, void *ptr) {
708  if (!(ctx->flags & REDISMODULE_CTX_AUTO_MEMORY)) return 0;
709 
710  int count = (ctx->amqueue_used+1)/2;
711  for (int j = 0; j < count; j++) {
712  for (int side = 0; side < 2; side++) {
713  /* For side = 0 check right side of the array, for
714  * side = 1 check the left side instead (zig-zag scanning). */
715  int i = (side == 0) ? (ctx->amqueue_used - 1 - j) : j;
716  if (ctx->amqueue[i].type == type &&
717  ctx->amqueue[i].ptr == ptr)
718  {
719  ctx->amqueue[i].type = REDISMODULE_AM_FREED;
720 
721  /* Switch the freed element and the last element, to avoid growing
722  * the queue unnecessarily if we allocate/free in a loop */
723  if (i != ctx->amqueue_used-1) {
724  ctx->amqueue[i] = ctx->amqueue[ctx->amqueue_used-1];
725  }
726 
727  /* Reduce the size of the queue because we either moved the top
728  * element elsewhere or freed it */
729  ctx->amqueue_used--;
730  return 1;
731  }
732  }
733  }
734  return 0;
735 }

737 /* Release all the objects in queue. */
738 void autoMemoryCollect(RedisModuleCtx *ctx) {
739  if (!(ctx->flags & REDISMODULE_CTX_AUTO_MEMORY)) return;
740  /* Clear the AUTO_MEMORY flag from the context, otherwise the functions
741  * we call to free the resources, will try to scan the auto release
742  * queue to mark the entries as freed. */
743  ctx->flags &= ~REDISMODULE_CTX_AUTO_MEMORY;
744  int j;
745  for (j = 0; j < ctx->amqueue_used; j++) {
746  void *ptr = ctx->amqueue[j].ptr;
747  switch(ctx->amqueue[j].type) {
748  case REDISMODULE_AM_STRING: decrRefCount(ptr); break;
749  case REDISMODULE_AM_REPLY: RM_FreeCallReply(ptr); break;
750  case REDISMODULE_AM_KEY: RM_CloseKey(ptr); break;
751  }
752  }
753  ctx->flags |= REDISMODULE_CTX_AUTO_MEMORY;
754  zfree(ctx->amqueue);
755  ctx->amqueue = NULL;
756  ctx->amqueue_len = 0;
757  ctx->amqueue_used = 0;
758 }

829 /* Every call to this function, will make the string 'str' requiring
830  * an additional call to RedisModule_FreeString() in order to really
831  * free the string. Note that the automatic freeing of the string obtained
832  * enabling modules automatic memory management counts for one
833  * RedisModule_FreeString() call (it is just executed automatically).
834  *
835  * Normally you want to call this function when, at the same time
836  * the following conditions are true:
837  *
838  * 1) You have automatic memory management enabled.
839  * 2) You want to create string objects.
840  * 3) Those string objects you create need to live *after* the callback
841  * function(for example a command implementation) creating them returns.
842  *
843  * Usually you want this in order to store the created string object
844  * into your own data structure, for example when implementing a new data
845  * type.
846  *
847  * Note that when memory management is turned off, you don't need
848  * any call to RetainString() since creating a string will always result
849  * into a string that lives after the callback function returns, if
850  * no FreeString() call is performed. */
851 void RM_RetainString(RedisModuleCtx *ctx, RedisModuleString *str) {
852  if (!autoMemoryFreed(ctx,REDISMODULE_AM_STRING,str)) {
853  /* Increment the string reference counting only if we can't
854  * just remove the object from the list of objects that should
855  * be reclaimed. Why we do that, instead of just incrementing
856  * the refcount in any case, and let the automatic FreeString()
857  * call at the end to bring the refcount back at the desired
858  * value? Because this way we ensure that the object refcount
859  * value is 1 (instead of going to 2 to be dropped later to 1)
860  * after the call to this function. This is needed for functions
861  * like RedisModule_StringAppendBuffer() to work. */
862  incrRefCount(str);
863  }
864 }

1061 /* When RedisModule_ReplyWithArray() is used with the argument
1062  * REDISMODULE_POSTPONED_ARRAY_LEN, because we don't know beforehand the number
1063  * of items we are going to output as elements of the array, this function
1064  * will take care to set the array length.
1065  *
1066  * Since it is possible to have multiple array replies pending with unknown
1067  * length, this function guarantees to always set the latest array length
1068  * that was created in a postponed way.
1069  *
1070  * For example in order to output an array like [1,[10,20,30]] we
1071  * could write:
1072  *
1073  * RedisModule_ReplyWithArray(ctx,REDISMODULE_POSTPONED_ARRAY_LEN);
1074  * RedisModule_ReplyWithLongLong(ctx,1);
1075  * RedisModule_ReplyWithArray(ctx,REDISMODULE_POSTPONED_ARRAY_LEN);
1076  * RedisModule_ReplyWithLongLong(ctx,10);
1077  * RedisModule_ReplyWithLongLong(ctx,20);
1078  * RedisModule_ReplyWithLongLong(ctx,30);
1079  * RedisModule_ReplySetArrayLength(ctx,3); // Set len of 10,20,30 array.
1080  * RedisModule_ReplySetArrayLength(ctx,2); // Set len of top array
1081  *
1082  * Note that in the above example there is no reason to postpone the array
1083  * length, since we produce a fixed number of elements, but in the practice
1084  * the code may use an interator or other ways of creating the output so
1085  * that is not easy to calculate in advance the number of elements.
1086  */
1087 void RM_ReplySetArrayLength(RedisModuleCtx *ctx, long len) {
1088  client *c = moduleGetReplyClient(ctx);
1089  if (c == NULL) return;
1090  if (ctx->postponed_arrays_count == 0) {
1091  serverLog(LL_WARNING,
1092  "API misuse detected in module %s: "
1093  "RedisModule_ReplySetArrayLength() called without previous "
1094  "RedisModule_ReplyWithArray(ctx,REDISMODULE_POSTPONED_ARRAY_LEN) "
1095  "call.", ctx->module->name);
1096  return;
1097  }
1098  ctx->postponed_arrays_count--;
1099  setDeferredMultiBulkLength(c,
1100  ctx->postponed_arrays[ctx->postponed_arrays_count],
1101  len);
1102  if (ctx->postponed_arrays_count == 0) {
1103  zfree(ctx->postponed_arrays);
1104  ctx->postponed_arrays = NULL;
1105  }
1106 }

1188  * of execution of the calling command implementation.
1189  *
1190  * The replicated commands are always wrapped into the MULTI/EXEC that
1191  * contains all the commands replicated in a given module command
1192  * execution. However the commands replicated with RedisModule_Call()
1193  * are the first items, the ones replicated with RedisModule_Replicate()
1194  * will all follow before the EXEC.
1195  *
1196  * Modules should try to use one interface or the other.
1197  *

1199  * so a set of format specifiers must be passed, followed by arguments
1200  * matching the provided format specifiers.
1201  *
1202  * Please refer to RedisModule_Call() for more information.
1203  *
1204  * The command returns REDISMODULE_ERR if the format specifiers are invalid
1205  * or the command name does not belong to a known command. */
1206 int RM_Replicate(RedisModuleCtx *ctx, const char *cmdname, const char *fmt, ...) {
1207  struct redisCommand *cmd;

1211

1217  argv = moduleCreateArgvFromUserFormat(cmdname,fmt,&argc,&flags,ap);
1218  va_end(ap);

1506 /* Prepare the key associated string value for DMA access, and returns
1507  * a pointer and size (by reference), that the user can use to read or
1508  * modify the string in-place accessing it directly via pointer.
1509  *
1510  * The 'mode' is composed by bitwise OR-ing the following flags:
1511  *
1512  * REDISMODULE_READ -- Read access
1513  * REDISMODULE_WRITE -- Write access
1514  *
1515  * If the DMA is not requested for writing, the pointer returned should
1516  * only be accessed in a read-only fashion.
1517  *
1518  * On error (wrong type) NULL is returned.
1519  *
1520  * DMA access rules:
1521  *
1522  * 1. No other key writing function should be called since the moment
1523  * the pointer is obtained, for all the time we want to use DMA access
1524  * to read or modify the string.
1525  *
1526  * 2. Each time RM_StringTruncate() is called, to continue with the DMA
1527  * access, RM_StringDMA() should be called again to re-obtain
1528  * a new pointer and length.
1529  *
1530  * 3. If the returned pointer is not NULL, but the length is zero, no
1531  * byte can be touched (the string is empty, or the key itself is empty)
1532  * so a RM_StringTruncate() call should be used if there is to enlarge
1533  * the string, and later call StringDMA() again to get the pointer.
1534  */
1535 char *RM_StringDMA(RedisModuleKey *key, size_t *len, int mode) {
1536  /* We need to return *some* pointer for empty keys, we just return
1537  * a string literal pointer, that is the advantage to be mapped into
1538  * a read only memory page, so the module will segfault if a write
1539  * attempt is performed. */
1540  char *emptystring = "<dma-empty-string>";
1541  if (key->value == NULL) {
1542  *len = 0;
1543  return emptystring;
1544  }
1545 
1546  if (key->value->type != OBJ_STRING) return NULL;
1547 
1548  /* For write access, and even for read access if the object is encoded,
1549  * we unshare the string (that has the side effect of decoding it). */
1550  if ((mode & REDISMODULE_WRITE) || key->value->encoding != OBJ_ENCODING_RAW)
1551  key->value = dbUnshareStringValue(key->db, key->key, key->value);
1552 
1553  *len = sdslen(key->value->ptr);
1554  return key->value->ptr;
1555 }

1557 /* If the string is open for writing and is of string type, resize it, padding
1558  * with zero bytes if the new length is greater than the old one.
1559  *
1560  * After this call, RM_StringDMA() must be called again to continue
1561  * DMA access with the new pointer.
1562  *
1563  * The function returns REDISMODULE_OK on success, and REDISMODULE_ERR on
1564  * error, that is, the key is not open for writing, is not a string
1565  * or resizing for more than 512 MB is requested.
1566  *
1567  * If the key is empty, a string key is created with the new string value
1568  * unless the new length value requested is zero. */
1569 int RM_StringTruncate(RedisModuleKey *key, size_t newlen) {
1570  if (!(key->mode & REDISMODULE_WRITE)) return REDISMODULE_ERR;
1571  if (key->value && key->value->type != OBJ_STRING) return REDISMODULE_ERR;
1572  if (newlen > 512*1024*1024) return REDISMODULE_ERR;
1573 
1574  /* Empty key and new len set to 0. Just return REDISMODULE_OK without
1575  * doing anything. */
1576  if (key->value == NULL && newlen == 0) return REDISMODULE_OK;
1577 
1578  if (key->value == NULL) {
1579  /* Empty key: create it with the new size. */
1580  robj *o = createObject(OBJ_STRING,sdsnewlen(NULL, newlen));
1581  setKey(key->db,key->key,o);
1582  key->value = o;
1583  decrRefCount(o);
1584  } else {
1585  /* Unshare and resize. */
1586  key->value = dbUnshareStringValue(key->db, key->key, key->value);
1587  size_t curlen = sdslen(key->value->ptr);
1588  if (newlen > curlen) {
1589  key->value->ptr = sdsgrowzero(key->value->ptr,newlen);
1590  } else if (newlen < curlen) {
1591  sdsrange(key->value->ptr,0,newlen-1);
1592  /* If the string is too wasteful, reallocate it. */
1593  if (sdslen(key->value->ptr) < sdsavail(key->value->ptr))
1594  key->value->ptr = sdsRemoveFreeSpace(key->value->ptr);
1595  }
1596  }
1597  return REDISMODULE_OK;
1598 }

2679 /* Turn a 9 chars name in the specified charset and a 10 bit encver into
2680  * a single 64 bit unsigned integer that represents this exact module name
2681  * and version. This final number is called a "type ID" and is used when
2682  * writing module exported values to RDB files, in order to re-associate the
2683  * value to the right module to load them during RDB loading.
2684  *
2685  * If the string is not of the right length or the charset is wrong, or
2686  * if encver is outside the unsigned 10 bit integer range, 0 is returned,
2687  * otherwise the function returns the right type ID.
2688  *
2689  * The resulting 64 bit integer is composed as follows:
2690  *
2691  * (high order bits) 6|6|6|6|6|6|6|6|6|10 (low order bits)
2692  *
2693  * The first 6 bits value is the first character, name[0], while the last
2694  * 6 bits value, immediately before the 10 bits integer, is name[8].
2695  * The last 10 bits are the encoding version.
2696  *
2697  * Note that a name and encver combo of "AAAAAAAAA" and 0, will produce
2698  * zero as return value, that is the same we use to signal errors, thus
2699  * this combination is invalid, and also useless since type names should
2700  * try to be vary to avoid collisions. */
2701 
2702 const char *ModuleTypeNameCharSet =
2703  "ABCDEFGHIJKLMNOPQRSTUVWXYZ"
2704  "abcdefghijklmnopqrstuvwxyz"
2705  "0123456789-_";
2706 
2707 uint64_t moduleTypeEncodeId(const char *name, int encver) {
2708  /* We use 64 symbols so that we can map each character into 6 bits
2709  * of the final output. */
2710  const char *cset = ModuleTypeNameCharSet;
2711  if (strlen(name) != 9) return 0;
2712  if (encver < 0 || encver > 1023) return 0;
2713 
2714  uint64_t id = 0;
2715  for (int j = 0; j < 9; j++) {
2716  char *p = strchr(cset,name[j]);
2717  if (!p) return 0;
2718  unsigned long pos = p-cset;
2719  id = (id << 6) | pos;
2720  }
2721  id = (id << 10) | encver;
2722  return id;
2723 }

2831  * still load old data produced by an older version if the rdb_load
