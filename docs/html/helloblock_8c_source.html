61 /* The thread entry point that actually executes the blocking part
62  * of the command HELLO.BLOCK. */
63 void *HelloBlock_ThreadMain(void *arg) {
64  void **targ = arg;
65  RedisModuleBlockedClient *bc = targ[0];
66  long long delay = (unsigned long)targ[1];
67  RedisModule_Free(targ);
68 
69  sleep(delay);
70  int *r = RedisModule_Alloc(sizeof(int));
71  *r = rand();
72  RedisModule_UnblockClient(bc,r);
73  return NULL;
74 }

76 /* HELLO.BLOCK <delay> <timeout> -- Block for <count> seconds, then reply with
77  * a random number. Timeout is the command timeout, so that you can test
78  * what happens when the delay is greater than the timeout. */
79 int HelloBlock_RedisCommand(RedisModuleCtx *ctx, RedisModuleString **argv, int argc) {
80  if (argc != 3) return RedisModule_WrongArity(ctx);
81  long long delay;
82  long long timeout;
83 
84  if (RedisModule_StringToLongLong(argv[1],&delay) != REDISMODULE_OK) {
85  return RedisModule_ReplyWithError(ctx,"ERR invalid count");
86  }
87 
88  if (RedisModule_StringToLongLong(argv[2],&timeout) != REDISMODULE_OK) {
89  return RedisModule_ReplyWithError(ctx,"ERR invalid count");
90  }
91 
92  pthread_t tid;
93  RedisModuleBlockedClient *bc = RedisModule_BlockClient(ctx,HelloBlock_Reply,HelloBlock_Timeout,HelloBlock_FreeData,timeout);
94 
95  /* Now that we setup a blocking client, we need to pass the control
96  * to the thread. However we need to pass arguments to the thread:
97  * the delay and a reference to the blocked client handle. */
98  void **targ = RedisModule_Alloc(sizeof(void*)*2);
99  targ[0] = bc;
100  targ[1] = (void*)(unsigned long) delay;
101 
102  if (pthread_create(&tid,NULL,HelloBlock_ThreadMain,targ) != 0) {
103  RedisModule_AbortBlock(bc);
104  return RedisModule_ReplyWithError(ctx,"-ERR Can't start thread");
105  }
106  return REDISMODULE_OK;
107 }

109 /* The thread entry point that actually executes the blocking part
110  * of the command HELLO.KEYS.
111  *
112  * Note: this implementation is very simple on purpose, so no duplicated
113  * keys (returned by SCAN) are filtered. However adding such a functionality
114  * would be trivial just using any data structure implementing a dictionary
115  * in order to filter the duplicated items. */
116 void *HelloKeys_ThreadMain(void *arg) {
117  RedisModuleBlockedClient *bc = arg;
118  RedisModuleCtx *ctx = RedisModule_GetThreadSafeContext(bc);
119  long long cursor = 0;
120  size_t replylen = 0;
121 
122  RedisModule_ReplyWithArray(ctx,REDISMODULE_POSTPONED_ARRAY_LEN);
123  do {
124  RedisModule_ThreadSafeContextLock(ctx);
125  RedisModuleCallReply *reply = RedisModule_Call(ctx,
126  "SCAN","l",(long long)cursor);
127  RedisModule_ThreadSafeContextUnlock(ctx);
128 
129  RedisModuleCallReply *cr_cursor =
130  RedisModule_CallReplyArrayElement(reply,0);
131  RedisModuleCallReply *cr_keys =
132  RedisModule_CallReplyArrayElement(reply,1);
133 
134  RedisModuleString *s = RedisModule_CreateStringFromCallReply(cr_cursor);
135  RedisModule_StringToLongLong(s,&cursor);
136  RedisModule_FreeString(ctx,s);
137 
138  size_t items = RedisModule_CallReplyLength(cr_keys);
139  for (size_t j = 0; j < items; j++) {
140  RedisModuleCallReply *ele =
141  RedisModule_CallReplyArrayElement(cr_keys,j);
142  RedisModule_ReplyWithCallReply(ctx,ele);
143  replylen++;
144  }
145  RedisModule_FreeCallReply(reply);
146  } while (cursor != 0);
147  RedisModule_ReplySetArrayLength(ctx,replylen);
148 
149  RedisModule_FreeThreadSafeContext(ctx);
150  RedisModule_UnblockClient(bc,NULL);
151  return NULL;
152 }

154 /* HELLO.KEYS -- Return all the keys in the current database without blocking
155  * the server. The keys do not represent a point-in-time state so only the keys
156  * that were in the database from the start to the end are guaranteed to be
157  * there. */
158 int HelloKeys_RedisCommand(RedisModuleCtx *ctx, RedisModuleString **argv, int argc) {
159  REDISMODULE_NOT_USED(argv);
160  if (argc != 1) return RedisModule_WrongArity(ctx);
161 
162  pthread_t tid;
163 
164  /* Note that when blocking the client we do not set any callback: no
165  * timeout is possible since we passed '0', nor we need a reply callback
166  * because we'll use the thread safe context to accumulate a reply. */
167  RedisModuleBlockedClient *bc = RedisModule_BlockClient(ctx,NULL,NULL,NULL,0);
168 
169  /* Now that we setup a blocking client, we need to pass the control
170  * to the thread. However we need to pass arguments to the thread:
171  * the reference to the blocked client handle. */
172  if (pthread_create(&tid,NULL,HelloKeys_ThreadMain,bc) != 0) {
173  RedisModule_AbortBlock(bc);
174  return RedisModule_ReplyWithError(ctx,"-ERR Can't start thread");
175  }
176  return REDISMODULE_OK;
177 }

179 /* This function must be present on each Redis module. It is used in order to
180  * register the commands into the Redis server. */
181 int RedisModule_OnLoad(RedisModuleCtx *ctx, RedisModuleString **argv, int argc) {
182  REDISMODULE_NOT_USED(argv);
183  REDISMODULE_NOT_USED(argc);
184 
185  if (RedisModule_Init(ctx,"helloblock",1,REDISMODULE_APIVER_1)
186  == REDISMODULE_ERR) return REDISMODULE_ERR;
187 
188  if (RedisModule_CreateCommand(ctx,"hello.block",
189  HelloBlock_RedisCommand,"",0,0,0) == REDISMODULE_ERR)
190  return REDISMODULE_ERR;
191  if (RedisModule_CreateCommand(ctx,"hello.keys",
192  HelloKeys_RedisCommand,"",0,0,0) == REDISMODULE_ERR)
193  return REDISMODULE_ERR;
194 
195  return REDISMODULE_OK;
196 }
