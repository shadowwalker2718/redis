36 /* Every stream item inside the listpack, has a flags field that is used to
37  * mark the entry as deleted, or having the same field as the "master"
38  * entry at the start of the listpack> */
39 #define STREAM_ITEM_FLAG_NONE 0 /* No special flags. */
40 #define STREAM_ITEM_FLAG_DELETED (1<<0) /* Entry is delted. Skip it. */
41 #define STREAM_ITEM_FLAG_SAMEFIELDS (1<<1) /* Same fields as master entry. */

47 /* Create a new stream data structure. */
48 stream *streamNew(void) {
49  stream *s = zmalloc(sizeof(*s));
50  s->rax = raxNew();
51  s->length = 0;
52  s->last_id.ms = 0;
53  s->last_id.seq = 0;
54  return s;
55 }

63 /* Generate the next stream item ID given the previous one. If the current
64  * milliseconds Unix time is greater than the previous one, just use this
65  * as time part and start with sequence part of zero. Otherwise we use the
66  * previous time (and never go backward) and increment the sequence. */
67 void streamNextID(streamID *last_id, streamID *new_id) {
68  uint64_t ms = mstime();
69  if (ms > last_id->ms) {
70  new_id->ms = ms;
71  new_id->seq = 0;
72  } else {
73  new_id->ms = last_id->ms;
74  new_id->seq = last_id->seq+1;
75  }
76 }

78 /* This is just a wrapper for lpAppend() to directly use a 64 bit integer
79  * instead of a string. */
80 unsigned char *lpAppendInteger(unsigned char *lp, int64_t value) {
81  char buf[LONG_STR_SIZE];
82  int slen = ll2string(buf,sizeof(buf),value);
83  return lpAppend(lp,(unsigned char*)buf,slen);

86 /* This is just a wrapper for lpReplace() to directly use a 64 bit integer
87  * instead of a string to replace the current element. The function returns
88  * the new listpack as return value, and also updates the current cursor
89  * by updating '*pos'. */
90 unsigned char *lpReplaceInteger(unsigned char *lp, unsigned char **pos, int64_t value) {
91  char buf[LONG_STR_SIZE];

93  return lpInsert(lp, (unsigned char*)buf, slen, *pos, LP_REPLACE, pos);

96 /* This is a wrapper function for lpGet() to directly get an integer value
97  * from the listpack (that may store numbers as a string), converting
98  * the string if needed. */
99 int64_t lpGetInteger(unsigned char *ele) {
100  int64_t v;
101  unsigned char *e = lpGet(ele,&v,NULL);
102  if (e == NULL) return v;
103  /* The following code path should never be used for how listpacks work:
104  * they should always be able to store an int64_t value in integer
105  * encoded form. However the implementation may change. */
106  long long ll;
107  int retval = string2ll((char*)e,v,&ll);
108  serverAssert(retval != 0);
109  v = ll;
110  return v;
111 }

126 /* Convert the specified stream entry ID as a 128 bit big endian number, so
127  * that the IDs can be sorted lexicographically. */
128 void streamEncodeID(void *buf, streamID *id) {
129  uint64_t e[2];
130  e[0] = htonu64(id->ms);
131  e[1] = htonu64(id->seq);
132  memcpy(buf,e,sizeof(e));
133 }
134 
135 /* This is the reverse of streamEncodeID(): the decoded ID will be stored
136  * in the 'id' structure passed by reference. The buffer 'buf' must point
137  * to a 128 bit big-endian encoded ID. */
138 void streamDecodeID(void *buf, streamID *id) {
139  uint64_t e[2];
140  memcpy(e,buf,sizeof(e));
141  id->ms = ntohu64(e[0]);
142  id->seq = ntohu64(e[1]);
143 }

145 /* Adds a new item into the stream 's' having the specified number of
146  * field-value pairs as specified in 'numfields' and stored into 'argv'.
147  * Returns the new entry ID populating the 'added_id' structure.
148  *
149  * If 'use_id' is not NULL, the ID is not auto-generated by the function,
150  * but instead the passed ID is uesd to add the new entry. In this case
151  * adding the entry may fail as specified later in this comment.
152  *
153  * The function returns C_OK if the item was added, this is always true
154  * if the ID was generated by the function. However the function may return
155  * C_ERR if an ID was given via 'use_id', but adding it failed since the
156  * current top ID is greater or equal. */
157 int streamAppendItem(stream *s, robj **argv, int numfields, streamID *added_id, streamID *use_id) {
158  /* If an ID was given, check that it's greater than the last entry ID
159  * or return an error. */
160  if (use_id && (use_id->ms < s->last_id.ms ||
161  (use_id->ms == s->last_id.ms &&
162  use_id->seq <= s->last_id.seq))) return C_ERR;
163 
164  /* Add the new entry. */
165  raxIterator ri;
166  raxStart(&ri,s->rax);
167  raxSeek(&ri,"$",NULL,0);
168 
169  size_t lp_bytes = 0; /* Total bytes in the tail listpack. */
170  unsigned char *lp = NULL; /* Tail listpack pointer. */
171 
172  /* Get a reference to the tail node listpack. */
173  if (raxNext(&ri)) {
174  lp = ri.data;
175  lp_bytes = lpBytes(lp);
176  }
177  raxStop(&ri);
178 
179  /* Generate the new entry ID. */
180  streamID id;
181  if (use_id)
182  id = *use_id;
183  else
184  streamNextID(&s->last_id,&id);
185 
186  /* We have to add the key into the radix tree in lexicographic order,
187  * to do so we consider the ID as a single 128 bit number written in
188  * big endian, so that the most significant bytes are the first ones. */
189  uint64_t rax_key[2]; /* Key in the radix tree containing the listpack.*/
190  streamID master_id; /* ID of the master entry in the listpack. */
191 
192  /* Create a new listpack and radix tree node if needed. Note that when
193  * a new listpack is created, we populate it with a "master entry". This
194  * is just a set of fields that is taken as refernce in order to compress
195  * the stream entries that we'll add inside the listpack.
196  *
197  * Note that while we use the first added entry fields to create
198  * the master entry, the first added entry is NOT represented in the master
199  * entry, which is a stand alone object. But of course, the first entry
200  * will compress well because it's used as reference.
201  *
202  * The master entry is composed like in the following example:
203  *
204  * +-------+---------+------------+---------+--/--+---------+---------+-+
205  * | count | deleted | num-fields | field_1 | field_2 | ... | field_N |0|
206  * +-------+---------+------------+---------+--/--+---------+---------+-+
207  *
208  * count and deleted just represent respectively the total number of
209  * entires inside the listpack that are valid, and marked as deleted
210  * (delted flag in the entry flags set). So the total number of items
211  * actually inside the listpack (both deleted and not) is count+deleted.
212  *
213  * The real entries will be encoded with an ID that is just the
214  * millisecond and sequence difference compared to the key stored at
215  * the radix tree node containing the listpack (delta encoding), and
216  * if the fields of the entry are the same as the master enty fields, the
217  * entry flags will specify this fact and the entry fields and number
218  * of fields will be omitted (see later in the code of this function).
219  *
220  * The "0" entry at the end is the same as the 'lp-count' entry in the
221  * regular stream entries (see below), and marks the fact that there are
222  * no more entires, when we scan the stream from right to left. */
223 
224  int flags = STREAM_ITEM_FLAG_NONE;
225  if (lp == NULL || lp_bytes > STREAM_BYTES_PER_LISTPACK) {
226  master_id = id;
227  streamEncodeID(rax_key,&id);
228  /* Create the listpack having the master entry ID and fields. */
229  lp = lpNew();
230  lp = lpAppendInteger(lp,1); /* One item, the one we are adding. */
231  lp = lpAppendInteger(lp,0); /* Zero deleted so far. */
232  lp = lpAppendInteger(lp,numfields);
233  for (int i = 0; i < numfields; i++) {
234  sds field = argv[i*2]->ptr;
235  lp = lpAppend(lp,(unsigned char*)field,sdslen(field));
236  }
237  lp = lpAppendInteger(lp,0); /* Master entry zero terminator. */
238  raxInsert(s->rax,(unsigned char*)&rax_key,sizeof(rax_key),lp,NULL);
239  /* The first entry we insert, has obviously the same fields of the
240  * master entry. */
241  flags |= STREAM_ITEM_FLAG_SAMEFIELDS;
242  } else {
243  serverAssert(ri.key_len == sizeof(rax_key));
244  memcpy(rax_key,ri.key,sizeof(rax_key));
245 
246  /* Read the master ID from the radix tree key. */
247  streamDecodeID(rax_key,&master_id);
248  unsigned char *lp_ele = lpFirst(lp);
249 
250  /* Update count and skip the deleted fields. */
251  int64_t count = lpGetInteger(lp_ele);
252  lp = lpReplaceInteger(lp,&lp_ele,count+1);
253  lp_ele = lpNext(lp,lp_ele); /* seek delted. */
254  lp_ele = lpNext(lp,lp_ele); /* seek master entry num fields. */
255 
256  /* Check if the entry we are adding, have the same fields
257  * as the master entry. */
258  int master_fields_count = lpGetInteger(lp_ele);
259  lp_ele = lpNext(lp,lp_ele);
260  if (numfields == master_fields_count) {
261  int i;
262  for (i = 0; i < master_fields_count; i++) {
263  sds field = argv[i*2]->ptr;
264  int64_t e_len;
265  unsigned char buf[LP_INTBUF_SIZE];
266  unsigned char *e = lpGet(lp_ele,&e_len,buf);
267  /* Stop if there is a mismatch. */
268  if (sdslen(field) != (size_t)e_len ||
269  memcmp(e,field,e_len) != 0) break;
270  lp_ele = lpNext(lp,lp_ele);
271  }
272  /* All fields are the same! We can compress the field names
273  * setting a single bit in the flags. */
274  if (i == master_fields_count) flags |= STREAM_ITEM_FLAG_SAMEFIELDS;
275  }
276  }
277 
278  /* Populate the listpack with the new entry. We use the following
279  * encoding:
280  *
281  * +-----+--------+----------+-------+-------+-/-+-------+-------+--------+
282  * |flags|entry-id|num-fields|field-1|value-1|...|field-N|value-N|lp-count|
283  * +-----+--------+----------+-------+-------+-/-+-------+-------+--------+
284  *
285  * However if the SAMEFIELD flag is set, we have just to populate
286  * the entry with the values, so it becomes:
287  *
288  * +-----+--------+-------+-/-+-------+--------+
289  * |flags|entry-id|value-1|...|value-N|lp-count|
290  * +-----+--------+-------+-/-+-------+--------+
291  *
292  * The entry-id field is actually two separated fields: the ms
293  * and seq difference compared to the master entry.
294  *
295  * The lp-count field is a number that states the number of listpack pieces
296  * that compose the entry, so that it's possible to travel the entry
297  * in reverse order: we can just start from the end of the listpack, read
298  * the entry, and jump back N times to seek the "flags" field to read
299  * the stream full entry. */
300  lp = lpAppendInteger(lp,flags);
301  lp = lpAppendInteger(lp,id.ms - master_id.ms);
302  lp = lpAppendInteger(lp,id.seq - master_id.seq);
303  if (!(flags & STREAM_ITEM_FLAG_SAMEFIELDS))
304  lp = lpAppendInteger(lp,numfields);
305  for (int i = 0; i < numfields; i++) {
306  sds field = argv[i*2]->ptr, value = argv[i*2+1]->ptr;
307  if (!(flags & STREAM_ITEM_FLAG_SAMEFIELDS))
308  lp = lpAppend(lp,(unsigned char*)field,sdslen(field));
309  lp = lpAppend(lp,(unsigned char*)value,sdslen(value));
310  }
311  /* Compute and store the lp-count field. */
312  int lp_count = numfields;
313  lp_count += 3; /* Add the 3 fixed fields flags + ms-diff + seq-diff. */
314  if (!(flags & STREAM_ITEM_FLAG_SAMEFIELDS)) {
315  /* If the item is not compressed, it also has the fields other than
316  * the values, and an additional num-fileds field. */
317  lp_count += numfields+1;
318  }
319  lp = lpAppendInteger(lp,lp_count);
320 
321  /* Insert back into the tree in order to update the listpack pointer. */
322  raxInsert(s->rax,(unsigned char*)&rax_key,sizeof(rax_key),lp,NULL);
323  s->length++;
324  s->last_id = id;
325  if (added_id) *added_id = id;
326  return C_OK;
327 }

329 /* Trim the stream 's' to have no more than maxlen elements, and return the
330  * number of elements removed from the stream. The 'approx' option, if non-zero,
331  * specifies that the trimming must be performed in a approximated way in
332  * order to maximize performances. This means that the stream may contain
333  * more elements than 'maxlen', and elements are only removed if we can remove
334  * a *whole* node of the radix tree. The elements are removed from the head
335  * of the stream (older elements).
336  *
337  * The function may return zero if:
338  *
339  * 1) The stream is already shorter or equal to the specified max length.
340  * 2) The 'approx' option is true and the head node had not enough elements
341  * to be deleted, leaving the stream with a number of elements >= maxlen.
342  */
343 int64_t streamTrimByLength(stream *s, size_t maxlen, int approx) {
344  if (s->length <= maxlen) return 0;
345 
346  raxIterator ri;
347  raxStart(&ri,s->rax);
348  raxSeek(&ri,"^",NULL,0);
349 
350  int64_t deleted = 0;
351  while(s->length > maxlen && raxNext(&ri)) {
352  unsigned char *lp = ri.data, *p = lpFirst(lp);
353  int64_t entries = lpGetInteger(p);
354 
355  /* Check if we can remove the whole node, and still have at
356  * least maxlen elements. */
357  if (s->length - entries >= maxlen) {
358  lpFree(lp);
359  raxRemove(s->rax,ri.key,ri.key_len,NULL);
360  raxSeek(&ri,">=",ri.key,ri.key_len);
361  s->length -= entries;
362  deleted += entries;
363  continue;
364  }
365 
366  /* If we cannot remove a whole element, and approx is true,
367  * stop here. */
368  if (approx) break;
369 
370  /* Otherwise, we have to mark single entries inside the listpack
371  * as deleted. We start by updating the entries/deleted counters. */
372  int64_t to_delete = s->length - maxlen;
373  serverAssert(to_delete < entries);
374  lp = lpReplaceInteger(lp,&p,entries-to_delete);
375  p = lpNext(lp,p); /* Seek deleted field. */
376  int64_t marked_deleted = lpGetInteger(p);
377  lp = lpReplaceInteger(lp,&p,marked_deleted+to_delete);
378  p = lpNext(lp,p); /* Seek num-of-fields in the master entry. */
379 
380  /* Skip all the master fields. */
381  int64_t master_fields_count = lpGetInteger(p);
382  p = lpNext(lp,p); /* Seek the first field. */
383  for (int64_t j = 0; j < master_fields_count; j++)
384  p = lpNext(lp,p); /* Skip all master fields. */
385  p = lpNext(lp,p); /* Skip the zero master entry terminator. */
386 
387  /* 'p' is now pointing to the first entry inside the listpack.
388  * We have to run entry after entry, marking entries as deleted
389  * if they are already not deleted. */
390  while(p) {
391  int flags = lpGetInteger(p);
392  int to_skip;
393 
394  /* Mark the entry as deleted. */
395  if (!(flags & STREAM_ITEM_FLAG_DELETED)) {
396  flags |= STREAM_ITEM_FLAG_DELETED;
397  lp = lpReplaceInteger(lp,&p,flags);
398  deleted++;
399  s->length--;
400  if (s->length <= maxlen) break; /* Enough entries deleted. */
401  }
402 
403  p = lpNext(lp,p); /* Skip ID ms delta. */
404  p = lpNext(lp,p); /* Skip ID seq delta. */
405  p = lpNext(lp,p); /* Seek num-fields or values (if compressed). */
406  if (flags & STREAM_ITEM_FLAG_SAMEFIELDS) {
407  to_skip = master_fields_count;
408  } else {
409  to_skip = lpGetInteger(p);
410  to_skip = 1+(to_skip*2);
411  }
412 
413  while(to_skip--) p = lpNext(lp,p); /* Skip the whole entry. */
414  p = lpNext(lp,p); /* Skip the final lp-count field. */
415  }
416 
417  /* Here we should perform garbage collection in case at this point
418  * there are too many entries deleted inside the listpack. */
419  entries -= to_delete;
420  marked_deleted += to_delete;
421  if (entries + marked_deleted > 10 && marked_deleted > entries/2) {
422  /* TODO: perform a garbage collection. */
423  }
424 
425  break; /* If we are here, there was enough to delete in the current
426  node, so no need to go to the next node. */
427  }
428 
429  raxStop(&ri);
430  return deleted;
431 }

433 /* Initialize the stream iterator, so that we can call iterating functions
434  * to get the next items. This requires a corresponding streamIteratorStop()
435  * at the end. The 'rev' parameter controls the direction. If it's zero the
436  * iteration is from the start to the end element (inclusive), otherwise
437  * if rev is non-zero, the iteration is reversed.
438  *
439  * Once the iterator is initalized, we iterate like this:
440  *
441  * streamIterator myiterator;
442  * streamIteratorStart(&myiterator,...);
443  * int64_t numfields;
444  * while(streamIteratorGetID(&myitereator,&ID,&numfields)) {
445  * while(numfields--) {
446  * unsigned char *key, *value;
447  * size_t key_len, value_len;
448  * streamIteratorGetField(&myiterator,&key,&value,&key_len,&value_len);
449  *
450  * ... do what you want with key and value ...
451  * }
452  * }
453  * streamIteratorStop(&myiterator); */
454 void streamIteratorStart(streamIterator *si, stream *s, streamID *start, streamID *end, int rev) {
455  /* Intialize the iterator and translates the iteration start/stop
456  * elements into a 128 big big-endian number. */
457  if (start) {
458  streamEncodeID(si->start_key,start);
459  } else {
460  si->start_key[0] = 0;
461  si->start_key[0] = 0;
462  }
463 
464  if (end) {
465  streamEncodeID(si->end_key,end);
466  } else {
467  si->end_key[0] = UINT64_MAX;
468  si->end_key[0] = UINT64_MAX;
469  }
470 
471  /* Seek the correct node in the radix tree. */
472  raxStart(&si->ri,s->rax);
473  if (!rev) {
474  if (start && (start->ms || start->seq)) {
475  raxSeek(&si->ri,"<=",(unsigned char*)si->start_key,
476  sizeof(si->start_key));
477  if (raxEOF(&si->ri)) raxSeek(&si->ri,"^",NULL,0);
478  } else {
479  raxSeek(&si->ri,"^",NULL,0);
480  }
481  } else {
482  if (end && (end->ms || end->seq)) {
483  raxSeek(&si->ri,"<=",(unsigned char*)si->end_key,
484  sizeof(si->end_key));
485  if (raxEOF(&si->ri)) raxSeek(&si->ri,"$",NULL,0);
486  } else {
487  raxSeek(&si->ri,"$",NULL,0);
488  }
489  }
490  si->lp = NULL; /* There is no current listpack right now. */
491  si->lp_ele = NULL; /* Current listpack cursor. */
492  si->rev = rev; /* Direction, if non-zero reversed, from end to start. */
493 }
494 
495 /* Return 1 and store the current item ID at 'id' if there are still
496  * elements within the iteration range, otherwise return 0 in order to
497  * signal the iteration terminated. */
498 int streamIteratorGetID(streamIterator *si, streamID *id, int64_t *numfields) {
499  while(1) { /* Will stop when element > stop_key or end of radix tree. */
500  /* If the current listpack is set to NULL, this is the start of the
501  * iteration or the previous listpack was completely iterated.
502  * Go to the next node. */
503  if (si->lp == NULL || si->lp_ele == NULL) {
504  if (!si->rev && !raxNext(&si->ri)) return 0;
505  else if (si->rev && !raxPrev(&si->ri)) return 0;
506  serverAssert(si->ri.key_len == sizeof(streamID));
507  /* Get the master ID. */
508  streamDecodeID(si->ri.key,&si->master_id);
509  /* Get the master fields count. */
510  si->lp = si->ri.data;
511  si->lp_ele = lpFirst(si->lp); /* Seek items count */
512  si->lp_ele = lpNext(si->lp,si->lp_ele); /* Seek deleted count. */
513  si->lp_ele = lpNext(si->lp,si->lp_ele); /* Seek num fields. */
514  si->master_fields_count = lpGetInteger(si->lp_ele);
515  si->lp_ele = lpNext(si->lp,si->lp_ele); /* Seek first field. */
516  si->master_fields_start = si->lp_ele;
517  /* Skip master fileds to seek the first entry. */
518  for (uint64_t i = 0; i < si->master_fields_count; i++)
519  si->lp_ele = lpNext(si->lp,si->lp_ele);
520  /* We are now pointing the zero term of the master entry. If
521  * we are iterating in reverse order, we need to seek the
522  * end of the listpack. */
523  if (si->rev) si->lp_ele = lpLast(si->lp);
524  } else if (si->rev) {
525  /* If we are itereating in the reverse order, and this is not
526  * the first entry emitted for this listpack, then we already
527  * emitted the current entry, and have to go back to the previous
528  * one. */
529  int lp_count = lpGetInteger(si->lp_ele);
530  while(lp_count--) si->lp_ele = lpPrev(si->lp,si->lp_ele);
531  /* Seek lp-count of prev entry. */
532  si->lp_ele = lpPrev(si->lp,si->lp_ele);
533  }
534 
535  /* For every radix tree node, iterate the corresponding listpack,
536  * returning elements when they are within range. */
537  while(1) {
538  if (!si->rev) {
539  /* If we are going forward, skip the previous entry
540  * lp-count field (or in case of the master entry, the zero
541  * term field) */
542  si->lp_ele = lpNext(si->lp,si->lp_ele);
543  if (si->lp_ele == NULL) break;
544  } else {
545  /* If we are going backward, read the number of elements this
546  * entry is composed of, and jump backward N times to seek
547  * its start. */

562  * ID and this entry ID. */

564  id->ms += lpGetInteger(si->lp_ele);
565  si->lp_ele = lpNext(si->lp,si->lp_ele);
566  id->seq += lpGetInteger(si->lp_ele);
567  si->lp_ele = lpNext(si->lp,si->lp_ele);

572  * flags. */

581  * deleted, emit it. */

584  !(flags & STREAM_ITEM_FLAG_DELETED))

590  si->master_fields_ptr = si->master_fields_start;
591  return 1; /* Valid item returned. */
592  }
593  } else {
594  if (memcmp(buf,si->end_key,sizeof(streamID)) <= 0 &&
595  !(flags & STREAM_ITEM_FLAG_DELETED))
596  {

598  return 0; /* We are already out of range. */
599  si->entry_flags = flags;
600  if (flags & STREAM_ITEM_FLAG_SAMEFIELDS)
601  si->master_fields_ptr = si->master_fields_start;
602  return 1; /* Valid item returned. */
603  }

607  * forward, or seek the previous entry if we are going
608  * backward. */
609  if (!si->rev) {

611  *numfields : *numfields*2;
612  for (int64_t i = 0; i < to_discard; i++)

614  } else {

616  while(prev_times--) si->lp_ele = lpPrev(si->lp,si->lp_ele);
617  }

649 /* Send the specified range to the client 'c'. The range the client will
650  * receive is between start and end inclusive, if 'count' is non zero, no more
651  * than 'count' elemnets are sent. The 'end' pointer can be NULL to mean that
652  * we want all the elements from 'start' till the end of the stream. If 'rev'
653  * is non zero, elements are produced in reversed order from end to start. */
654 size_t streamReplyWithRange(client *c, stream *s, streamID *start, streamID *end, size_t count, int rev) {
655  void *arraylen_ptr = addDeferredMultiBulkLength(c);
656  size_t arraylen = 0;

659  streamID id;

662  while(streamIteratorGetID(&si,&id,&numfields)) {
663  /* Emit a two elements array for each item. The first is
664  * the ID, the second is an array of field-value pairs. */
665  sds replyid = sdscatfmt(sdsempty(),"+%U-%U\r\n",id.ms,id.seq);
666  addReplyMultiBulkLen(c,2);
667  addReplySds(c,replyid);
668  addReplyMultiBulkLen(c,numfields*2);
669 
670  /* Emit the field-value pairs. */
671  while(numfields--) {
672  unsigned char *key, *value;

674  streamIteratorGetField(&si,&key,&value,&key_len,&value_len);
675  addReplyBulkCBuffer(c,key,key_len);
676  addReplyBulkCBuffer(c,value,value_len);
677  }
678  arraylen++;
679  if (count && count == arraylen) break;
680  }
681  streamIteratorStop(&si);
682  setDeferredMultiBulkLength(c,arraylen_ptr,arraylen);
683  return arraylen;

706 /* Helper function to convert a string to an unsigned long long value.

724 /* Parse a stream ID in the format given by clients to Redis, that is
725  * <ms>.<seq>, and converts it into a streamID structure. If
726  * the specified ID is invalid C_ERR is returned and an error is reported
727  * to the client, otherwise C_OK is returned. The ID may be in incomplete
728  * form, just stating the milliseconds time part of the stream. In such a case
729  * the missing part is set according to the value of 'missing_seq' parameter.
730  * The IDs "-" and "+" specify respectively the minimum and maximum IDs
731  * that can be represented.
732  *
733  * If 'c' is set to NULL, no reply is sent to the client. */
734 int streamParseIDOrReply(client *c, robj *o, streamID *id, uint64_t missing_seq) {
735  char buf[128];
736  if (sdslen(o->ptr) > sizeof(buf)-1) goto invalid;
737  memcpy(buf,o->ptr,sdslen(o->ptr)+1);
738 
739  /* Handle the "-" and "+" special cases. */
740  if (buf[0] == '-' && buf[1] == '\0') {
741  id->ms = 0;
742  id->seq = 0;
743  return C_OK;
744  } else if (buf[0] == '+' && buf[1] == '\0') {
745  id->ms = UINT64_MAX;
746  id->seq = UINT64_MAX;
747  return C_OK;
748  }
749 
750  /* Parse <ms>.<seq> form. */
751  char *dot = strchr(buf,'-');
752  if (dot) *dot = '\0';
753  unsigned long long ms, seq;
754  if (string2ull(buf,&ms) == 0) goto invalid;
755  if (dot && string2ull(dot+1,&seq) == 0) goto invalid;
756  if (!dot) seq = missing_seq;
757  id->ms = ms;
758  id->seq = seq;
759  return C_OK;
760 
761 invalid:
762  if (c) addReplyError(c,"Invalid stream ID specified as stream "
763  "command argument");
764  return C_ERR;
765 }

920  * [RETRY <milliseconds> <ttl>] STREAMS key_1 key_2 ... key_N

974  * served with just the messages that will arrive into the stream
