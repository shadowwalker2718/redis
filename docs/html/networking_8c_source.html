155 /* This function is called every time we are going to transmit new data
156  * to the client. The behavior is the following:
157  *
158  * If the client should receive new data (normal clients will) the function
159  * returns C_OK, and make sure to install the write handler in our event
160  * loop so that when the socket is writable new data gets written.
161  *
162  * If the client should not receive new data, because it is a fake client
163  * (used to load AOF in memory), a master or because the setup of the write
164  * handler failed, the function returns C_ERR.
165  *
166  * The function may return C_OK without actually installing the write
167  * event handler in the following cases:
168  *
169  * 1) The event handler should already be installed since the output buffer
170  * already contains something.
171  * 2) The client is a slave but not yet online, so we want to just accumulate
172  * writes in the buffer but not actually sending them yet.
173  *
174  * Typically gets called every time a reply is built, before adding more
175  * data to the clients output buffers. If the function returns C_ERR no
176  * data should be appended to the output buffers. */
177 int prepareClientToWrite(client *c) {
178  /* If it's the Lua client we always return ok without installing any
179  * handler since there is no socket at all. */
180  if (c->flags & (CLIENT_LUA|CLIENT_MODULE)) return C_OK;
181 
182  /* CLIENT REPLY OFF / SKIP handling: don't send replies. */
183  if (c->flags & (CLIENT_REPLY_OFF|CLIENT_REPLY_SKIP)) return C_ERR;
184 
185  /* Masters don't receive replies, unless CLIENT_MASTER_FORCE_REPLY flag
186  * is set. */
187  if ((c->flags & CLIENT_MASTER) &&
188  !(c->flags & CLIENT_MASTER_FORCE_REPLY)) return C_ERR;
189 
190  if (c->fd <= 0) return C_ERR; /* Fake client for AOF loading. */
191 
192  /* Schedule the client to write the output buffers to the socket only
193  * if not already done (there were no pending writes already and the client
194  * was yet not flagged), and, for slaves, if the slave can actually
195  * receive writes at this stage. */
196  if (!clientHasPendingReplies(c) &&
197  !(c->flags & CLIENT_PENDING_WRITE) &&
198  (c->replstate == REPL_STATE_NONE ||
199  (c->replstate == SLAVE_STATE_ONLINE && !c->repl_put_online_on_ack)))
200  {
201  /* Here instead of installing the write handler, we just flag the
202  * client and put it into a list of clients that have something
203  * to write to the socket. This way before re-entering the event
204  * loop, we can try to directly write to the client sockets avoiding
205  * a system call. We'll only really install the write handler if
206  * we'll not be able to write the whole reply at once. */
207  c->flags |= CLIENT_PENDING_WRITE;
208  listAddNodeHead(server.clients_pending_write,c);
209  }
210 
211  /* Authorize the caller to queue in the output buffer of this client. */
212  return C_OK;
213 }

219 int _addReplyToBuffer(client *c, const char *s, size_t len) {
220  size_t available = sizeof(c->buf)-c->bufpos;
221 
222  if (c->flags & CLIENT_CLOSE_AFTER_REPLY) return C_OK;
223 
224  /* If there already are entries in the reply list, we cannot
225  * add anything more to the static buffer. */
226  if (listLength(c->reply) > 0) return C_ERR;
227 
228  /* Check that the buffer has enough space available for this string. */
229  if (len > available) return C_ERR;
230 
231  memcpy(c->buf+c->bufpos,s,len);
232  c->bufpos+=len;
233  return C_OK;
234 }

236 void _addReplyObjectToList(client *c, robj *o) {
237  if (c->flags & CLIENT_CLOSE_AFTER_REPLY) return;
238 
239  if (listLength(c->reply) == 0) {
240  sds s = sdsdup(o->ptr);
241  listAddNodeTail(c->reply,s);
242  c->reply_bytes += sdslen(s);
243  } else {
244  listNode *ln = listLast(c->reply);
245  sds tail = listNodeValue(ln);
246 
247  /* Append to this object when possible. If tail == NULL it was
248  * set via addDeferredMultiBulkLength(). */
249  if (tail && sdslen(tail)+sdslen(o->ptr) <= PROTO_REPLY_CHUNK_BYTES) {
250  tail = sdscatsds(tail,o->ptr);
251  listNodeValue(ln) = tail;
252  c->reply_bytes += sdslen(o->ptr);
253  } else {
254  sds s = sdsdup(o->ptr);
255  listAddNodeTail(c->reply,s);
256  c->reply_bytes += sdslen(s);
257  }
258  }
259  asyncCloseClientOnOutputBufferLimitReached(c);
260 }

262 /* This method takes responsibility over the sds. When it is no longer
263  * needed it will be free'd, otherwise it ends up in a robj. */
264 void _addReplySdsToList(client *c, sds s) {
265  if (c->flags & CLIENT_CLOSE_AFTER_REPLY) {
266  sdsfree(s);
267  return;
268  }
269 
270  if (listLength(c->reply) == 0) {
271  listAddNodeTail(c->reply,s);
272  c->reply_bytes += sdslen(s);
273  } else {
274  listNode *ln = listLast(c->reply);
275  sds tail = listNodeValue(ln);
276 
277  /* Append to this object when possible. If tail == NULL it was
278  * set via addDeferredMultiBulkLength(). */
279  if (tail && sdslen(tail)+sdslen(s) <= PROTO_REPLY_CHUNK_BYTES) {
280  tail = sdscatsds(tail,s);
281  listNodeValue(ln) = tail;
282  c->reply_bytes += sdslen(s);
283  sdsfree(s);
284  } else {
285  listAddNodeTail(c->reply,s);
286  c->reply_bytes += sdslen(s);
287  }
288  }
289  asyncCloseClientOnOutputBufferLimitReached(c);
290 }

323 void addReply(client *c, robj *obj) {
324  if (prepareClientToWrite(c) != C_OK) return;
325 
326  /* This is an important place where we can avoid copy-on-write
327  * when there is a saving child running, avoiding touching the
328  * refcount field of the object if it's not needed.
329  *
330  * If the encoding is RAW and there is room in the static buffer
331  * we'll be able to send the object to the client without
332  * messing with its page. */
333  if (sdsEncodedObject(obj)) {
334  if (_addReplyToBuffer(c,obj->ptr,sdslen(obj->ptr)) != C_OK)
335  _addReplyObjectToList(c,obj);
336  } else if (obj->encoding == OBJ_ENCODING_INT) {
337  /* Optimization: if there is room in the static buffer for 32 bytes
338  * (more than the max chars a 64 bit integer can take as string) we
339  * avoid decoding the object and go for the lower level approach. */
340  if (listLength(c->reply) == 0 && (sizeof(c->buf) - c->bufpos) >= 32) {
341  char buf[32];
342  int len;
343 
344  len = ll2string(buf,sizeof(buf),(long)obj->ptr);
345  if (_addReplyToBuffer(c,buf,len) == C_OK)
346  return;
347  /* else... continue with the normal code path, but should never
348  * happen actually since we verified there is room. */
349  }
350  obj = getDecodedObject(obj);
351  if (_addReplyToBuffer(c,obj->ptr,sdslen(obj->ptr)) != C_OK)
352  _addReplyObjectToList(c,obj);
353  decrRefCount(obj);
354  } else {
355  serverPanic("Wrong obj->encoding in addReply()");
356  }
357 }

443 /* Populate the length object and try gluing it to the next chunk. */
444 void setDeferredMultiBulkLength(client *c, void *node, long length) {
445  listNode *ln = (listNode*)node;
446  sds len, next;
447 
448  /* Abort when *node is NULL: when the client should not accept writes
449  * we return NULL in addDeferredMultiBulkLength() */
450  if (node == NULL) return;
451 
452  len = sdscatprintf(sdsnewlen("*",1),"%ld\r\n",length);
453  listNodeValue(ln) = len;
454  c->reply_bytes += sdslen(len);
455  if (ln->next != NULL) {
456  next = listNodeValue(ln->next);
457 
458  /* Only glue when the next node is non-NULL (an sds in this case) */
459  if (next != NULL) {
460  len = sdscatsds(len,next);
461  listDelNode(c->reply,ln->next);
462  listNodeValue(ln) = len;
463  /* No need to update c->reply_bytes: we are just moving the same
464  * amount of bytes from one node to another. */
465  }
466  }
467  asyncCloseClientOnOutputBufferLimitReached(c);
468 }

534 /* Create the length prefix of a bulk reply, example: $2234 */
535 void addReplyBulkLen(client *c, robj *obj) {
536  size_t len;
537 
538  if (sdsEncodedObject(obj)) {
539  len = sdslen(obj->ptr);
540  } else {
541  long n = (long)obj->ptr;
542 
543  /* Compute how many bytes will take this integer as a radix 10 string */
544  len = 1;
545  if (n < 0) {
546  len++;
547  n = -n;
548  }
549  while((n = n/10) != 0) {
550  len++;
551  }
552  }
553 
554  if (len < OBJ_SHARED_BULKHDR_LEN)
555  addReply(c,shared.bulkhdr[len]);
556  else
557  addReplyLongLongWithPrefix(c,len,'$');
558 }

808 void freeClient(client *c) {
809  listNode *ln;
810 
811  /* If it is our master that's beging disconnected we should make sure
812  * to cache the state to try a partial resynchronization later.
813  *
814  * Note that before doing this we make sure that the client is not in
815  * some unexpected state, by checking its flags. */
816  if (server.master && c->flags & CLIENT_MASTER) {
817  serverLog(LL_WARNING,"Connection with master lost.");
818  if (!(c->flags & (CLIENT_CLOSE_AFTER_REPLY|
819  CLIENT_CLOSE_ASAP|
820  CLIENT_BLOCKED|
821  CLIENT_UNBLOCKED)))
822  {
823  replicationCacheMaster(c);
824  return;
825  }
826  }
827 
828  /* Log link disconnection with slave */
829  if ((c->flags & CLIENT_SLAVE) && !(c->flags & CLIENT_MONITOR)) {
830  serverLog(LL_WARNING,"Connection with slave %s lost.",
831  replicationGetSlaveName(c));
832  }
833 
834  /* Free the query buffer */
835  sdsfree(c->querybuf);
836  sdsfree(c->pending_querybuf);
837  c->querybuf = NULL;
838 
839  /* Deallocate structures used to block on blocking ops. */
840  if (c->flags & CLIENT_BLOCKED) unblockClient(c);
841  dictRelease(c->bpop.keys);
842 
843  /* UNWATCH all the keys */
844  unwatchAllKeys(c);
845  listRelease(c->watched_keys);
846 
847  /* Unsubscribe from all the pubsub channels */
848  pubsubUnsubscribeAllChannels(c,0);
849  pubsubUnsubscribeAllPatterns(c,0);
850  dictRelease(c->pubsub_channels);
851  listRelease(c->pubsub_patterns);
852 
853  /* Free data structures. */
854  listRelease(c->reply);
855  freeClientArgv(c);
856 
857  /* Unlink the client: this will close the socket, remove the I/O
858  * handlers, and remove references of the client from different
859  * places where active clients may be referenced. */
860  unlinkClient(c);
861 
862  /* Master/slave cleanup Case 1:
863  * we lost the connection with a slave. */
864  if (c->flags & CLIENT_SLAVE) {
865  if (c->replstate == SLAVE_STATE_SEND_BULK) {
866  if (c->repldbfd != -1) close(c->repldbfd);
867  if (c->replpreamble) sdsfree(c->replpreamble);
868  }
869  list *l = (c->flags & CLIENT_MONITOR) ? server.monitors : server.slaves;
870  ln = listSearchKey(l,c);
871  serverAssert(ln != NULL);
872  listDelNode(l,ln);
873  /* We need to remember the time when we started to have zero
874  * attached slaves, as after some time we'll free the replication
875  * backlog. */
876  if (c->flags & CLIENT_SLAVE && listLength(server.slaves) == 0)
877  server.repl_no_slaves_since = server.unixtime;
878  refreshGoodSlavesCount();
879  }
880 
881  /* Master/slave cleanup Case 2:
882  * we lost the connection with the master. */
883  if (c->flags & CLIENT_MASTER) replicationHandleMasterDisconnection();
884 
885  /* If this client was scheduled for async freeing we need to remove it
886  * from the queue. */
887  if (c->flags & CLIENT_CLOSE_ASAP) {
888  ln = listSearchKey(server.clients_to_close,c);
889  serverAssert(ln != NULL);
890  listDelNode(server.clients_to_close,ln);
891  }
892 
893  /* Release other dynamically allocated client structure fields,
894  * and finally release the client structure itself. */
895  if (c->name) decrRefCount(c->name);
896  zfree(c->argv);
897  freeClientMultiState(c);
898  sdsfree(c->peerid);
899  zfree(c);
900 }

923 /* Write data in output buffers to client. Return C_OK if the client
924  * is still valid after the call, C_ERR if it was freed. */
925 int writeToClient(int fd, client *c, int handler_installed) {
926  ssize_t nwritten = 0, totwritten = 0;
927  size_t objlen;
928  sds o;
929 
930  while(clientHasPendingReplies(c)) {
931  if (c->bufpos > 0) {
932  nwritten = write(fd,c->buf+c->sentlen,c->bufpos-c->sentlen);
933  if (nwritten <= 0) break;
934  c->sentlen += nwritten;
935  totwritten += nwritten;
936 
937  /* If the buffer was sent, set bufpos to zero to continue with
938  * the remainder of the reply. */
939  if ((int)c->sentlen == c->bufpos) {
940  c->bufpos = 0;
941  c->sentlen = 0;
942  }
943  } else {
944  o = listNodeValue(listFirst(c->reply));
945  objlen = sdslen(o);
946 
947  if (objlen == 0) {
948  listDelNode(c->reply,listFirst(c->reply));
949  continue;
950  }
951 
952  nwritten = write(fd, o + c->sentlen, objlen - c->sentlen);
953  if (nwritten <= 0) break;
954  c->sentlen += nwritten;
955  totwritten += nwritten;
956 
957  /* If we fully sent the object on head go to the next one */
958  if (c->sentlen == objlen) {
959  listDelNode(c->reply,listFirst(c->reply));
960  c->sentlen = 0;
961  c->reply_bytes -= objlen;
962  /* If there are no longer objects in the list, we expect
963  * the count of reply bytes to be exactly zero. */
964  if (listLength(c->reply) == 0)
965  serverAssert(c->reply_bytes == 0);
966  }
967  }
968  /* Note that we avoid to send more than NET_MAX_WRITES_PER_EVENT
969  * bytes, in a single threaded server it's a good idea to serve
970  * other clients as well, even if a very large request comes from
971  * super fast link that is always able to accept data (in real world
972  * scenario think about 'KEYS *' against the loopback interface).
973  *
974  * However if we are over the maxmemory limit we ignore that and
975  * just deliver as much data as it is possible to deliver. */
976  if (totwritten > NET_MAX_WRITES_PER_EVENT &&
977  (server.maxmemory == 0 ||
978  zmalloc_used_memory() < server.maxmemory)) break;
979  }
980  server.stat_net_output_bytes += totwritten;
981  if (nwritten == -1) {
982  if (errno == EAGAIN) {
983  nwritten = 0;
984  } else {
985  serverLog(LL_VERBOSE,
986  "Error writing to client: %s", strerror(errno));
987  freeClient(c);
988  return C_ERR;
989  }
990  }
991  if (totwritten > 0) {
992  /* For clients representing masters we don't count sending data
993  * as an interaction, since we always send REPLCONF ACK commands
994  * that take some time to just fill the socket output buffer.
995  * We just rely on data / pings received for timeout detection. */
996  if (!(c->flags & CLIENT_MASTER)) c->lastinteraction = server.unixtime;
997  }
998  if (!clientHasPendingReplies(c)) {
999  c->sentlen = 0;
1000  if (handler_installed) aeDeleteFileEvent(server.el,c->fd,AE_WRITABLE);
1001 
1002  /* Close connection after entire reply has been sent. */
1003  if (c->flags & CLIENT_CLOSE_AFTER_REPLY) {
1004  freeClient(c);
1005  return C_ERR;
1006  }
1007  }
1008  return C_OK;
1009 }

1074  * to be executed inside the client structure. Returns C_OK if the command
1075  * is ready to be executed, or C_ERR if there is still protocol to read to
1076  * have a well formed command. The function also returns C_ERR when there is
1077  * a protocol error: in such a case the client structure is setup to reply

1079 int processInlineBuffer(client *c) {

1083  size_t querylen;

1086  newline = strchr(c->querybuf,'\n');
1087

1093  }
1094  return C_ERR;
1095  }
1096

1098  if (newline && newline != c->querybuf && *(newline-1) == '\r')
1099  newline--;

1102  querylen = newline-(c->querybuf);

1108  setProtocolError("unbalanced quotes in inline request",c,0);

1114  * RDB file. */
1115  if (querylen == 0 && c->flags & CLIENT_SLAVE)
1116  c->repl_ack_time = server.unixtime;
1117

1119  sdsrange(c->querybuf,querylen+2,-1);

1124  c->argv = zmalloc(sizeof(robj*)*argc);
1125  }

1130  c->argv[c->argc] = createObject(OBJ_STRING,argv[j]);
1131  c->argc++;
1132  } else {
1133  sdsfree(argv[j]);
1134  }
1135  }
1136  zfree(argv);

1320 /* This function is called every time, in the client structure 'c', there is
1321  * more query buffer to process, because we read more data from the socket
1322  * or because a client was blocked and later reactivated, so there could be
1323  * pending query buffer, already representing a full command, to process. */
1324 void processInputBuffer(client *c) {
1325  server.current_client = c;
1326  /* Keep processing while there is something in the input buffer */
1327  while(sdslen(c->querybuf)) {
1328  /* Return if clients are paused. */
1329  if (!(c->flags & CLIENT_SLAVE) && clientsArePaused()) break;
1330 
1331  /* Immediately abort if the client is in the middle of something. */
1332  if (c->flags & CLIENT_BLOCKED) break;
1333 
1334  /* CLIENT_CLOSE_AFTER_REPLY closes the connection once the reply is
1335  * written to the client. Make sure to not let the reply grow after
1336  * this flag has been set (i.e. don't process more commands).
1337  *
1338  * The same applies for clients we want to terminate ASAP. */
1339  if (c->flags & (CLIENT_CLOSE_AFTER_REPLY|CLIENT_CLOSE_ASAP)) break;
1340 
1341  /* Determine request type when unknown. */
1342  if (!c->reqtype) {
1343  if (c->querybuf[0] == '*') {
1344  c->reqtype = PROTO_REQ_MULTIBULK;
1345  } else {
1346  c->reqtype = PROTO_REQ_INLINE;
1347  }
1348  }
1349 
1350  if (c->reqtype == PROTO_REQ_INLINE) {
1351  if (processInlineBuffer(c) != C_OK) break;
1352  } else if (c->reqtype == PROTO_REQ_MULTIBULK) {
1353  if (processMultibulkBuffer(c) != C_OK) break;
1354  } else {
1355  serverPanic("Unknown request type");
1356  }
1357 
1358  /* Multibulk processing could see a <= 0 length. */
1359  if (c->argc == 0) {
1360  resetClient(c);
1361  } else {
1362  /* Only reset the client when the command was executed. */
1363  if (processCommand(c) == C_OK) {
1364  if (c->flags & CLIENT_MASTER && !(c->flags & CLIENT_MULTI)) {
1365  /* Update the applied replication offset of our master. */
1366  c->reploff = c->read_reploff - sdslen(c->querybuf);
1367  }
1368 
1369  /* Don't reset the client structure for clients blocked in a
1370  * module blocking command, so that the reply callback will
1371  * still be able to access the client argv and argc field.
1372  * The client will be reset in unblockClientFromModule(). */
1373  if (!(c->flags & CLIENT_BLOCKED) || c->btype != BLOCKED_MODULE)
1374  resetClient(c);

1377  * result into a slave, that may be the active client, to be
1378  * freed. */

1381  }

1391

1393  /* If this is a multi bulk request, and we are processing a bulk reply
1394  * that is large enough, try to maximize the probability that the query
1395  * buffer contains exactly the SDS string representing the object, even
1396  * at the risk of requiring more read(2) calls. This way the function
1397  * processMultiBulkBuffer() can avoid copying buffers to create the
1398  * Redis Object representing the argument. */
1399  if (c->reqtype == PROTO_REQ_MULTIBULK && c->multibulklen && c->bulklen != -1
1400  && c->bulklen >= PROTO_MBULK_BIG_ARG)

1408  if (c->querybuf_peak < qblen) c->querybuf_peak = qblen;

1413  return;

1416  freeClient(c);

1419  } else if (nread == 0) {

1423  } else if (c->flags & CLIENT_MASTER) {

1425  * of the master. We'll use this buffer later in order to have a

1427  c->pending_querybuf = sdscatlen(c->pending_querybuf,
1428  c->querybuf+qblen,nread);

1438  bytes = sdscatrepr(bytes,c->querybuf,64);
1439  serverLog(LL_WARNING,"Closing client that reached max query buffer length: %s (qbuf initial bytes: %s)", ci, bytes);
1440  sdsfree(ci);
1441  sdsfree(bytes);
1442  freeClient(c);
1443  return;
1444  }

1447  * compute the difference between the applied offset before and after
1448  * processing the buffer, to understand how much of the replication stream
1449  * was actually applied to the master state: this quantity, and its
1450  * corresponding part of the replication stream, will be propagated to
1451  * the sub-slaves and to the replication backlog. */
1452  if (!(c->flags & CLIENT_MASTER)) {
1453  processInputBuffer(c);
1454  } else {
1455  size_t prev_offset = c->reploff;
1456  processInputBuffer(c);
1457  size_t applied = c->reploff - prev_offset;
1458  if (applied) {
1459  replicationFeedSlavesFromMasterStream(server.slaves,
1460  c->pending_querybuf, applied);
1461  sdsrange(c->pending_querybuf,applied,-1);
1462  }
1463  }

1904 /* The function checks if the client reached output buffer soft or hard
1905  * limit, and also update the state needed to check the soft limit as
1906  * a side effect.
1907  *
1908  * Return value: non-zero if the client reached the soft or the hard limit.
1909  * Otherwise zero is returned. */
1910 int checkClientOutputBufferLimits(client *c) {
1911  int soft = 0, hard = 0, class;
1912  unsigned long used_mem = getClientOutputBufferMemoryUsage(c);
1913 
1914  class = getClientType(c);
1915  /* For the purpose of output buffer limiting, masters are handled
1916  * like normal clients. */
1917  if (class == CLIENT_TYPE_MASTER) class = CLIENT_TYPE_NORMAL;
1918 
1919  if (server.client_obuf_limits[class].hard_limit_bytes &&
1920  used_mem >= server.client_obuf_limits[class].hard_limit_bytes)
1921  hard = 1;
1922  if (server.client_obuf_limits[class].soft_limit_bytes &&
1923  used_mem >= server.client_obuf_limits[class].soft_limit_bytes)
1924  soft = 1;
1925

1927  * specified amount of seconds. */
1928  if (soft) {

1931  soft = 0; /* First time we see the soft limit reached */
1932  } else {
1933  time_t elapsed = server.unixtime - c->obuf_soft_limit_reached_time;
1934 
1935  if (elapsed <=
1936  server.client_obuf_limits[class].soft_limit_seconds) {
1937  soft = 0; /* The client still did not reached the max number of
1938  seconds for the soft limit to be considered
1939  reached. */
1940  }
1941  }
1942  } else {
1943  c->obuf_soft_limit_reached_time = 0;
1944  }
