35 /*-----------------------------------------------------------------------------
36  * Incremental collection of expired keys.
37  *
38  * When keys are accessed they are expired on-access. However we need a
39  * mechanism in order to ensure keys are eventually removed when expired even
40  * if no access is performed on them.
41  *----------------------------------------------------------------------------*/
42 
43 /* Helper function for the activeExpireCycle() function.
44  * This function will try to expire the key that is stored in the hash table
45  * entry 'de' of the 'expires' hash table of a Redis database.
46  *
47  * If the key is found to be expired, it is removed from the database and
48  * 1 is returned. Otherwise no operation is performed and 0 is returned.
49  *
50  * When a key is expired, server.stat_expiredkeys is incremented.
51  *
52  * The parameter 'now' is the current time in milliseconds as is passed
53  * to the function to avoid too many gettimeofday() syscalls. */
54 int activeExpireCycleTryExpire(redisDb *db, dictEntry *de, long long now) {
55  long long t = dictGetSignedIntegerVal(de);
56  if (now > t) {
57  sds key = dictGetKey(de);
58  robj *keyobj = createStringObject(key,sdslen(key));
59 
60  propagateExpire(db,keyobj,server.lazyfree_lazy_expire);
61  if (server.lazyfree_lazy_expire)
62  dbAsyncDelete(db,keyobj);
63  else
64  dbSyncDelete(db,keyobj);
65  notifyKeyspaceEvent(NOTIFY_EXPIRED,
66  "expired",keyobj,db->id);
67  decrRefCount(keyobj);
68  server.stat_expiredkeys++;
69  return 1;
70  } else {
71  return 0;
72  }
73 }

75 /* Try to expire a few timed out keys. The algorithm used is adaptive and
76  * will use few CPU cycles if there are few expiring keys, otherwise
77  * it will get more aggressive to avoid that too much memory is used by
78  * keys that can be removed from the keyspace.
79  *
80  * No more than CRON_DBS_PER_CALL databases are tested at every
81  * iteration.
82  *
83  * This kind of call is used when Redis detects that timelimit_exit is
84  * true, so there is more work to do, and we do it more incrementally from
85  * the beforeSleep() function of the event loop.
86  *
87  * Expire cycle type:
88  *
89  * If type is ACTIVE_EXPIRE_CYCLE_FAST the function will try to run a
90  * "fast" expire cycle that takes no longer than EXPIRE_FAST_CYCLE_DURATION
91  * microseconds, and is not repeated again before the same amount of time.
92  *
93  * If type is ACTIVE_EXPIRE_CYCLE_SLOW, that normal expire cycle is
94  * executed, where the time limit is a percentage of the REDIS_HZ period
95  * as specified by the ACTIVE_EXPIRE_CYCLE_SLOW_TIME_PERC define. */
96 
97 void activeExpireCycle(int type) {
98  /* This function has some global state in order to continue the work
99  * incrementally across calls. */
100  static unsigned int current_db = 0; /* Last DB tested. */
101  static int timelimit_exit = 0; /* Time limit hit in previous call? */
102  static long long last_fast_cycle = 0; /* When last fast cycle ran. */
103 
104  int j, iteration = 0;
105  int dbs_per_call = CRON_DBS_PER_CALL;
106  long long start = ustime(), timelimit, elapsed;
107 
108  /* When clients are paused the dataset should be static not just from the
109  * POV of clients not being able to write, but also from the POV of
110  * expires and evictions of keys not being performed. */
111  if (clientsArePaused()) return;
112 
113  if (type == ACTIVE_EXPIRE_CYCLE_FAST) {
114  /* Don't start a fast cycle if the previous cycle did not exited
115  * for time limt. Also don't repeat a fast cycle for the same period
116  * as the fast cycle total duration itself. */
117  if (!timelimit_exit) return;
118  if (start < last_fast_cycle + ACTIVE_EXPIRE_CYCLE_FAST_DURATION*2) return;
119  last_fast_cycle = start;
120  }
121 
122  /* We usually should test CRON_DBS_PER_CALL per iteration, with
123  * two exceptions:
124  *
125  * 1) Don't test more DBs than we have.
126  * 2) If last time we hit the time limit, we want to scan all DBs
127  * in this iteration, as there is work to do in some DB and we don't want
128  * expired keys to use memory for too much time. */
129  if (dbs_per_call > server.dbnum || timelimit_exit)
130  dbs_per_call = server.dbnum;
131 
132  /* We can use at max ACTIVE_EXPIRE_CYCLE_SLOW_TIME_PERC percentage of CPU time
133  * per iteration. Since this function gets called with a frequency of
134  * server.hz times per second, the following is the max amount of
135  * microseconds we can spend in this function. */
136  timelimit = 1000000*ACTIVE_EXPIRE_CYCLE_SLOW_TIME_PERC/server.hz/100;
137  timelimit_exit = 0;
138  if (timelimit <= 0) timelimit = 1;
139 
140  if (type == ACTIVE_EXPIRE_CYCLE_FAST)
141  timelimit = ACTIVE_EXPIRE_CYCLE_FAST_DURATION; /* in microseconds. */
142 
143  for (j = 0; j < dbs_per_call && timelimit_exit == 0; j++) {
144  int expired;
145  redisDb *db = server.db+(current_db % server.dbnum);
146 
147  /* Increment the DB now so we are sure if we run out of time
148  * in the current DB we'll restart from the next. This allows to
149  * distribute the time evenly across DBs. */
150  current_db++;
151 
152  /* Continue to expire if at the end of the cycle more than 25%
153  * of the keys were expired. */
154  do {
155  unsigned long num, slots;
156  long long now, ttl_sum;
157  int ttl_samples;
158  iteration++;
159 
160  /* If there is nothing to expire try next DB ASAP. */
161  if ((num = dictSize(db->expires)) == 0) {
162  db->avg_ttl = 0;
163  break;
164  }
165  slots = dictSlots(db->expires);
166  now = mstime();
167 
168  /* When there are less than 1% filled slots getting random
169  * keys is expensive, so stop here waiting for better times...
170  * The dictionary will be resized asap. */
171  if (num && slots > DICT_HT_INITIAL_SIZE &&
172  (num*100/slots < 1)) break;
173 
174  /* The main collection cycle. Sample random keys among keys
175  * with an expire set, checking for expired ones. */
176  expired = 0;
177  ttl_sum = 0;
178  ttl_samples = 0;
179 
180  if (num > ACTIVE_EXPIRE_CYCLE_LOOKUPS_PER_LOOP)
181  num = ACTIVE_EXPIRE_CYCLE_LOOKUPS_PER_LOOP;
182 
183  while (num--) {
184  dictEntry *de;
185  long long ttl;
186 
187  if ((de = dictGetRandomKey(db->expires)) == NULL) break;
188  ttl = dictGetSignedIntegerVal(de)-now;
189  if (activeExpireCycleTryExpire(db,de,now)) expired++;
190  if (ttl > 0) {
191  /* We want the average TTL of keys yet not expired. */
192  ttl_sum += ttl;
193  ttl_samples++;
194  }
195  }
196 
197  /* Update the average TTL stats for this database. */
198  if (ttl_samples) {
199  long long avg_ttl = ttl_sum/ttl_samples;
200 
201  /* Do a simple running average with a few samples.
202  * We just use the current estimate with a weight of 2%
203  * and the previous estimate with a weight of 98%. */
204  if (db->avg_ttl == 0) db->avg_ttl = avg_ttl;
205  db->avg_ttl = (db->avg_ttl/50)*49 + (avg_ttl/50);
206  }
207 
208  /* We can't block forever here even if there are many keys to
209  * expire. So after a given amount of milliseconds return to the
210  * caller waiting for the other active expire cycle. */
211  if ((iteration & 0xf) == 0) { /* check once every 16 iterations. */
212  elapsed = ustime()-start;
213  if (elapsed > timelimit) {
214  timelimit_exit = 1;
215  break;
216  }
217  }
218  /* We don't repeat the cycle if there are less than 25% of keys
219  * found expired in the current DB. */
220  } while (expired > ACTIVE_EXPIRE_CYCLE_LOOKUPS_PER_LOOP/4);
221  }
222 
223  elapsed = ustime()-start;
224  latencyAddSampleIfNeeded("expire-cycle",elapsed/1000);
225 }

227 /*-----------------------------------------------------------------------------
228  * Expires of keys created in writable slaves
229  *
230  * Normally slaves do not process expires: they wait the masters to synthesize
231  * DEL operations in order to retain consistency. However writable slaves are
232  * an exception: if a key is created in the slave and an expire is assigned
233  * to it, we need a way to expire such a key, since the master does not know
234  * anything about such a key.
235  *
236  * In order to do so, we track keys created in the slave side with an expire
237  * set, and call the expireSlaveKeys() function from time to time in order to
238  * reclaim the keys if they already expired.
239  *
240  * Note that the use case we are trying to cover here, is a popular one where
241  * slaves are put in writable mode in order to compute slow operations in
242  * the slave side that are mostly useful to actually read data in a more
243  * processed way. Think at sets intersections in a tmp key, with an expire so
244  * that it is also used as a cache to avoid intersecting every time.
245  *
246  * This implementation is currently not perfect but a lot better than leaking
247  * the keys as implemented in 3.2.
248  *----------------------------------------------------------------------------*/
249 
250 /* The dictionary where we remember key names and database ID of keys we may
251  * want to expire from the slave. Since this function is not often used we
252  * don't even care to initialize the database at startup. We'll do it once
253  * the feature is used the first time, that is, when rememberSlaveKeyWithExpire()
254  * is called.
255  *
256  * The dictionary has an SDS string representing the key as the hash table
257  * key, while the value is a 64 bit unsigned integer with the bits corresponding
258  * to the DB where the keys may exist set to 1. Currently the keys created
259  * with a DB id > 63 are not expired, but a trivial fix is to set the bitmap
260  * to the max 64 bit unsigned value when we know there is a key with a DB
261  * ID greater than 63, and check all the configured DBs in such a case. */
262 dict *slaveKeysWithExpire = NULL;
263 
264 /* Check the set of keys created by the master with an expire set in order to
265  * check if they should be evicted. */
266 void expireSlaveKeys(void) {
267  if (slaveKeysWithExpire == NULL ||
268  dictSize(slaveKeysWithExpire) == 0) return;
269 
270  int cycles = 0, noexpire = 0;
271  mstime_t start = mstime();
272  while(1) {
273  dictEntry *de = dictGetRandomKey(slaveKeysWithExpire);
274  sds keyname = dictGetKey(de);
275  uint64_t dbids = dictGetUnsignedIntegerVal(de);
276  uint64_t new_dbids = 0;
277 
278  /* Check the key against every database corresponding to the
279  * bits set in the value bitmap. */
280  int dbid = 0;
281  while(dbids && dbid < server.dbnum) {
282  if ((dbids & 1) != 0) {
283  redisDb *db = server.db+dbid;
284  dictEntry *expire = dictFind(db->expires,keyname);
285  int expired = 0;
286 
287  if (expire &&
288  activeExpireCycleTryExpire(server.db+dbid,expire,start))
289  {
290  expired = 1;
291  }
292 
293  /* If the key was not expired in this DB, we need to set the
294  * corresponding bit in the new bitmap we set as value.
295  * At the end of the loop if the bitmap is zero, it means we
296  * no longer need to keep track of this key. */
297  if (expire && !expired) {
298  noexpire++;
299  new_dbids |= (uint64_t)1 << dbid;
300  }
301  }
302  dbid++;
303  dbids >>= 1;
304  }
305 
306  /* Set the new bitmap as value of the key, in the dictionary
307  * of keys with an expire set directly in the writable slave. Otherwise
308  * if the bitmap is zero, we no longer need to keep track of it. */
309  if (new_dbids)
310  dictSetUnsignedIntegerVal(de,new_dbids);
311  else
312  dictDelete(slaveKeysWithExpire,keyname);
313 
314  /* Stop conditions: found 3 keys we cna't expire in a row or
315  * time limit was reached. */
316  cycles++;
317  if (noexpire > 3) break;
318  if ((cycles % 64) == 0 && mstime()-start > 1) break;
319  if (dictSize(slaveKeysWithExpire) == 0) break;
320  }
321 }

323 /* Track keys that received an EXPIRE or similar command in the context
324  * of a writable slave. */
325 void rememberSlaveKeyWithExpire(redisDb *db, robj *key) {
326  if (slaveKeysWithExpire == NULL) {
327  static dictType dt = {
328  dictSdsHash, /* hash function */
329  NULL, /* key dup */
330  NULL, /* val dup */
331  dictSdsKeyCompare, /* key compare */
332  dictSdsDestructor, /* key destructor */
333  NULL /* val destructor */
334  };
335  slaveKeysWithExpire = dictCreate(&dt,NULL);
336  }
337  if (db->id > 63) return;
338 
339  dictEntry *de = dictAddOrFind(slaveKeysWithExpire,key->ptr);
340  /* If the entry was just created, set it to a copy of the SDS string
341  * representing the key: we don't want to need to take those keys
342  * in sync with the main DB. The keys will be removed by expireSlaveKeys()
343  * as it scans to find keys to remove. */
344  if (de->key == key->ptr) {
345  de->key = sdsdup(key->ptr);
346  dictSetUnsignedIntegerVal(de,0);
347  }
348 
349  uint64_t dbids = dictGetUnsignedIntegerVal(de);
350  dbids |= (uint64_t)1 << db->id;
351  dictSetUnsignedIntegerVal(de,dbids);
352 }

354 /* Return the number of keys we are tracking. */
355 size_t getSlaveKeyWithExpireCount(void) {
356  if (slaveKeysWithExpire == NULL) return 0;
357  return dictSize(slaveKeysWithExpire);
358 }

360 /* Remove the keys in the hash table. We need to do that when data is
361  * flushed from the server. We may receive new keys from the master with
362  * the same name/db and it is no longer a good idea to expire them.
363  *
364  * Note: technically we should handle the case of a single DB being flushed
365  * but it is not worth it since anyway race conditions using the same set
366  * of key names in a wriatable slave and in its master will lead to
367  * inconsistencies. This is just a best-effort thing we do. */
368 void flushSlaveKeysWithExpireList(void) {
369  if (slaveKeysWithExpire) {
370  dictRelease(slaveKeysWithExpire);
371  slaveKeysWithExpire = NULL;
372  }
373 }

379 /* This is the generic command implementation for EXPIRE, PEXPIRE, EXPIREAT
380  * and PEXPIREAT. Because the commad second argument may be relative or absolute
381  * the "basetime" argument is used to signal what the base time is (either 0
382  * for *AT variants of the command, or the current time for relative expires).
383  *
384  * unit is either UNIT_SECONDS or UNIT_MILLISECONDS, and is only used for
385  * the argv[2] parameter. The basetime is always specified in milliseconds. */
386 void expireGenericCommand(client *c, long long basetime, int unit) {
387  robj *key = c->argv[1], *param = c->argv[2];
388  long long when; /* unix time in milliseconds when the key will expire. */
389 
390  if (getLongLongFromObjectOrReply(c, param, &when, NULL) != C_OK)
391  return;
392 
393  if (unit == UNIT_SECONDS) when *= 1000;
394  when += basetime;
395 
396  /* No key, return zero. */
397  if (lookupKeyWrite(c->db,key) == NULL) {
398  addReply(c,shared.czero);
399  return;
400  }
401 
402  /* EXPIRE with negative TTL, or EXPIREAT with a timestamp into the past
403  * should never be executed as a DEL when load the AOF or in the context
404  * of a slave instance.
405  *
406  * Instead we take the other branch of the IF statement setting an expire
407  * (possibly in the past) and wait for an explicit DEL from the master. */
408  if (when <= mstime() && !server.loading && !server.masterhost) {
409  robj *aux;
410 
411  int deleted = server.lazyfree_lazy_expire ? dbAsyncDelete(c->db,key) :
412  dbSyncDelete(c->db,key);
413  serverAssertWithInfo(c,key,deleted);
414  server.dirty++;
415 
416  /* Replicate/AOF this as an explicit DEL or UNLINK. */
417  aux = server.lazyfree_lazy_expire ? shared.unlink : shared.del;
418  rewriteClientCommandVector(c,2,aux,key);
419  signalModifiedKey(c->db,key);
420  notifyKeyspaceEvent(NOTIFY_GENERIC,"del",key,c->db->id);
421  addReply(c, shared.cone);
422  return;
423  } else {
424  setExpire(c,c->db,key,when);
425  addReply(c,shared.cone);
426  signalModifiedKey(c->db,key);
427  notifyKeyspaceEvent(NOTIFY_GENERIC,"expire",key,c->db->id);
428  server.dirty++;
429  return;
430  }
431 }

453 /* Implements TTL and PTTL */
454 void ttlGenericCommand(client *c, int output_ms) {
455  long long expire, ttl = -1;
456 
457  /* If the key does not exist at all, return -2 */
458  if (lookupKeyReadWithFlags(c->db,c->argv[1],LOOKUP_NOTOUCH) == NULL) {
459  addReplyLongLong(c,-2);
460  return;
461  }
462  /* The key exists. Return -1 if it has no expire, or the actual
463  * TTL value otherwise. */
464  expire = getExpire(c->db,c->argv[1]);
465  if (expire != -1) {
466  ttl = expire-mstime();
467  if (ttl < 0) ttl = 0;
468  }
469  if (ttl == -1) {
470  addReplyLongLong(c,-1);
471  } else {
472  addReplyLongLong(c,output_ms ? ttl : ((ttl+500)/1000));
473  }
474 }
