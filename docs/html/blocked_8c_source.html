32  * API:
33  *
34  * getTimeoutFromObjectOrReply() is just an utility function to parse a
35  * timeout argument since blocking operations usually require a timeout.
36  *
37  * blockClient() set the CLIENT_BLOCKED flag in the client, and set the
38  * specified block type 'btype' filed to one of BLOCKED_* macros.
39  *
40  * unblockClient() unblocks the client doing the following:
41  * 1) It calls the btype-specific function to cleanup the state.
42  * 2) It unblocks the client by unsetting the CLIENT_BLOCKED flag.
43  * 3) It puts the client into a list of just unblocked clients that are
44  * processed ASAP in the beforeSleep() event loop callback, so that
45  * if there is some query buffer to process, we do it. This is also
46  * required because otherwise there is no 'readable' event fired, we
47  * already read the pending commands. We also set the CLIENT_UNBLOCKED
48  * flag to remember the client is in the unblocked_clients list.
49  *
50  * processUnblockedClients() is called inside the beforeSleep() function
51  * to process the query buffer from unblocked clients and remove the clients
52  * from the blocked_clients queue.
53  *
54  * replyToBlockedClientTimedOut() is called by the cron function when
55  * a client blocked reaches the specified timeout (if the timeout is set
56  * to 0, no timeout is processed).
57  * It usually just needs to send a reply to the client.
58  *
59  * When implementing a new type of blocking opeation, the implementation
60  * should modify unblockClient() and replyToBlockedClientTimedOut() in order
61  * to handle the btype-specific behavior of this two functions.
62  * If the blocking operation waits for certain keys to change state, the
63  * clusterRedirectBlockedClientIfNeeded() function should also be updated.
64  */

70 /* Get a timeout value from an object and store it into 'timeout'.
71  * The final timeout is always stored as milliseconds as a time where the
72  * timeout will expire, however the parsing is performed according to
73  * the 'unit' that can be seconds or milliseconds.
74  *
75  * Note that if the timeout is zero (usually from the point of view of
76  * commands API this means no timeout) the value stored into 'timeout'
77  * is zero. */
78 int getTimeoutFromObjectOrReply(client *c, robj *object, mstime_t *timeout, int unit) {
79  long long tval;
80 
81  if (getLongLongFromObjectOrReply(c,object,&tval,
82  "timeout is not an integer or out of range") != C_OK)
83  return C_ERR;
84 
85  if (tval < 0) {
86  addReplyError(c,"timeout is negative");
87  return C_ERR;
88  }
89 
90  if (tval > 0) {
91  if (unit == UNIT_SECONDS) tval *= 1000;
92  tval += mstime();
93  }
94  *timeout = tval;
95 
96  return C_OK;
97 }

99 /* Block a client for the specific operation type. Once the CLIENT_BLOCKED
100  * flag is set client query buffer is not longer processed, but accumulated,
101  * and will be processed when the client is unblocked. */
102 void blockClient(client *c, int btype) {
103  c->flags |= CLIENT_BLOCKED;
104  c->btype = btype;
105  server.blocked_clients++;
106  server.blocked_clients_by_type[btype]++;
107 }

109 /* This function is called in the beforeSleep() function of the event loop
110  * in order to process the pending input buffer of clients that were
111  * unblocked after a blocking operation. */
112 void processUnblockedClients(void) {
113  listNode *ln;
114  client *c;
115 
116  while (listLength(server.unblocked_clients)) {
117  ln = listFirst(server.unblocked_clients);
118  serverAssert(ln != NULL);
119  c = ln->value;
120  listDelNode(server.unblocked_clients,ln);
121  c->flags &= ~CLIENT_UNBLOCKED;
122 
123  /* Process remaining data in the input buffer, unless the client
124  * is blocked again. Actually processInputBuffer() checks that the
125  * client is not blocked before to proceed, but things may change and
126  * the code is conceptually more correct this way. */
127  if (!(c->flags & CLIENT_BLOCKED)) {
128  if (c->querybuf && sdslen(c->querybuf) > 0) {
129  processInputBuffer(c);
130  }
131  }
132  }
133 }

135 /* Unblock a client calling the right function depending on the kind
136  * of operation the client is blocking for. */
137 void unblockClient(client *c) {
138  if (c->btype == BLOCKED_LIST || c->btype == BLOCKED_STREAM) {
139  unblockClientWaitingData(c);
140  } else if (c->btype == BLOCKED_WAIT) {
141  unblockClientWaitingReplicas(c);
142  } else if (c->btype == BLOCKED_MODULE) {
143  unblockClientFromModule(c);
144  } else {
145  serverPanic("Unknown btype in unblockClient().");
146  }
147  /* Clear the flags, and put the client in the unblocked list so that
148  * we'll process new commands in its query buffer ASAP. */
149  server.blocked_clients--;
150  server.blocked_clients_by_type[c->btype]--;
151  c->flags &= ~CLIENT_BLOCKED;
152  c->btype = BLOCKED_NONE;
153  /* The client may already be into the unblocked list because of a previous
154  * blocking operation, don't add back it into the list multiple times. */
155  if (!(c->flags & CLIENT_UNBLOCKED)) {
156  c->flags |= CLIENT_UNBLOCKED;
157  listAddNodeTail(server.unblocked_clients,c);
158  }
159 }

161 /* This function gets called when a blocked client timed out in order to
162  * send it a reply of some kind. After this function is called,
163  * unblockClient() will be called with the same client as argument. */
164 void replyToBlockedClientTimedOut(client *c) {
165  if (c->btype == BLOCKED_LIST || c->btype == BLOCKED_STREAM) {
166  addReply(c,shared.nullmultibulk);
167  } else if (c->btype == BLOCKED_WAIT) {
168  addReplyLongLong(c,replicationCountAcksByOffset(c->bpop.reploffset));
169  } else if (c->btype == BLOCKED_MODULE) {
170  moduleBlockedClientTimedOut(c);
171  } else {
172  serverPanic("Unknown btype in replyToBlockedClientTimedOut().");
173  }
174 }

176 /* Mass-unblock clients because something changed in the instance that makes
177  * blocking no longer safe. For example clients blocked in list operations
178  * in an instance which turns from master to slave is unsafe, so this function
179  * is called when a master turns into a slave.
180  *
181  * The semantics is to send an -UNBLOCKED error to the client, disconnecting
182  * it at the same time. */
183 void disconnectAllBlockedClients(void) {
184  listNode *ln;
185  listIter li;
186 
187  listRewind(server.clients,&li);
188  while((ln = listNext(&li))) {
189  client *c = listNodeValue(ln);
190 
191  if (c->flags & CLIENT_BLOCKED) {
192  addReplySds(c,sdsnew(
193  "-UNBLOCKED force unblock from blocking operation, "
194  "instance state changed (master -> slave?)\r\n"));
195  unblockClient(c);
196  c->flags |= CLIENT_CLOSE_AFTER_REPLY;
197  }
198  }

201 /* This function should be called by Redis every time a single command,
202  * a MULTI/EXEC block, or a Lua script, terminated its execution after
203  * being called by a client.
204  *
205  * All the keys with at least one client blocked that received at least
206  * one new element via some PUSH/XADD operation are accumulated into
207  * the server.ready_keys list. This function will run the list and will
208  * serve clients accordingly. Note that the function will iterate again and
209  * again as a result of serving BRPOPLPUSH we can have new blocking clients
210  * to serve because of the PUSH side of BRPOPLPUSH. */
211 void handleClientsBlockedOnKeys(void) {
212  while(listLength(server.ready_keys) != 0) {
213  list *l;
214 
215  /* Point server.ready_keys to a fresh list and save the current one
216  * locally. This way as we run the old list we are free to call
217  * signalKeyAsReady() that may push new elements in server.ready_keys
218  * when handling clients blocked into BRPOPLPUSH. */
219  l = server.ready_keys;
220  server.ready_keys = listCreate();
221 
222  while(listLength(l) != 0) {
223  listNode *ln = listFirst(l);
224  readyList *rl = ln->value;
225 
226  /* First of all remove this key from db->ready_keys so that
227  * we can safely call signalKeyAsReady() against this key. */
228  dictDelete(rl->db->ready_keys,rl->key);
229 
230  /* Serve clients blocked on list key. */
231  robj *o = lookupKeyWrite(rl->db,rl->key);
232  if (o != NULL && o->type == OBJ_LIST) {
233  dictEntry *de;
234 
235  /* We serve clients in the same order they blocked for
236  * this key, from the first blocked to the last. */
237  de = dictFind(rl->db->blocking_keys,rl->key);
238  if (de) {
239  list *clients = dictGetVal(de);
240  int numclients = listLength(clients);
241 
242  while(numclients--) {
243  listNode *clientnode = listFirst(clients);
244  client *receiver = clientnode->value;
245 
246  if (receiver->btype != BLOCKED_LIST) {
247  /* Put on the tail, so that at the next call
248  * we'll not run into it again. */
249  listDelNode(clients,clientnode);
250  listAddNodeTail(clients,receiver);
251  continue;
252  }
253 
254  robj *dstkey = receiver->bpop.target;
255  int where = (receiver->lastcmd &&
256  receiver->lastcmd->proc == blpopCommand) ?
257  LIST_HEAD : LIST_TAIL;
258  robj *value = listTypePop(o,where);
259 
260  if (value) {
261  /* Protect receiver->bpop.target, that will be
262  * freed by the next unblockClient()
263  * call. */
264  if (dstkey) incrRefCount(dstkey);
265  unblockClient(receiver);
266 
267  if (serveClientBlockedOnList(receiver,
268  rl->key,dstkey,rl->db,value,
269  where) == C_ERR)
270  {
271  /* If we failed serving the client we need
272  * to also undo the POP operation. */
273  listTypePush(o,value,where);
274  }
275 
276  if (dstkey) decrRefCount(dstkey);
277  decrRefCount(value);
278  } else {
279  break;
280  }
281  }
282  }
283 
284  if (listTypeLength(o) == 0) {
285  dbDelete(rl->db,rl->key);
286  }
287  /* We don't call signalModifiedKey() as it was already called
288  * when an element was pushed on the list. */
289  }
290 
291  /* Serve clients blocked on stream key. */
292  else if (o != NULL && o->type == OBJ_STREAM) {
293  dictEntry *de = dictFind(rl->db->blocking_keys,rl->key);
294  stream *s = o->ptr;
295 
296  /* We need to provide the new data arrived on the stream
297  * to all the clients that are waiting for an offset smaller
298  * than the current top item. */
299  if (de) {
300  list *clients = dictGetVal(de);
301  listNode *ln;
302  listIter li;
303  listRewind(clients,&li);
304 
305  while((ln = listNext(&li))) {
306  client *receiver = listNodeValue(ln);
307  if (receiver->btype != BLOCKED_STREAM) continue;
308  streamID *gt = dictFetchValue(receiver->bpop.keys,
309  rl->key);
310  if (s->last_id.ms > gt->ms ||
311  (s->last_id.ms == gt->ms &&
312  s->last_id.seq > gt->seq))
313  {
314  streamID start = *gt;
315  start.seq++; /* Can't overflow, it's an uint64_t */
316  /* Note that after we unblock the client, 'gt'
317  * is no longer valid, so we must do it after
318  * we copied the ID into the 'start' variable. */
319  unblockClient(receiver);
320 
321  /* Emit the two elements sub-array consisting of
322  * the name of the stream and the data we
323  * extracted from it. Wrapped in a single-item
324  * array, since we have just one key. */
325  addReplyMultiBulkLen(receiver,1);
326  addReplyMultiBulkLen(receiver,2);
327  addReplyBulk(receiver,rl->key);
328  streamReplyWithRange(receiver,s,&start,NULL,
329  receiver->bpop.xread_count,0);
330  }
331  }
332  }
333  }
334 
335  /* Free this item. */
336  decrRefCount(rl->key);
337  zfree(rl);
338  listDelNode(l,ln);
339  }
340  listRelease(l); /* We have the new list on place at this point. */
341  }
342 }

344 /* This is how the current blocking lists/streams work, we use BLPOP as
345  * example, but the concept is the same for other list ops and XREAD.
346  * - If the user calls BLPOP and the key exists and contains a non empty list
347  * then LPOP is called instead. So BLPOP is semantically the same as LPOP
348  * if blocking is not required.
349  * - If instead BLPOP is called and the key does not exists or the list is
350  * empty we need to block. In order to do so we remove the notification for
351  * new data to read in the client socket (so that we'll not serve new
352  * requests if the blocking request is not served). Also we put the client
353  * in a dictionary (db->blocking_keys) mapping keys to a list of clients
354  * blocking for this keys.
355  * - If a PUSH operation against a key with blocked clients waiting is
356  * performed, we mark this key as "ready", and after the current command,
357  * MULTI/EXEC block, or script, is executed, we serve all the clients waiting
358  * for this list, from the one that blocked first, to the last, accordingly
359  * to the number of elements we have in the ready list.
360  */
361 
362 /* Set a client in blocking mode for the specified key (list or stream), with
363  * the specified timeout. The 'type' argument is BLOCKED_LIST or BLOCKED_STREAM
364  * depending on the kind of operation we are waiting for an empty key in
365  * order to awake the client. The client is blocked for all the 'numkeys'
366  * keys as in the 'keys' argument. When we block for stream keys, we also
367  * provide an array of streamID structures: clients will be unblocked only
368  * when items with an ID greater or equal to the specified one is appended
369  * to the stream. */
370 void blockForKeys(client *c, int btype, robj **keys, int numkeys, mstime_t timeout, robj *target, streamID *ids) {
371  dictEntry *de;
372  list *l;
373  int j;
374 
375  c->bpop.timeout = timeout;
376  c->bpop.target = target;
377 
378  if (target != NULL) incrRefCount(target);
379 
380  for (j = 0; j < numkeys; j++) {
381  /* The value associated with the key name in the bpop.keys dictionary
382  * is NULL for lists, or the stream ID for streams. */
383  void *key_data = NULL;
384  if (btype == BLOCKED_STREAM) {
385  key_data = zmalloc(sizeof(streamID));
386  memcpy(key_data,ids+j,sizeof(streamID));
387  }
388 
389  /* If the key already exists in the dictionary ignore it. */
390  if (dictAdd(c->bpop.keys,keys[j],key_data) != DICT_OK) {
391  zfree(key_data);
392  continue;
393  }
394  incrRefCount(keys[j]);
395 
396  /* And in the other "side", to map keys -> clients */
397  de = dictFind(c->db->blocking_keys,keys[j]);
398  if (de == NULL) {
399  int retval;
400 
401  /* For every key we take a list of clients blocked for it */
402  l = listCreate();
403  retval = dictAdd(c->db->blocking_keys,keys[j],l);
404  incrRefCount(keys[j]);
405  serverAssertWithInfo(c,keys[j],retval == DICT_OK);
406  } else {
407  l = dictGetVal(de);
408  }
409  listAddNodeTail(l,c);
410  }
411  blockClient(c,btype);
412 }

414 /* Unblock a client that's waiting in a blocking operation such as BLPOP.
415  * You should never call this function directly, but unblockClient() instead. */
416 void unblockClientWaitingData(client *c) {
417  dictEntry *de;
418  dictIterator *di;
419  list *l;
420 
421  serverAssertWithInfo(c,NULL,dictSize(c->bpop.keys) != 0);
422  di = dictGetIterator(c->bpop.keys);
423  /* The client may wait for multiple keys, so unblock it for every key. */
424  while((de = dictNext(di)) != NULL) {
425  robj *key = dictGetKey(de);
426 
427  /* Remove this client from the list of clients waiting for this key. */
428  l = dictFetchValue(c->db->blocking_keys,key);
429  serverAssertWithInfo(c,key,l != NULL);
430  listDelNode(l,listSearchKey(l,c));
431  /* If the list is empty we need to remove it to avoid wasting memory */
432  if (listLength(l) == 0)
433  dictDelete(c->db->blocking_keys,key);
434  }
435  dictReleaseIterator(di);
436 
437  /* Cleanup the client structure */
438  dictEmpty(c->bpop.keys,NULL);
439  if (c->bpop.target) {
440  decrRefCount(c->bpop.target);
441  c->bpop.target = NULL;
442  }
443  if (c->bpop.xread_group) {
444  decrRefCount(c->bpop.xread_group);
445  c->bpop.xread_group = NULL;
446  }
447 }

449 /* If the specified key has clients blocked waiting for list pushes, this
450  * function will put the key reference into the server.ready_keys list.
451  * Note that db->ready_keys is a hash table that allows us to avoid putting
452  * the same key again and again in the list in case of multiple pushes
453  * made by a script or in the context of MULTI/EXEC.
454  *
455  * The list will be finally processed by handleClientsBlockedOnLists() */
456 void signalKeyAsReady(redisDb *db, robj *key) {
457  readyList *rl;
458 
459  /* No clients blocking for this key? No need to queue it. */
460  if (dictFind(db->blocking_keys,key) == NULL) return;
461 
462  /* Key was already signaled? No need to queue it again. */
463  if (dictFind(db->ready_keys,key) != NULL) return;
464 
465  /* Ok, we need to queue this key into server.ready_keys. */
466  rl = zmalloc(sizeof(*rl));
467  rl->key = key;
468  rl->db = db;
469  incrRefCount(key);
470  listAddNodeTail(server.ready_keys,rl);
471 
472  /* We also add the key in the db->ready_keys dictionary in order
473  * to avoid adding it multiple times into a list with a simple O(1)
474  * check. */
475  incrRefCount(key);
476  serverAssert(dictAdd(db->ready_keys,key,NULL) == DICT_OK);
477 }
