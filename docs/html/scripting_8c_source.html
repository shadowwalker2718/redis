83 /* Perform the SHA1 of the input string. We use this both for hashing script
84  * bodies in order to obtain the Lua function name, and in the implementation
85  * of redis.sha1().
86  *
87  * 'digest' should point to a 41 bytes buffer: 40 for SHA1 converted into an
88  * hexadecimal number, plus 1 byte for null term. */
89 void sha1hex(char *digest, char *script, size_t len) {
90  SHA1_CTX ctx;
91  unsigned char hash[20];
92  char *cset = "0123456789abcdef";
93  int j;
94 
95  SHA1Init(&ctx);
96  SHA1Update(&ctx,(unsigned char*)script,len);
97  SHA1Final(hash,&ctx);
98 
99  for (j = 0; j < 20; j++) {
100  digest[j*2] = cset[((hash[j]&0xF0)>>4)];
101  digest[j*2+1] = cset[(hash[j]&0xF)];
102  }
103  digest[40] = '\0';
104 }

110 /* Take a Redis reply in the Redis protocol format and convert it into a
111  * Lua type. Thanks to this function, and the introduction of not connected
112  * clients, it is trivial to implement the redis() lua function.
113  *
114  * Basically we take the arguments, execute the Redis command in the context
115  * of a non connected client, then take the generated reply and convert it
116  * into a suitable Lua type. With this trick the scripting feature does not
117  * need the introduction of a full Redis internals API. The script
118  * is like a normal client that bypasses all the slow I/O paths.
119  *
120  * Note: in this function we do not do any sanity check as the reply is
121  * generated by Redis directly. This allows us to go faster.
122  *
123  * Errors are returned as a table with a single 'err' field set to the
124  * error string.
125  */
126 
127 char *redisProtocolToLuaType(lua_State *lua, char* reply) {
128  char *p = reply;
129 
130  switch(*p) {
131  case ':': p = redisProtocolToLuaType_Int(lua,reply); break;
132  case '$': p = redisProtocolToLuaType_Bulk(lua,reply); break;
133  case '+': p = redisProtocolToLuaType_Status(lua,reply); break;
134  case '-': p = redisProtocolToLuaType_Error(lua,reply); break;
135  case '*': p = redisProtocolToLuaType_MultiBulk(lua,reply); break;
136  }
137  return p;
138 }

149 char *redisProtocolToLuaType_Bulk(lua_State *lua, char *reply) {
150  char *p = strchr(reply+1,'\r');

152

154  if (bulklen == -1) {

156  return p+2;
157  } else {
158  lua_pushlstring(lua,p+2,bulklen);
159  return p+2+bulklen+2;

183 char *redisProtocolToLuaType_MultiBulk(lua_State *lua, char *reply) {
184  char *p = strchr(reply+1,'\r');
185  long long mbulklen;
186  int j = 0;
187 
188  string2ll(reply+1,p-reply-1,&mbulklen);
189  p += 2;
190  if (mbulklen == -1) {
191  lua_pushboolean(lua,0);
192  return p;
193  }
194  lua_newtable(lua);
195  for (j = 0; j < mbulklen; j++) {
196  lua_pushnumber(lua,j+1);
197  p = redisProtocolToLuaType(lua,p);
198  lua_settable(lua,-3);
199  }
200  return p;

203 /* This function is used in order to push an error on the Lua stack in the
204  * format used by redis.pcall to return errors, which is a lua table
205  * with a single "err" field set to the error string. Note that this
206  * table is never a valid reply by proper commands, since the returned
207  * tables are otherwise always indexed by integers, never by strings. */
208 void luaPushError(lua_State *lua, char *error) {
209  lua_Debug dbg;
210 
211  /* If debugging is active and in step mode, log errors resulting from
212  * Redis commands. */
213  if (ldb.active && ldb.step) {
214  ldbLog(sdscatprintf(sdsempty(),"<error> %s",error));
215  }
216 
217  lua_newtable(lua);
218  lua_pushstring(lua,"err");
219 
220  /* Attempt to figure out where this function was called, if possible */
221  if(lua_getstack(lua, 1, &dbg) && lua_getinfo(lua, "nSl", &dbg)) {
222  sds msg = sdscatprintf(sdsempty(), "%s: %d: %s",
223  dbg.source, dbg.currentline, error);
224  lua_pushstring(lua, msg);
225  sdsfree(msg);
226  } else {
227  lua_pushstring(lua, error);
228  }
229  lua_settable(lua,-3);
230 }
231 
232 /* In case the error set into the Lua stack by luaPushError() was generated
233  * by the non-error-trapping version of redis.pcall(), which is redis.call(),
234  * this function will raise the Lua error so that the execution of the
235  * script will be halted. */
236 int luaRaiseError(lua_State *lua) {
237  lua_pushstring(lua,"err");
238  lua_gettable(lua,-2);
239  return lua_error(lua);
240 }

242 /* Sort the array currently in the stack. We do this to make the output
243  * of commands like KEYS or SMEMBERS something deterministic when called
244  * from Lua (to play well with AOf/replication).
245  *
246  * The array is sorted using table.sort itself, and assuming all the
247  * list elements are strings. */
248 void luaSortArray(lua_State *lua) {
249  /* Initial Stack: array */
250  lua_getglobal(lua,"table");
251  lua_pushstring(lua,"sort");
252  lua_gettable(lua,-2); /* Stack: array, table, table.sort */
253  lua_pushvalue(lua,-3); /* Stack: array, table, table.sort, array */
254  if (lua_pcall(lua,1,0,0)) {
255  /* Stack: array, table, error */
256 
257  /* We are not interested in the error, we assume that the problem is
258  * that there are 'false' elements inside the array, so we try
259  * again with a slower function but able to handle this case, that
260  * is: table.sort(table, __redis__compare_helper) */
261  lua_pop(lua,1); /* Stack: array, table */
262  lua_pushstring(lua,"sort"); /* Stack: array, table, sort */
263  lua_gettable(lua,-2); /* Stack: array, table, table.sort */
264  lua_pushvalue(lua,-3); /* Stack: array, table, table.sort, array */
265  lua_getglobal(lua,"__redis__compare_helper");
266  /* Stack: array, table, table.sort, array, __redis__compare_helper */
267  lua_call(lua,2,0);
268  }
269  /* Stack: array (sorted), table */
270  lua_pop(lua,1); /* Stack: array (sorted) */
271 }

277 void luaReplyToRedisReply(client *c, lua_State *lua) {
278  int t = lua_type(lua,-1);
279 
280  switch(t) {
281  case LUA_TSTRING:
282  addReplyBulkCBuffer(c,(char*)lua_tostring(lua,-1),lua_strlen(lua,-1));
283  break;
284  case LUA_TBOOLEAN:
285  addReply(c,lua_toboolean(lua,-1) ? shared.cone : shared.nullbulk);
286  break;
287  case LUA_TNUMBER:
288  addReplyLongLong(c,(long long)lua_tonumber(lua,-1));
289  break;
290  case LUA_TTABLE:
291  /* We need to check if it is an array, an error, or a status reply.
292  * Error are returned as a single element table with 'err' field.
293  * Status replies are returned as single element table with 'ok'
294  * field. */
295  lua_pushstring(lua,"err");
296  lua_gettable(lua,-2);
297  t = lua_type(lua,-1);
298  if (t == LUA_TSTRING) {
299  sds err = sdsnew(lua_tostring(lua,-1));
300  sdsmapchars(err,"\r\n"," ",2);
301  addReplySds(c,sdscatprintf(sdsempty(),"-%s\r\n",err));
302  sdsfree(err);
303  lua_pop(lua,2);
304  return;
305  }
306 
307  lua_pop(lua,1);
308  lua_pushstring(lua,"ok");
309  lua_gettable(lua,-2);
310  t = lua_type(lua,-1);
311  if (t == LUA_TSTRING) {
312  sds ok = sdsnew(lua_tostring(lua,-1));
313  sdsmapchars(ok,"\r\n"," ",2);
314  addReplySds(c,sdscatprintf(sdsempty(),"+%s\r\n",ok));
315  sdsfree(ok);
316  lua_pop(lua,1);
317  } else {
318  void *replylen = addDeferredMultiBulkLength(c);
319  int j = 1, mbulklen = 0;
320 
321  lua_pop(lua,1); /* Discard the 'ok' field value we popped */
322  while(1) {
323  lua_pushnumber(lua,j++);
324  lua_gettable(lua,-2);
325  t = lua_type(lua,-1);
326  if (t == LUA_TNIL) {
327  lua_pop(lua,1);
328  break;
329  }
330  luaReplyToRedisReply(c, lua);
331  mbulklen++;
332  }
333  setDeferredMultiBulkLength(c,replylen,mbulklen);
334  }
335  break;
336  default:
337  addReply(c,shared.nullbulk);
338  }
339  lua_pop(lua,1);
340 }

346 #define LUA_CMD_OBJCACHE_SIZE 32
347 #define LUA_CMD_OBJCACHE_MAX_LEN 64
348 int luaRedisGenericCommand(lua_State *lua, int raise_error) {
349  int j, argc = lua_gettop(lua);
350  struct redisCommand *cmd;
351  client *c = server.lua_client;
352  sds reply;
353 
354  /* Cached across calls. */
355  static robj **argv = NULL;
356  static int argv_size = 0;
357  static robj *cached_objects[LUA_CMD_OBJCACHE_SIZE];
358  static size_t cached_objects_len[LUA_CMD_OBJCACHE_SIZE];
359  static int inuse = 0; /* Recursive calls detection. */
360 
361  /* Reflect MULTI state */
362  if (server.lua_multi_emitted || (server.lua_caller->flags & CLIENT_MULTI)) {
363  c->flags |= CLIENT_MULTI;
364  } else {
365  c->flags &= ~CLIENT_MULTI;
366  }
367 
368  /* By using Lua debug hooks it is possible to trigger a recursive call
369  * to luaRedisGenericCommand(), which normally should never happen.
370  * To make this function reentrant is futile and makes it slower, but
371  * we should at least detect such a misuse, and abort. */
372  if (inuse) {
373  char *recursion_warning =
374  "luaRedisGenericCommand() recursive call detected. "
375  "Are you doing funny stuff with Lua debug hooks?";
376  serverLog(LL_WARNING,"%s",recursion_warning);
377  luaPushError(lua,recursion_warning);
378  return 1;
379  }
380  inuse++;
381 
382  /* Require at least one argument */
383  if (argc == 0) {
384  luaPushError(lua,
385  "Please specify at least one argument for redis.call()");
386  inuse--;
387  return raise_error ? luaRaiseError(lua) : 1;
388  }
389 
390  /* Build the arguments vector */
391  if (argv_size < argc) {
392  argv = zrealloc(argv,sizeof(robj*)*argc);
393  argv_size = argc;
394  }
395 
396  for (j = 0; j < argc; j++) {
397  char *obj_s;
398  size_t obj_len;
399  char dbuf[64];
400 
401  if (lua_type(lua,j+1) == LUA_TNUMBER) {
402  /* We can't use lua_tolstring() for number -> string conversion
403  * since Lua uses a format specifier that loses precision. */
404  lua_Number num = lua_tonumber(lua,j+1);
405 
406  obj_len = snprintf(dbuf,sizeof(dbuf),"%.17g",(double)num);
407  obj_s = dbuf;
408  } else {
409  obj_s = (char*)lua_tolstring(lua,j+1,&obj_len);
410  if (obj_s == NULL) break; /* Not a string. */
411  }
412 
413  /* Try to use a cached object. */
414  if (j < LUA_CMD_OBJCACHE_SIZE && cached_objects[j] &&
415  cached_objects_len[j] >= obj_len)
416  {
417  sds s = cached_objects[j]->ptr;
418  argv[j] = cached_objects[j];
419  cached_objects[j] = NULL;
420  memcpy(s,obj_s,obj_len+1);
421  sdssetlen(s, obj_len);
422  } else {
423  argv[j] = createStringObject(obj_s, obj_len);
424  }
425  }
426 
427  /* Check if one of the arguments passed by the Lua script
428  * is not a string or an integer (lua_isstring() return true for
429  * integers as well). */
430  if (j != argc) {
431  j--;
432  while (j >= 0) {
433  decrRefCount(argv[j]);
434  j--;
435  }
436  luaPushError(lua,
437  "Lua redis() command arguments must be strings or integers");
438  inuse--;
439  return raise_error ? luaRaiseError(lua) : 1;
440  }
441 
442  /* Setup our fake client for command execution */
443  c->argv = argv;
444  c->argc = argc;
445 
446  /* Log the command if debugging is active. */
447  if (ldb.active && ldb.step) {
448  sds cmdlog = sdsnew("<redis>");
449  for (j = 0; j < c->argc; j++) {
450  if (j == 10) {
451  cmdlog = sdscatprintf(cmdlog," ... (%d more)",
452  c->argc-j-1);
453  break;
454  } else {
455  cmdlog = sdscatlen(cmdlog," ",1);
456  cmdlog = sdscatsds(cmdlog,c->argv[j]->ptr);
457  }
458  }
459  ldbLog(cmdlog);
460  }
461 
462  /* Command lookup */
463  cmd = lookupCommand(argv[0]->ptr);
464  if (!cmd || ((cmd->arity > 0 && cmd->arity != argc) ||
465  (argc < -cmd->arity)))
466  {
467  if (cmd)
468  luaPushError(lua,
469  "Wrong number of args calling Redis command From Lua script");
470  else
471  luaPushError(lua,"Unknown Redis command called from Lua script");
472  goto cleanup;
473  }
474  c->cmd = c->lastcmd = cmd;
475 
476  /* There are commands that are not allowed inside scripts. */
477  if (cmd->flags & CMD_NOSCRIPT) {
478  luaPushError(lua, "This Redis command is not allowed from scripts");
479  goto cleanup;
480  }
481 
482  /* Write commands are forbidden against read-only slaves, or if a
483  * command marked as non-deterministic was already called in the context
484  * of this script. */
485  if (cmd->flags & CMD_WRITE) {
486  if (server.lua_random_dirty && !server.lua_replicate_commands) {
487  luaPushError(lua,
488  "Write commands not allowed after non deterministic commands. Call redis.replicate_commands() at the start of your script in order to switch to single commands replication mode.");
489  goto cleanup;
490  } else if (server.masterhost && server.repl_slave_ro &&
491  !server.loading &&
492  !(server.lua_caller->flags & CLIENT_MASTER))
493  {
494  luaPushError(lua, shared.roslaveerr->ptr);
495  goto cleanup;
496  } else if (server.stop_writes_on_bgsave_err &&
497  server.saveparamslen > 0 &&
498  server.lastbgsave_status == C_ERR)
499  {
500  luaPushError(lua, shared.bgsaveerr->ptr);
501  goto cleanup;
502  }
503  }
504 
505  /* If we reached the memory limit configured via maxmemory, commands that
506  * could enlarge the memory usage are not allowed, but only if this is the
507  * first write in the context of this script, otherwise we can't stop
508  * in the middle. */
509  if (server.maxmemory && server.lua_write_dirty == 0 &&
510  (cmd->flags & CMD_DENYOOM))
511  {
512  if (freeMemoryIfNeeded() == C_ERR) {
513  luaPushError(lua, shared.oomerr->ptr);
514  goto cleanup;
515  }
516  }
517 
518  if (cmd->flags & CMD_RANDOM) server.lua_random_dirty = 1;
519  if (cmd->flags & CMD_WRITE) server.lua_write_dirty = 1;
520 
521  /* If this is a Redis Cluster node, we need to make sure Lua is not
522  * trying to access non-local keys, with the exception of commands
523  * received from our master or when loading the AOF back in memory. */
524  if (server.cluster_enabled && !server.loading &&
525  !(server.lua_caller->flags & CLIENT_MASTER))
526  {
527  /* Duplicate relevant flags in the lua client. */
528  c->flags &= ~(CLIENT_READONLY|CLIENT_ASKING);
529  c->flags |= server.lua_caller->flags & (CLIENT_READONLY|CLIENT_ASKING);
530  if (getNodeByQuery(c,c->cmd,c->argv,c->argc,NULL,NULL) !=
531  server.cluster->myself)
532  {
533  luaPushError(lua,
534  "Lua script attempted to access a non local key in a "
535  "cluster node");
536  goto cleanup;
537  }
538  }
539 
540  /* If we are using single commands replication, we need to wrap what
541  * we propagate into a MULTI/EXEC block, so that it will be atomic like
542  * a Lua script in the context of AOF and slaves. */
543  if (server.lua_replicate_commands &&
544  !server.lua_multi_emitted &&
545  !(server.lua_caller->flags & CLIENT_MULTI) &&
546  server.lua_write_dirty &&
547  server.lua_repl != PROPAGATE_NONE)
548  {
549  execCommandPropagateMulti(server.lua_caller);
550  server.lua_multi_emitted = 1;
551  }
552 
553  /* Run the command */
554  int call_flags = CMD_CALL_SLOWLOG | CMD_CALL_STATS;
555  if (server.lua_replicate_commands) {
556  /* Set flags according to redis.set_repl() settings. */
557  if (server.lua_repl & PROPAGATE_AOF)
558  call_flags |= CMD_CALL_PROPAGATE_AOF;
559  if (server.lua_repl & PROPAGATE_REPL)
560  call_flags |= CMD_CALL_PROPAGATE_REPL;
561  }
562  call(c,call_flags);
563 
564  /* Convert the result of the Redis command into a suitable Lua type.
565  * The first thing we need is to create a single string from the client
566  * output buffers. */
567  if (listLength(c->reply) == 0 && c->bufpos < PROTO_REPLY_CHUNK_BYTES) {
568  /* This is a fast path for the common case of a reply inside the
569  * client static buffer. Don't create an SDS string but just use
570  * the client buffer directly. */
571  c->buf[c->bufpos] = '\0';
572  reply = c->buf;
573  c->bufpos = 0;
574  } else {
575  reply = sdsnewlen(c->buf,c->bufpos);
576  c->bufpos = 0;
577  while(listLength(c->reply)) {
578  sds o = listNodeValue(listFirst(c->reply));
579 
580  reply = sdscatsds(reply,o);
581  listDelNode(c->reply,listFirst(c->reply));
582  }
583  }
584  if (raise_error && reply[0] != '-') raise_error = 0;
585  redisProtocolToLuaType(lua,reply);
586 
587  /* If the debugger is active, log the reply from Redis. */
588  if (ldb.active && ldb.step)
589  ldbLogRedisReply(reply);
590 
591  /* Sort the output array if needed, assuming it is a non-null multi bulk
592  * reply as expected. */
593  if ((cmd->flags & CMD_SORT_FOR_SCRIPT) &&
594  (server.lua_replicate_commands == 0) &&
595  (reply[0] == '*' && reply[1] != '-')) {
596  luaSortArray(lua);
597  }
598  if (reply != c->buf) sdsfree(reply);
599  c->reply_bytes = 0;

1105 /* The following implementation is the one shipped with Lua itself but with
1106  * rand() replaced by redisLrand48(). */
1107 int redis_math_random (lua_State *L) {
1108  /* the `%' avoids the (rare) case of r==1, and is needed also because on
1109  some systems (SunOS!) `rand()' may return a value larger than RAND_MAX */
1110  lua_Number r = (lua_Number)(redisLrand48()%REDIS_LRAND48_MAX) /
1111  (lua_Number)REDIS_LRAND48_MAX;
1112  switch (lua_gettop(L)) { /* check number of arguments */
1113  case 0: { /* no arguments */
1114  lua_pushnumber(L, r); /* Number between 0 and 1 */
1115  break;
1116  }
1117  case 1: { /* only upper limit */
1118  int u = luaL_checkint(L, 1);
1119  luaL_argcheck(L, 1<=u, 1, "interval is empty");
1120  lua_pushnumber(L, floor(r*u)+1); /* int between 1 and `u' */
1121  break;
1122  }
1123  case 2: { /* lower and upper limits */
1124  int l = luaL_checkint(L, 1);
1125  int u = luaL_checkint(L, 2);
1126  luaL_argcheck(L, l<=u, 2, "interval is empty");
1127  lua_pushnumber(L, floor(r*(u-l+1))+l); /* int between `l' and `u' */
1128  break;
1129  }
1130  default: return luaL_error(L, "wrong number of arguments");
1131  }
1132  return 1;
1133 }

1171  sdsfree(sha);

1177  funcdef = sdscatlen(funcdef,funcname,42);
1178  funcdef = sdscatlen(funcdef,"() ",3);

1239 void evalGenericCommand(client *c, int evalsha) {
1240  lua_State *lua = server.lua;
1241  char funcname[43];
1242  long long numkeys;
1243  int delhook = 0, err;
1244 
1245  /* When we replicate whole scripts, we want the same PRNG sequence at
1246  * every call so that our PRNG is not affected by external state. */
1247  redisSrand48(0);
1248 
1249  /* We set this flag to zero to remember that so far no random command
1250  * was called. This way we can allow the user to call commands like
1251  * SRANDMEMBER or RANDOMKEY from Lua scripts as far as no write command
1252  * is called (otherwise the replication and AOF would end with non
1253  * deterministic sequences).
1254  *
1255  * Thanks to this flag we'll raise an error every time a write command
1256  * is called after a random command was used. */
1257  server.lua_random_dirty = 0;
1258  server.lua_write_dirty = 0;
1259  server.lua_replicate_commands = server.lua_always_replicate_commands;
1260  server.lua_multi_emitted = 0;
1261  server.lua_repl = PROPAGATE_AOF|PROPAGATE_REPL;
1262 
1263  /* Get the number of arguments that are keys */
1264  if (getLongLongFromObjectOrReply(c,c->argv[2],&numkeys,NULL) != C_OK)
1265  return;
1266  if (numkeys > (c->argc - 3)) {
1267  addReplyError(c,"Number of keys can't be greater than number of args");
1268  return;
1269  } else if (numkeys < 0) {
1270  addReplyError(c,"Number of keys can't be negative");
1271  return;
1272  }
1273 
1274  /* We obtain the script SHA1, then check if this function is already
1275  * defined into the Lua state */
1276  funcname[0] = 'f';
1277  funcname[1] = '_';
1278  if (!evalsha) {
1279  /* Hash the code if this is an EVAL call */
1280  sha1hex(funcname+2,c->argv[1]->ptr,sdslen(c->argv[1]->ptr));
1281  } else {
1282  /* We already have the SHA if it is a EVALSHA */
1283  int j;
1284  char *sha = c->argv[1]->ptr;
1285 
1286  /* Convert to lowercase. We don't use tolower since the function
1287  * managed to always show up in the profiler output consuming
1288  * a non trivial amount of time. */
1289  for (j = 0; j < 40; j++)
1290  funcname[j+2] = (sha[j] >= 'A' && sha[j] <= 'Z') ?
1291  sha[j]+('a'-'A') : sha[j];
1292  funcname[42] = '\0';
1293  }
1294 
1295  /* Push the pcall error handler function on the stack. */
1296  lua_getglobal(lua, "__redis__err__handler");
1297 
1298  /* Try to lookup the Lua function */
1299  lua_getglobal(lua, funcname);
1300  if (lua_isnil(lua,-1)) {
1301  lua_pop(lua,1); /* remove the nil from the stack */
1302  /* Function not defined... let's define it if we have the
1303  * body of the function. If this is an EVALSHA call we can just
1304  * return an error. */
1305  if (evalsha) {
1306  lua_pop(lua,1); /* remove the error handler from the stack. */
1307  addReply(c, shared.noscripterr);
1308  return;
1309  }
1310  if (luaCreateFunction(c,lua,c->argv[1]) == NULL) {
1311  lua_pop(lua,1); /* remove the error handler from the stack. */
1312  /* The error is sent to the client by luaCreateFunction()
1313  * itself when it returns NULL. */
1314  return;
1315  }
1316  /* Now the following is guaranteed to return non nil */
1317  lua_getglobal(lua, funcname);
1318  serverAssert(!lua_isnil(lua,-1));
1319  }
1320 
1321  /* Populate the argv and keys table accordingly to the arguments that
1322  * EVAL received. */
1323  luaSetGlobalArray(lua,"KEYS",c->argv+3,numkeys);
1324  luaSetGlobalArray(lua,"ARGV",c->argv+3+numkeys,c->argc-3-numkeys);
1325 
1326  /* Select the right DB in the context of the Lua client */
1327  selectDb(server.lua_client,c->db->id);
1328 
1329  /* Set a hook in order to be able to stop the script execution if it
1330  * is running for too much time.
1331  * We set the hook only if the time limit is enabled as the hook will
1332  * make the Lua script execution slower.
1333  *
1334  * If we are debugging, we set instead a "line" hook so that the
1335  * debugger is call-back at every line executed by the script. */
1336  server.lua_caller = c;
1337  server.lua_time_start = mstime();
1338  server.lua_kill = 0;
1339  if (server.lua_time_limit > 0 && server.masterhost == NULL &&
1340  ldb.active == 0)
1341  {
1342  lua_sethook(lua,luaMaskCountHook,LUA_MASKCOUNT,100000);
1343  delhook = 1;
1344  } else if (ldb.active) {
1345  lua_sethook(server.lua,luaLdbLineHook,LUA_MASKLINE|LUA_MASKCOUNT,100000);
1346  delhook = 1;
1347  }
1348 
1349  /* At this point whether this script was never seen before or if it was
1350  * already defined, we can call it. We have zero arguments and expect
1351  * a single return value. */
1352  err = lua_pcall(lua,0,1,-2);
1353 
1354  /* Perform some cleanup that we need to do both on error and success. */
1355  if (delhook) lua_sethook(lua,NULL,0,0); /* Disable hook */
1356  if (server.lua_timedout) {
1357  server.lua_timedout = 0;
1358  /* Restore the readable handler that was unregistered when the
1359  * script timeout was detected. */
1360  aeCreateFileEvent(server.el,c->fd,AE_READABLE,
1361  readQueryFromClient,c);
1362  }
1363  server.lua_caller = NULL;
1364 
1365  /* Call the Lua garbage collector from time to time to avoid a
1366  * full cycle performed by Lua, which adds too latency.
1367  *
1368  * The call is performed every LUA_GC_CYCLE_PERIOD executed commands
1369  * (and for LUA_GC_CYCLE_PERIOD collection steps) because calling it
1370  * for every command uses too much CPU. */
1371  #define LUA_GC_CYCLE_PERIOD 50
1372  {
1373  static long gc_count = 0;
1374 
1375  gc_count++;
1376  if (gc_count == LUA_GC_CYCLE_PERIOD) {
1377  lua_gc(lua,LUA_GCSTEP,LUA_GC_CYCLE_PERIOD);
1378  gc_count = 0;
1379  }
1380  }
1381 
1382  if (err) {
1383  addReplyErrorFormat(c,"Error running script (call to %s): %s\n",
1384  funcname, lua_tostring(lua,-1));
1385  lua_pop(lua,2); /* Consume the Lua reply and remove error handler. */
1386  } else {
1387  /* On success convert the Lua return value into Redis protocol, and
1388  * send it to * the client. */
1389  luaReplyToRedisReply(c,lua); /* Convert and consume the reply. */
1390  lua_pop(lua,1); /* Remove the error handler. */
1391  }
1392 
1393  /* If we are using single commands replication, emit EXEC if there
1394  * was at least a write. */
1395  if (server.lua_replicate_commands) {
1396  preventCommandPropagation(c);
1397  if (server.lua_multi_emitted) {
1398  robj *propargv[1];
1399  propargv[0] = createStringObject("EXEC",4);
1400  alsoPropagate(server.execCommand,c->db->id,propargv,1,
1401  PROPAGATE_AOF|PROPAGATE_REPL);
1402  decrRefCount(propargv[0]);
1403  }
1404  }

1407  * we are sure that the script was already in the context of all the
1408  * attached slaves *and* the current AOF file if enabled.
1409  *
1410  * To do so we use a cache of SHA1s of scripts that we already propagated
1411  * as full EVAL, that's called the Replication Script Cache.
1412  *
1413  * For repliation, everytime a new slave attaches to the master, we need to
1414  * flush our cache of scripts that can be replicated as EVALSHA, while
1415  * for AOF we need to do so every time we rewrite the AOF file. */
1416  if (evalsha && !server.lua_replicate_commands) {
1417  if (!replicationScriptCacheExists(c->argv[1]->ptr)) {
1418  /* This script is not in our script cache, replicate it as
1419  * EVAL, then add it into the script cache, as from now on
1420  * slaves and AOF know about it. */
1421  robj *script = dictFetchValue(server.lua_scripts,c->argv[1]->ptr);
1422 
1423  replicationScriptCacheAdd(c->argv[1]->ptr);
1424  serverAssertWithInfo(c,NULL,script != NULL);
1425  rewriteClientCommandArgument(c,0,
1426  resetRefCount(createStringObject("EVAL",4)));
1427  rewriteClientCommandArgument(c,1,script);
1428  forceCommandPropagation(c,PROPAGATE_REPL|PROPAGATE_AOF);
1429  }
1430  }

1612 /* Start a debugging session before calling EVAL implementation.
1613  * The techique we use is to capture the client socket file descriptor,
1614  * in order to perform direct I/O with it from within Lua hooks. This
1615  * way we don't have to re-enter Redis in order to handle I/O.
1616  *
1617  * The function returns 1 if the caller should proceed to call EVAL,
1618  * and 0 if instead the caller should abort the operation (this happens
1619  * for the parent in a forked session, since it's up to the children
1620  * to continue, or when fork returned an error).
1621  *
1622  * The caller should call ldbEndSession() only if ldbStartSession()
1623  * returned 1. */
1624 int ldbStartSession(client *c) {
1625  ldb.forked = (c->flags & CLIENT_LUA_DEBUG_SYNC) == 0;
1626  if (ldb.forked) {
1627  pid_t cp = fork();
1628  if (cp == -1) {
1629  addReplyError(c,"Fork() failed: can't run EVAL in debugging mode.");
1630  return 0;
1631  } else if (cp == 0) {
1632  /* Child. Let's ignore important signals handled by the parent. */
1633  struct sigaction act;
1634  sigemptyset(&act.sa_mask);
1635  act.sa_flags = 0;
1636  act.sa_handler = SIG_IGN;
1637  sigaction(SIGTERM, &act, NULL);
1638  sigaction(SIGINT, &act, NULL);
1639 
1640  /* Log the creation of the child and close the listening
1641  * socket to make sure if the parent crashes a reset is sent
1642  * to the clients. */
1643  serverLog(LL_WARNING,"Redis forked for debugging eval");
1644  closeListeningSockets(0);
1645  } else {
1646  /* Parent */
1647  listAddNodeTail(ldb.children,(void*)(unsigned long)cp);
1648  freeClientAsync(c); /* Close the client in the parent side. */
1649  return 0;
1650  }
1651  } else {
1652  serverLog(LL_WARNING,
1653  "Redis synchronous debugging eval session started");
1654  }
1655 
1656  /* Setup our debugging session. */
1657  anetBlock(NULL,ldb.fd);
1658  anetSendTimeout(NULL,ldb.fd,5000);
1659  ldb.active = 1;
1660 
1661  /* First argument of EVAL is the script itself. We split it into different
1662  * lines since this is the way the debugger accesses the source code. */
1663  sds srcstring = sdsdup(c->argv[1]->ptr);
1664  size_t srclen = sdslen(srcstring);
1665  while(srclen && (srcstring[srclen-1] == '\n' ||
1666  srcstring[srclen-1] == '\r'))
1667  {
1668  srcstring[--srclen] = '\0';
1669  }
1670  sdssetlen(srcstring,srclen);
1671  ldb.src = sdssplitlen(srcstring,sdslen(srcstring),"\n",1,&ldb.lines);
1672  sdsfree(srcstring);
1673  return 1;
1674 }

1877 /* Append an human readable representation of the Lua value at position 'idx'
1878  * on the stack of the 'lua' state, to the SDS string passed as argument.
1879  * The new SDS string with the represented value attached is returned.
1880  * Used in order to implement ldbLogStackValue().
1881  *
1882  * The element is not automatically removed from the stack, nor it is
1883  * converted to a different type. */
1884 #define LDB_MAX_VALUES_DEPTH (LUA_MINSTACK/2)
1885 sds ldbCatStackValueRec(sds s, lua_State *lua, int idx, int level) {
1886  int t = lua_type(lua,idx);
1887 
1888  if (level++ == LDB_MAX_VALUES_DEPTH)
1889  return sdscat(s,"<max recursion level reached! Nested table?>");
1890 
1891  switch(t) {
1892  case LUA_TSTRING:
1893  {
1894  size_t strl;
1895  char *strp = (char*)lua_tolstring(lua,idx,&strl);
1896  s = sdscatrepr(s,strp,strl);
1897  }
1898  break;
1899  case LUA_TBOOLEAN:
1900  s = sdscat(s,lua_toboolean(lua,idx) ? "true" : "false");
1901  break;
1902  case LUA_TNUMBER:
1903  s = sdscatprintf(s,"%g",(double)lua_tonumber(lua,idx));
1904  break;
1905  case LUA_TNIL:
1906  s = sdscatlen(s,"nil",3);
1907  break;
1908  case LUA_TTABLE:
1909  {
1910  int expected_index = 1; /* First index we expect in an array. */
1911  int is_array = 1; /* Will be set to null if check fails. */
1912  /* Note: we create two representations at the same time, one
1913  * assuming the table is an array, one assuming it is not. At the
1914  * end we know what is true and select the right one. */
1915  sds repr1 = sdsempty();
1916  sds repr2 = sdsempty();
1917  lua_pushnil(lua); /* The first key to start the iteration is nil. */
1918  while (lua_next(lua,idx-1)) {
1919  /* Test if so far the table looks like an array. */
1920  if (is_array &&
1921  (lua_type(lua,-2) != LUA_TNUMBER ||
1922  lua_tonumber(lua,-2) != expected_index)) is_array = 0;
1923  /* Stack now: table, key, value */
1924  /* Array repr. */
1925  repr1 = ldbCatStackValueRec(repr1,lua,-1,level);
1926  repr1 = sdscatlen(repr1,"; ",2);
1927  /* Full repr. */
1928  repr2 = sdscatlen(repr2,"[",1);
1929  repr2 = ldbCatStackValueRec(repr2,lua,-2,level);
1930  repr2 = sdscatlen(repr2,"]=",2);
1931  repr2 = ldbCatStackValueRec(repr2,lua,-1,level);
1932  repr2 = sdscatlen(repr2,"; ",2);
1933  lua_pop(lua,1); /* Stack: table, key. Ready for next iteration. */
1934  expected_index++;
1935  }
1936  /* Strip the last " ;" from both the representations. */
1937  if (sdslen(repr1)) sdsrange(repr1,0,-3);
1938  if (sdslen(repr2)) sdsrange(repr2,0,-3);
1939  /* Select the right one and discard the other. */
1940  s = sdscatlen(s,"{",1);
1941  s = sdscatsds(s,is_array ? repr1 : repr2);
1942  s = sdscatlen(s,"}",1);
1943  sdsfree(repr1);
1944  sdsfree(repr2);
1945  }
1946  break;
1947  case LUA_TFUNCTION:
1948  case LUA_TUSERDATA:
1949  case LUA_TTHREAD:
1950  case LUA_TLIGHTUSERDATA:
1951  {
1952  const void *p = lua_topointer(lua,idx);
1953  char *typename = "unknown";
1954  if (t == LUA_TFUNCTION) typename = "function";
1955  else if (t == LUA_TUSERDATA) typename = "userdata";
1956  else if (t == LUA_TTHREAD) typename = "thread";
1957  else if (t == LUA_TLIGHTUSERDATA) typename = "light-userdata";
1958  s = sdscatprintf(s,"\"%s@%p\"",typename,p);
1959  }
1960  break;
1961  default:
1962  s = sdscat(s,"\"<unknown-lua-type>\"");
1963  break;
1964  }
1965  return s;
1966 }
