171 /* Return the name of the dotfile for the specified 'dotfilename'.
172  * Normally it just concatenates user $HOME to the file specified
173  * in 'dotfilename'. However if the environment varialbe 'envoverride'
174  * is set, its value is taken as the path.

176  * The function returns NULL (if the file is /dev/null or cannot be
177  * obtained for some error), or an SDS string that must be freed by
178  * the user. */
179 static sds getDotfilePath(char *envoverride, char *dotfilename) {
180  char *path = NULL;
181  sds dotPath = NULL;

184  path = getenv(envoverride);

196  dotPath = sdscatprintf(sdsempty(), "%s/%s", home, dotfilename);

202 /* URL-style percent decoding. */
203 #define isHexChar(c) (isdigit(c) || (c >= 'a' && c <= 'f'))
204 #define decodeHexChar(c) (isdigit(c) ? c - '0' : c - 'a' + 10)
205 #define decodeHex(h, l) ((decodeHexChar(h) << 4) + decodeHexChar(l))
206 
207 static sds percentDecode(const char *pe, size_t len) {
208  const char *end = pe + len;
209  sds ret = sdsempty();
210  const char *curr = pe;
211 
212  while (curr < end) {
213  if (*curr == '%') {
214  if ((end - curr) < 2) {
215  fprintf(stderr, "Incomplete URI encoding\n");
216  exit(1);
217  }
218 
219  char h = tolower(*(++curr));
220  char l = tolower(*(++curr));
221  if (!isHexChar(h) || !isHexChar(l)) {
222  fprintf(stderr, "Illegal character in URI encoding\n");
223  exit(1);
224  }
225  char c = decodeHex(h, l);
226  ret = sdscatlen(ret, &c, 1);
227  curr++;
228  } else {
229  ret = sdscatlen(ret, curr++, 1);
230  }
231  }
232 
233  return ret;
234 }

236 /* Parse a URI and extract the server connection information.
237  * URI scheme is based on the the provisional specification[1] excluding support
238  * for query parameters. Valid URIs are:
239  * scheme: "redis://"
240  * authority: [<username> ":"] <password> "@"] [<hostname> [":" <port>]]
241  * path: ["/" [<db>]]
242  *
243  * [1]: https://www.iana.org/assignments/uri-schemes/prov/redis */
244 static void parseRedisUri(const char *uri) {
245 
246  const char *scheme = "redis://";
247  const char *curr = uri;
248  const char *end = uri + strlen(uri);
249  const char *userinfo, *username, *port, *host, *path;
250 
251  /* URI must start with a valid scheme. */
252  if (strncasecmp(scheme, curr, strlen(scheme))) {
253  fprintf(stderr,"Invalid URI scheme\n");
254  exit(1);
255  }
256  curr += strlen(scheme);
257  if (curr == end) return;
258 
259  /* Extract user info. */
260  if ((userinfo = strchr(curr,'@'))) {
261  if ((username = strchr(curr, ':')) && username < userinfo) {
262  /* If provided, username is ignored. */
263  curr = username + 1;
264  }
265 
266  config.auth = percentDecode(curr, userinfo - curr);
267  curr = userinfo + 1;
268  }
269  if (curr == end) return;
270 
271  /* Extract host and port. */
272  path = strchr(curr, '/');
273  if (*curr != '/') {
274  host = path ? path - 1 : end;
275  if ((port = strchr(curr, ':'))) {
276  config.hostport = atoi(port + 1);
277  host = port - 1;
278  }
279  config.hostip = sdsnewlen(curr, host - curr + 1);
280  }
281  curr = path ? path + 1 : end;
282  if (curr == end) return;
283 
284  /* Extract database number. */
285  config.dbnum = atoi(curr);
286 }

350 /* cliInitHelp() setups the helpEntries array with the command and group
351  * names from the help.h file. However the Redis instance we are connecting
352  * to may support more commands, so this function integrates the previous
353  * entries with additional entries obtained using the COMMAND command
354  * available in recent versions of Redis. */
355 static void cliIntegrateHelp(void) {
356  if (cliConnect(0) == REDIS_ERR) return;
357 
358  redisReply *reply = redisCommand(context, "COMMAND");
359  if(reply == NULL || reply->type != REDIS_REPLY_ARRAY) return;
360 
361  /* Scan the array reported by COMMAND and fill only the entries that
362  * don't already match what we have. */
363  for (size_t j = 0; j < reply->elements; j++) {
364  redisReply *entry = reply->element[j];
365  if (entry->type != REDIS_REPLY_ARRAY || entry->elements < 4 ||
366  entry->element[0]->type != REDIS_REPLY_STRING ||
367  entry->element[1]->type != REDIS_REPLY_INTEGER ||
368  entry->element[3]->type != REDIS_REPLY_INTEGER) return;
369  char *cmdname = entry->element[0]->str;
370  int i;
371 
372  for (i = 0; i < helpEntriesLen; i++) {
373  helpEntry *he = helpEntries+i;
374  if (!strcasecmp(he->argv[0],cmdname))
375  break;
376  }
377  if (i != helpEntriesLen) continue;
378 
379  helpEntriesLen++;
380  helpEntries = zrealloc(helpEntries,sizeof(helpEntry)*helpEntriesLen);
381  helpEntry *new = helpEntries+(helpEntriesLen-1);
382 
383  new->argc = 1;
384  new->argv = zmalloc(sizeof(sds));
385  new->argv[0] = sdsnew(cmdname);
386  new->full = new->argv[0];
387  new->type = CLI_HELP_COMMAND;
388  sdstoupper(new->argv[0]);
389 
390  struct commandHelp *ch = zmalloc(sizeof(*ch));
391  ch->name = new->argv[0];
392  ch->params = sdsempty();
393  int args = llabs(entry->element[1]->integer);
394  if (entry->element[3]->integer == 1) {
395  ch->params = sdscat(ch->params,"key ");
396  args--;
397  }
398  while(args--) ch->params = sdscat(ch->params,"arg ");
399  if (entry->element[1]->integer < 0)
400  ch->params = sdscat(ch->params,"...options...");
401  ch->summary = "Help not available";
402  ch->group = 0;
403  ch->since = "not known";
404  new->org = ch;
405  }
406  freeReplyObject(reply);
407 }

513 /* Linenoise hints callback. */
514 static char *hintsCallback(const char *buf, int *color, int *bold) {
515  if (!pref.hints) return NULL;
516 
517  int i, argc, buflen = strlen(buf);
518  sds *argv = sdssplitargs(buf,&argc);
519  int endspace = buflen && isspace(buf[buflen-1]);
520 
521  /* Check if the argument list is empty and return ASAP. */
522  if (argc == 0) {
523  sdsfreesplitres(argv,argc);
524  return NULL;
525  }
526 
527  for (i = 0; i < helpEntriesLen; i++) {
528  if (!(helpEntries[i].type & CLI_HELP_COMMAND)) continue;
529 
530  if (strcasecmp(argv[0],helpEntries[i].full) == 0)
531  {
532  *color = 90;
533  *bold = 0;
534  sds hint = sdsnew(helpEntries[i].org->params);
535 
536  /* Remove arguments from the returned hint to show only the
537  * ones the user did not yet typed. */
538  int toremove = argc-1;
539  while(toremove > 0 && sdslen(hint)) {
540  if (hint[0] == '[') break;
541  if (hint[0] == ' ') toremove--;
542  sdsrange(hint,1,-1);
543  }
544 
545  /* Add an initial space if needed. */
546  if (!endspace) {
547  sds newhint = sdsnewlen(" ",1);
548  newhint = sdscatsds(newhint,hint);
549  sdsfree(hint);
550  hint = newhint;
551  }
552 
553  sdsfreesplitres(argv,argc);
554  return hint;
555  }
556  }
557  sdsfreesplitres(argv,argc);
558  return NULL;
559 }

642 static sds cliFormatReplyTTY(redisReply *r, char *prefix) {
643  sds out = sdsempty();
644  switch (r->type) {
645  case REDIS_REPLY_ERROR:
646  out = sdscatprintf(out,"(error) %s\n", r->str);
647  break;
648  case REDIS_REPLY_STATUS:
649  out = sdscat(out,r->str);
650  out = sdscat(out,"\n");
651  break;
652  case REDIS_REPLY_INTEGER:
653  out = sdscatprintf(out,"(integer) %lld\n",r->integer);
654  break;
655  case REDIS_REPLY_STRING:
656  /* If you are producing output for the standard output we want
657  * a more interesting output with quoted characters and so forth */
658  out = sdscatrepr(out,r->str,r->len);
659  out = sdscat(out,"\n");
660  break;
661  case REDIS_REPLY_NIL:
662  out = sdscat(out,"(nil)\n");
663  break;
664  case REDIS_REPLY_ARRAY:
665  if (r->elements == 0) {
666  out = sdscat(out,"(empty list or set)\n");
667  } else {
668  unsigned int i, idxlen = 0;
669  char _prefixlen[16];
670  char _prefixfmt[16];
671  sds _prefix;
672  sds tmp;
673 
674  /* Calculate chars needed to represent the largest index */
675  i = r->elements;
676  do {
677  idxlen++;
678  i /= 10;
679  } while(i);
680 
681  /* Prefix for nested multi bulks should grow with idxlen+2 spaces */
682  memset(_prefixlen,' ',idxlen+2);
683  _prefixlen[idxlen+2] = '\0';
684  _prefix = sdscat(sdsnew(prefix),_prefixlen);
685 
686  /* Setup prefix format for every entry */
687  snprintf(_prefixfmt,sizeof(_prefixfmt),"%%s%%%ud) ",idxlen);
688 
689  for (i = 0; i < r->elements; i++) {
690  /* Don't use the prefix for the first element, as the parent
691  * caller already prepended the index number. */
692  out = sdscatprintf(out,_prefixfmt,i == 0 ? "" : prefix,i+1);
693 
694  /* Format the multi bulk entry */
695  tmp = cliFormatReplyTTY(r->element[i],_prefix);
696  out = sdscatlen(out,tmp,sdslen(tmp));
697  sdsfree(tmp);
698  }
699  sdsfree(_prefix);
700  }
701  break;
702  default:
703  fprintf(stderr,"Unknown reply type: %d\n", r->type);
704  exit(1);
705  }
706  return out;

840 static int cliReadReply(int output_raw_strings) {
841  void *_reply;
842  redisReply *reply;
843  sds out = NULL;
844  int output = 1;
845 
846  if (redisGetReply(context,&_reply) != REDIS_OK) {
847  if (config.shutdown) {
848  redisFree(context);
849  context = NULL;
850  return REDIS_OK;
851  }
852  if (config.interactive) {
853  /* Filter cases where we should reconnect */
854  if (context->err == REDIS_ERR_IO &&
855  (errno == ECONNRESET || errno == EPIPE))
856  return REDIS_ERR;
857  if (context->err == REDIS_ERR_EOF)
858  return REDIS_ERR;
859  }
860  cliPrintContextError();
861  exit(1);
862  return REDIS_ERR; /* avoid compiler warning */
863  }
864 
865  reply = (redisReply*)_reply;
866 
867  config.last_cmd_type = reply->type;
868 
869  /* Check if we need to connect to a different node and reissue the
870  * request. */
871  if (config.cluster_mode && reply->type == REDIS_REPLY_ERROR &&
872  (!strncmp(reply->str,"MOVED",5) || !strcmp(reply->str,"ASK")))
873  {
874  char *p = reply->str, *s;
875  int slot;
876 
877  output = 0;
878  /* Comments show the position of the pointer as:
879  *
880  * [S] for pointer 's'
881  * [P] for pointer 'p'
882  */
883  s = strchr(p,' '); /* MOVED[S]3999 127.0.0.1:6381 */
884  p = strchr(s+1,' '); /* MOVED[S]3999[P]127.0.0.1:6381 */
885  *p = '\0';
886  slot = atoi(s+1);
887  s = strrchr(p+1,':'); /* MOVED 3999[P]127.0.0.1[S]6381 */
888  *s = '\0';
889  sdsfree(config.hostip);
890  config.hostip = sdsnew(p+1);
891  config.hostport = atoi(s+1);
892  if (config.interactive)
893  printf("-> Redirected to slot [%d] located at %s:%d\n",
894  slot, config.hostip, config.hostport);
895  config.cluster_reissue_command = 1;
896  cliRefreshPrompt();
897  }
898 
899  if (output) {
900  if (output_raw_strings) {
901  out = cliFormatReplyRaw(reply);
902  } else {
903  if (config.output == OUTPUT_RAW) {
904  out = cliFormatReplyRaw(reply);
905  out = sdscat(out,"\n");
906  } else if (config.output == OUTPUT_STANDARD) {
907  out = cliFormatReplyTTY(reply,"");
908  } else if (config.output == OUTPUT_CSV) {
909  out = cliFormatReplyCSV(reply);
910  out = sdscat(out,"\n");
911  }
912  }
913  fwrite(out,sdslen(out),1,stdout);
914  sdsfree(out);
915  }
916  freeReplyObject(reply);
917  return REDIS_OK;
918 }

962  if (argc == 3 && !strcasecmp(argv[0],"script") &&
963  !strcasecmp(argv[1],"debug"))
964  {

966  config.enable_ldb_on_eval = 1;
967  } else {
968  config.enable_ldb_on_eval = 0;
969  }
970  }
971

974  config.eval_ldb = 1;
975  config.output = OUTPUT_RAW;
976  }

981  argvlen[j] = sdslen(argv[j]);

996  }

1005 
1006  if (cliReadReply(output_raw) != REDIS_OK) {
1007  zfree(argvlen);
1008  return REDIS_ERR;
1009  } else {
1010  /* Store database number when SELECT was successfully executed. */
1011  if (!strcasecmp(command,"select") && argc == 2 && config.last_cmd_type != REDIS_REPLY_ERROR) {
1012  config.dbnum = atoi(argv[1]);

1017  }

1019  fflush(stdout); /* Make it grep friendly */
1020  }
1021 
1022  zfree(argvlen);
1023  return REDIS_OK;
1024 }

1796 /* Sends SYNC and reads the number of bytes in the payload. Used both by
1797  * slaveMode() and getRDB(). */
1798 unsigned long long sendSync(int fd) {
1799  /* To start we need to send the SYNC command and return the payload.
1800  * The hiredis client lib does not understand this part of the protocol
1801  * and we don't want to mess with its buffers, so everything is performed
1802  * using direct low-level I/O. */
1803  char buf[4096], *p;
1804  ssize_t nread;
1805 
1806  /* Send the SYNC command. */
1807  if (write(fd,"SYNC\r\n",6) != 6) {
1808  fprintf(stderr,"Error writing to master\n");
1809  exit(1);
1810  }
1811 
1812  /* Read $<payload>\r\n, making sure to read just up to "\n" */
1813  p = buf;
1814  while(1) {
1815  nread = read(fd,p,1);
1816  if (nread <= 0) {
1817  fprintf(stderr,"Error reading bulk length while SYNCing\n");
1818  exit(1);
1819  }
1820  if (*p == '\n' && p != buf) break;
1821  if (*p != '\n') p++;
1822  }
1823  *p = '\0';
1824  if (buf[0] == '-') {
1825  printf("SYNC with master failed: %s\n", buf);

1936  * time. */
1937  while(!done) {

1939

1964  fprintf(stderr, "Error reading replies from server\n");
1965  exit(1);

1967  if (reply) {

1969  fprintf(stderr,"%s\n", reply->str);
1970  errors++;

1972  reply->len == 20) {
1973  /* Check if this is the reply to our final ECHO
1974  * command. If so everything was received
1975  * from the server. */
1976  if (memcmp(reply->str,magic,20) == 0) {
1977  printf("Last reply received from server.\n");
1978  done = 1;
1979  replies--;
1980  }
1981  }

1984  }
1985  } while(reply);
1986  }
1987

1989  if (mask & AE_WRITABLE) {

2009  if (obuf_len != 0) break; /* Can't accept more data. */

2012  if (obuf_len == 0 && !eof) {

2026  * ECHO command that we can match in the replies
2027  * to make sure everything was read from the server. */

2035  } else if (nread == -1) {
2036  fprintf(stderr, "Error reading from stdin: %s\n",
2037  strerror(errno));
2038  exit(1);
2039  } else {
2040  obuf_len = nread;
2041  obuf_pos = 0;
2042  }
