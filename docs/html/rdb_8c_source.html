52  va_list ap;

64  char *argv[2] = {"",server.rdb_filename};

68  }

108 /* Saves an encoded length. The first two bits in the first byte are used to
109  * hold the encoding type. See the RDB_* definitions for more information
110  * on the types of encoding. */
111 int rdbSaveLen(rio *rdb, uint64_t len) {
112  unsigned char buf[2];
113  size_t nwritten;
114 
115  if (len < (1<<6)) {
116  /* Save a 6 bit len */
117  buf[0] = (len&0xFF)|(RDB_6BITLEN<<6);
118  if (rdbWriteRaw(rdb,buf,1) == -1) return -1;
119  nwritten = 1;
120  } else if (len < (1<<14)) {
121  /* Save a 14 bit len */
122  buf[0] = ((len>>8)&0xFF)|(RDB_14BITLEN<<6);
123  buf[1] = len&0xFF;
124  if (rdbWriteRaw(rdb,buf,2) == -1) return -1;
125  nwritten = 2;
126  } else if (len <= UINT32_MAX) {
127  /* Save a 32 bit len */
128  buf[0] = RDB_32BITLEN;
129  if (rdbWriteRaw(rdb,buf,1) == -1) return -1;
130  uint32_t len32 = htonl(len);
131  if (rdbWriteRaw(rdb,&len32,4) == -1) return -1;
132  nwritten = 1+4;
133  } else {
134  /* Save a 64 bit len */
135  buf[0] = RDB_64BITLEN;
136  if (rdbWriteRaw(rdb,buf,1) == -1) return -1;
137  len = htonu64(len);
138  if (rdbWriteRaw(rdb,&len,8) == -1) return -1;
139  nwritten = 1+8;
140  }
141  return nwritten;
142 }

145 /* Load an encoded length. If the loaded length is a normal length as stored
146  * with rdbSaveLen(), the read length is set to '*lenptr'. If instead the
147  * loaded length describes a special encoding that follows, then '*isencoded'
148  * is set to 1 and the encoding format is stored at '*lenptr'.
149  *
150  * See the RDB_ENC_* definitions in rdb.h for more information on special
151  * encodings.
152  *
153  * The function returns -1 on error, 0 on success. */
154 int rdbLoadLenByRef(rio *rdb, int *isencoded, uint64_t *lenptr) {
155  unsigned char buf[2];
156  int type;
157 
158  if (isencoded) *isencoded = 0;
159  if (rioRead(rdb,buf,1) == 0) return -1;
160  type = (buf[0]&0xC0)>>6;
161  if (type == RDB_ENCVAL) {
162  /* Read a 6 bit encoding type. */
163  if (isencoded) *isencoded = 1;
164  *lenptr = buf[0]&0x3F;
165  } else if (type == RDB_6BITLEN) {
166  /* Read a 6 bit len. */
167  *lenptr = buf[0]&0x3F;
168  } else if (type == RDB_14BITLEN) {
169  /* Read a 14 bit len. */
170  if (rioRead(rdb,buf+1,1) == 0) return -1;
171  *lenptr = ((buf[0]&0x3F)<<8)|buf[1];
172  } else if (buf[0] == RDB_32BITLEN) {
173  /* Read a 32 bit len. */
174  uint32_t len;
175  if (rioRead(rdb,&len,4) == 0) return -1;
176  *lenptr = ntohl(len);
177  } else if (buf[0] == RDB_64BITLEN) {
178  /* Read a 64 bit len. */
179  uint64_t len;
180  if (rioRead(rdb,&len,8) == 0) return -1;
181  *lenptr = ntohu64(len);
182  } else {
183  rdbExitReportCorruptRDB(
184  "Unknown length encoding %d in rdbLoadLen()",type);
185  return -1; /* Never reached. */
186  }
187  return 0;
188 }

201 /* Encodes the "value" argument as integer when it fits in the supported ranges
202  * for encoded types. If the function successfully encodes the integer, the
203  * representation is stored in the buffer pointer to by "enc" and the string
204  * length is returned. Otherwise 0 is returned. */
205 int rdbEncodeInteger(long long value, unsigned char *enc) {
206  if (value >= -(1<<7) && value <= (1<<7)-1) {
207  enc[0] = (RDB_ENCVAL<<6)|RDB_ENC_INT8;
208  enc[1] = value&0xFF;
209  return 2;
210  } else if (value >= -(1<<15) && value <= (1<<15)-1) {
211  enc[0] = (RDB_ENCVAL<<6)|RDB_ENC_INT16;
212  enc[1] = value&0xFF;
213  enc[2] = (value>>8)&0xFF;
214  return 3;
215  } else if (value >= -((long long)1<<31) && value <= ((long long)1<<31)-1) {
216  enc[0] = (RDB_ENCVAL<<6)|RDB_ENC_INT32;
217  enc[1] = value&0xFF;
218  enc[2] = (value>>8)&0xFF;
219  enc[3] = (value>>16)&0xFF;
220  enc[4] = (value>>24)&0xFF;
221  return 5;
222  } else {
223  return 0;
224  }
225 }

227 /* Loads an integer-encoded object with the specified encoding type "enctype".
228  * The returned value changes according to the flags, see
229  * rdbGenerincLoadStringObject() for more info. */
230 void *rdbLoadIntegerObject(rio *rdb, int enctype, int flags, size_t *lenptr) {
231  int plain = flags & RDB_LOAD_PLAIN;
232  int sds = flags & RDB_LOAD_SDS;
233  int encode = flags & RDB_LOAD_ENC;
234  unsigned char enc[4];
235  long long val;
236 
237  if (enctype == RDB_ENC_INT8) {
238  if (rioRead(rdb,enc,1) == 0) return NULL;
239  val = (signed char)enc[0];
240  } else if (enctype == RDB_ENC_INT16) {
241  uint16_t v;
242  if (rioRead(rdb,enc,2) == 0) return NULL;
243  v = enc[0]|(enc[1]<<8);
244  val = (int16_t)v;
245  } else if (enctype == RDB_ENC_INT32) {
246  uint32_t v;
247  if (rioRead(rdb,enc,4) == 0) return NULL;
248  v = enc[0]|(enc[1]<<8)|(enc[2]<<16)|(enc[3]<<24);
249  val = (int32_t)v;
250  } else {
251  val = 0; /* anti-warning */
252  rdbExitReportCorruptRDB("Unknown RDB integer encoding type %d",enctype);
253  }
254  if (plain || sds) {
255  char buf[LONG_STR_SIZE], *p;
256  int len = ll2string(buf,sizeof(buf),val);
257  if (lenptr) *lenptr = len;
258  p = plain ? zmalloc(len) : sdsnewlen(NULL,len);
259  memcpy(p,buf,len);
260  return p;
261  } else if (encode) {
262  return createStringObjectFromLongLong(val);
263  } else {
264  return createObject(OBJ_STRING,sdsfromlonglong(val));
265  }
266 }

268 /* String objects in the form "2391" "-100" without any space and with a
269  * range of values that can fit in an 8, 16 or 32 bit signed value can be
270  * encoded as integers to save space */
271 int rdbTryIntegerEncoding(char *s, size_t len, unsigned char *enc) {
272  long long value;
273  char *endptr, buf[32];
274 
275  /* Check if it's possible to encode this value as a number */
276  value = strtoll(s, &endptr, 10);
277  if (endptr[0] != '\0') return 0;
278  ll2string(buf,32,value);
279 
280  /* If the number converted back into a string is not identical
281  * then it's not possible to encode the string as integer */
282  if (strlen(buf) != len || memcmp(buf,s,len)) return 0;
283 
284  return rdbEncodeInteger(value,enc);
285 }

287 ssize_t rdbSaveLzfBlob(rio *rdb, void *data, size_t compress_len,
288  size_t original_len) {
289  unsigned char byte;
290  ssize_t n, nwritten = 0;
291 
292  /* Data compressed! Let's save it on disk */
293  byte = (RDB_ENCVAL<<6)|RDB_ENC_LZF;
294  if ((n = rdbWriteRaw(rdb,&byte,1)) == -1) goto writeerr;
295  nwritten += n;
296 
297  if ((n = rdbSaveLen(rdb,compress_len)) == -1) goto writeerr;
298  nwritten += n;
299 
300  if ((n = rdbSaveLen(rdb,original_len)) == -1) goto writeerr;
301  nwritten += n;
302 
303  if ((n = rdbWriteRaw(rdb,data,compress_len)) == -1) goto writeerr;
304  nwritten += n;
305 
306  return nwritten;
307 
308 writeerr:
309  return -1;
310 }

312 ssize_t rdbSaveLzfStringObject(rio *rdb, unsigned char *s, size_t len) {
313  size_t comprlen, outlen;
314  void *out;
315 
316  /* We require at least four bytes compression for this to be worth it */
317  if (len <= 4) return 0;
318  outlen = len-4;
319  if ((out = zmalloc(outlen+1)) == NULL) return 0;
320  comprlen = lzf_compress(s, len, out, outlen);
321  if (comprlen == 0) {
322  zfree(out);
323  return 0;
324  }
325  ssize_t nwritten = rdbSaveLzfBlob(rdb, out, comprlen, len);
326  zfree(out);
327  return nwritten;
328 }

330 /* Load an LZF compressed string in RDB format. The returned value
331  * changes according to 'flags'. For more info check the
332  * rdbGenericLoadStringObject() function. */
333 void *rdbLoadLzfStringObject(rio *rdb, int flags, size_t *lenptr) {
334  int plain = flags & RDB_LOAD_PLAIN;
335  int sds = flags & RDB_LOAD_SDS;
336  uint64_t len, clen;
337  unsigned char *c = NULL;
338  char *val = NULL;
339 
340  if ((clen = rdbLoadLen(rdb,NULL)) == RDB_LENERR) return NULL;
341  if ((len = rdbLoadLen(rdb,NULL)) == RDB_LENERR) return NULL;
342  if ((c = zmalloc(clen)) == NULL) goto err;
343 
344  /* Allocate our target according to the uncompressed size. */
345  if (plain) {
346  val = zmalloc(len);
347  if (lenptr) *lenptr = len;
348  } else {
349  val = sdsnewlen(NULL,len);
350  }
351 
352  /* Load the compressed representation and uncompress it to target. */
353  if (rioRead(rdb,c,clen) == 0) goto err;
354  if (lzf_decompress(c,clen,val,len) == 0) {
355  if (rdbCheckMode) rdbCheckSetError("Invalid LZF compressed string");
356  goto err;
357  }
358  zfree(c);
359 
360  if (plain || sds) {
361  return val;
362  } else {
363  return createObject(OBJ_STRING,val);
364  }
365 err:
366  zfree(c);
367  if (plain)
368  zfree(val);
369  else
370  sdsfree(val);
371  return NULL;
372 }

374 /* Save a string object as [len][data] on disk. If the object is a string
375  * representation of an integer value we try to save it in a special form */
376 ssize_t rdbSaveRawString(rio *rdb, unsigned char *s, size_t len) {
377  int enclen;
378  ssize_t n, nwritten = 0;
379 
380  /* Try integer encoding */
381  if (len <= 11) {
382  unsigned char buf[5];
383  if ((enclen = rdbTryIntegerEncoding((char*)s,len,buf)) > 0) {
384  if (rdbWriteRaw(rdb,buf,enclen) == -1) return -1;
385  return enclen;
386  }
387  }
388 
389  /* Try LZF compression - under 20 bytes it's unable to compress even
390  * aaaaaaaaaaaaaaaaaa so skip it */
391  if (server.rdb_compression && len > 20) {
392  n = rdbSaveLzfStringObject(rdb,s,len);
393  if (n == -1) return -1;
394  if (n > 0) return n;
395  /* Return value of 0 means data can't be compressed, save the old way */
396  }
397 
398  /* Store verbatim */
399  if ((n = rdbSaveLen(rdb,len)) == -1) return -1;
400  nwritten += n;
401  if (len > 0) {
402  if (rdbWriteRaw(rdb,s,len) == -1) return -1;
403  nwritten += len;
404  }
405  return nwritten;
406 }

409 ssize_t rdbSaveLongLongAsStringObject(rio *rdb, long long value) {
410  unsigned char buf[32];
411  ssize_t n, nwritten = 0;
412  int enclen = rdbEncodeInteger(value,buf);

421  if ((n = rdbWriteRaw(rdb,buf,enclen)) == -1) return -1;
422  nwritten += n;
423  }
424  return nwritten;

439 /* Load a string object from an RDB file according to flags:
440  *
441  * RDB_LOAD_NONE (no flags): load an RDB object, unencoded.
442  * RDB_LOAD_ENC: If the returned type is a Redis object, try to
443  * encode it in a special way to be more memory
444  * efficient. When this flag is passed the function
445  * no longer guarantees that obj->ptr is an SDS string.
446  * RDB_LOAD_PLAIN: Return a plain string allocated with zmalloc()
447  * instead of a Redis object with an sds in it.
448  * RDB_LOAD_SDS: Return an SDS string instead of a Redis object.
449  *
450  * On I/O error NULL is returned.
451  */
452 void *rdbGenericLoadStringObject(rio *rdb, int flags, size_t *lenptr) {
453  int encode = flags & RDB_LOAD_ENC;
454  int plain = flags & RDB_LOAD_PLAIN;
455  int sds = flags & RDB_LOAD_SDS;
456  int isencoded;
457  uint64_t len;
458 
459  len = rdbLoadLen(rdb,&isencoded);
460  if (isencoded) {
461  switch(len) {
462  case RDB_ENC_INT8:
463  case RDB_ENC_INT16:
464  case RDB_ENC_INT32:
465  return rdbLoadIntegerObject(rdb,len,flags,lenptr);
466  case RDB_ENC_LZF:
467  return rdbLoadLzfStringObject(rdb,flags,lenptr);
468  default:
469  rdbExitReportCorruptRDB("Unknown RDB string encoding type %d",len);
470  }
471  }
472 
473  if (len == RDB_LENERR) return NULL;
474  if (plain || sds) {
475  void *buf = plain ? zmalloc(len) : sdsnewlen(NULL,len);
476  if (lenptr) *lenptr = len;
477  if (len && rioRead(rdb,buf,len) == 0) {
478  if (plain)
479  zfree(buf);
480  else
481  sdsfree(buf);
482  return NULL;
483  }
484  return buf;
485  } else {
486  robj *o = encode ? createStringObject(NULL,len) :
487  createRawStringObject(NULL,len);
488  if (len && rioRead(rdb,o->ptr,len) == 0) {
489  decrRefCount(o);
490  return NULL;
491  }
492  return o;
493  }
494 }

504 /* Save a double value. Doubles are saved as strings prefixed by an unsigned
505  * 8 bit integer specifying the length of the representation.
506  * This 8 bit integer has special values in order to specify the following
507  * conditions:
508  * 253: not a number
509  * 254: + inf
510  * 255: - inf
511  */
512 int rdbSaveDoubleValue(rio *rdb, double val) {
513  unsigned char buf[128];
514  int len;
515 
516  if (isnan(val)) {
517  buf[0] = 253;
518  len = 1;
519  } else if (!isfinite(val)) {
520  len = 1;
521  buf[0] = (val < 0) ? 255 : 254;
522  } else {
523 #if (DBL_MANT_DIG >= 52) && (LLONG_MAX == 0x7fffffffffffffffLL)
524  /* Check if the float is in a safe range to be casted into a
525  * long long. We are assuming that long long is 64 bit here.
526  * Also we are assuming that there are no implementations around where
527  * double has precision < 52 bit.
528  *
529  * Under this assumptions we test if a double is inside an interval
530  * where casting to long long is safe. Then using two castings we
531  * make sure the decimal part is zero. If all this is true we use
532  * integer printing function that is much faster. */
533  double min = -4503599627370495; /* (2^52)-1 */
534  double max = 4503599627370496; /* -(2^52) */
535  if (val > min && val < max && val == ((double)((long long)val)))
536  ll2string((char*)buf+1,sizeof(buf)-1,(long long)val);
537  else
538 #endif
539  snprintf((char*)buf+1,sizeof(buf)-1,"%.17g",val);
540  buf[0] = strlen((char*)buf+1);
541  len = buf[0]+1;
542  }
543  return rdbWriteRaw(rdb,buf,len);
544 }

619  case OBJ_HASH:

645 /* Save a Redis object. Returns -1 on error, number of bytes written on success. */
646 ssize_t rdbSaveObject(rio *rdb, robj *o) {
647  ssize_t n = 0, nwritten = 0;
648 
649  if (o->type == OBJ_STRING) {
650  /* Save a string value */
651  if ((n = rdbSaveStringObject(rdb,o)) == -1) return -1;
652  nwritten += n;
653  } else if (o->type == OBJ_LIST) {
654  /* Save a list value */
655  if (o->encoding == OBJ_ENCODING_QUICKLIST) {
656  quicklist *ql = o->ptr;
657  quicklistNode *node = ql->head;
658 
659  if ((n = rdbSaveLen(rdb,ql->len)) == -1) return -1;
660  nwritten += n;
661 
662  while(node) {
663  if (quicklistNodeIsCompressed(node)) {
664  void *data;
665  size_t compress_len = quicklistGetLzf(node, &data);
666  if ((n = rdbSaveLzfBlob(rdb,data,compress_len,node->sz)) == -1) return -1;
667  nwritten += n;
668  } else {
669  if ((n = rdbSaveRawString(rdb,node->zl,node->sz)) == -1) return -1;
670  nwritten += n;
671  }
672  node = node->next;
673  }
674  } else {
675  serverPanic("Unknown list encoding");
676  }
677  } else if (o->type == OBJ_SET) {
678  /* Save a set value */
679  if (o->encoding == OBJ_ENCODING_HT) {
680  dict *set = o->ptr;
681  dictIterator *di = dictGetIterator(set);
682  dictEntry *de;
683 
684  if ((n = rdbSaveLen(rdb,dictSize(set))) == -1) return -1;
685  nwritten += n;
686 
687  while((de = dictNext(di)) != NULL) {
688  sds ele = dictGetKey(de);
689  if ((n = rdbSaveRawString(rdb,(unsigned char*)ele,sdslen(ele)))
690  == -1) return -1;
691  nwritten += n;
692  }
693  dictReleaseIterator(di);
694  } else if (o->encoding == OBJ_ENCODING_INTSET) {
695  size_t l = intsetBlobLen((intset*)o->ptr);
696 
697  if ((n = rdbSaveRawString(rdb,o->ptr,l)) == -1) return -1;
698  nwritten += n;
699  } else {
700  serverPanic("Unknown set encoding");
701  }
702  } else if (o->type == OBJ_ZSET) {
703  /* Save a sorted set value */
704  if (o->encoding == OBJ_ENCODING_ZIPLIST) {
705  size_t l = ziplistBlobLen((unsigned char*)o->ptr);
706 
707  if ((n = rdbSaveRawString(rdb,o->ptr,l)) == -1) return -1;
708  nwritten += n;
709  } else if (o->encoding == OBJ_ENCODING_SKIPLIST) {
710  zset *zs = o->ptr;
711  zskiplist *zsl = zs->zsl;
712 
713  if ((n = rdbSaveLen(rdb,zsl->length)) == -1) return -1;
714  nwritten += n;
715 
716  /* We save the skiplist elements from the greatest to the smallest
717  * (that's trivial since the elements are already ordered in the
718  * skiplist): this improves the load process, since the next loaded
719  * element will always be the smaller, so adding to the skiplist
720  * will always immediately stop at the head, making the insertion
721  * O(1) instead of O(log(N)). */
722  zskiplistNode *zn = zsl->tail;
723  while (zn != NULL) {
724  if ((n = rdbSaveRawString(rdb,
725  (unsigned char*)zn->ele,sdslen(zn->ele))) == -1)
726  {
727  return -1;
728  }
729  nwritten += n;
730  if ((n = rdbSaveBinaryDoubleValue(rdb,zn->score)) == -1)
731  return -1;
732  nwritten += n;
733  zn = zn->backward;
734  }
735  } else {
736  serverPanic("Unknown sorted set encoding");
737  }
738  } else if (o->type == OBJ_HASH) {
739  /* Save a hash value */
740  if (o->encoding == OBJ_ENCODING_ZIPLIST) {
741  size_t l = ziplistBlobLen((unsigned char*)o->ptr);
742 
743  if ((n = rdbSaveRawString(rdb,o->ptr,l)) == -1) return -1;
744  nwritten += n;
745 
746  } else if (o->encoding == OBJ_ENCODING_HT) {
747  dictIterator *di = dictGetIterator(o->ptr);
748  dictEntry *de;
749 
750  if ((n = rdbSaveLen(rdb,dictSize((dict*)o->ptr))) == -1) return -1;
751  nwritten += n;
752 
753  while((de = dictNext(di)) != NULL) {
754  sds field = dictGetKey(de);
755  sds value = dictGetVal(de);
756 
757  if ((n = rdbSaveRawString(rdb,(unsigned char*)field,
758  sdslen(field))) == -1) return -1;
759  nwritten += n;
760  if ((n = rdbSaveRawString(rdb,(unsigned char*)value,
761  sdslen(value))) == -1) return -1;
762  nwritten += n;
763  }
764  dictReleaseIterator(di);
765  } else {
766  serverPanic("Unknown hash encoding");
767  }
768  } else if (o->type == OBJ_STREAM) {
769  /* Store how many listpacks we have inside the radix tree. */
770  stream *s = o->ptr;
771  rax *rax = s->rax;
772  if ((n = rdbSaveLen(rdb,raxSize(rax))) == -1) return -1;
773  nwritten += n;
774 
775  /* Serialize all the listpacks inside the radix tree as they are,
776  * when loading back, we'll use the first entry of each listpack
777  * to insert it back into the radix tree. */
778  raxIterator ri;
779  raxStart(&ri,rax);
780  raxSeek(&ri,"^",NULL,0);
781  while (raxNext(&ri)) {
782  unsigned char *lp = ri.data;
783  size_t lp_bytes = lpBytes(lp);
784  if ((n = rdbSaveRawString(rdb,ri.key,ri.key_len)) == -1) return -1;
785  nwritten += n;
786  if ((n = rdbSaveRawString(rdb,lp,lp_bytes)) == -1) return -1;
787  nwritten += n;
788  }
789  raxStop(&ri);
790 
791  /* Save the number of elements inside the stream. We cannot obtain
792  * this easily later, since our macro nodes should be checked for
793  * number of items: not a great CPU / space tradeoff. */
794  if ((n = rdbSaveLen(rdb,s->length)) == -1) return -1;
795  nwritten += n;
796  /* Save the last entry ID. */
797  if ((n = rdbSaveLen(rdb,s->last_id.ms)) == -1) return -1;
798  nwritten += n;
799  if ((n = rdbSaveLen(rdb,s->last_id.seq)) == -1) return -1;
800  nwritten += n;
801  } else if (o->type == OBJ_MODULE) {
802  /* Save a module-specific value. */
803  RedisModuleIO io;
804  moduleValue *mv = o->ptr;
805  moduleType *mt = mv->type;
806  moduleInitIOContext(io,mt,rdb);
807 
808  /* Write the "module" identifier as prefix, so that we'll be able
809  * to call the right module during loading. */
810  int retval = rdbSaveLen(rdb,mt->id);
811  if (retval == -1) return -1;
812  io.bytes += retval;
813 
814  /* Then write the module-specific representation + EOF marker. */
815  mt->rdb_save(&io,mv->value);
816  retval = rdbSaveLen(rdb,RDB_MODULE_OPCODE_EOF);
817  if (retval == -1) return -1;
818  io.bytes += retval;
819 
820  if (io.ctx) {
821  moduleFreeContext(io.ctx);
822  zfree(io.ctx);
823  }
824  return io.error ? -1 : (ssize_t)io.bytes;
825  } else {
826  serverPanic("Unknown object type");
827  }
828  return nwritten;
829 }

908 /* Produces a dump of the database in RDB format sending it to the specified
909  * Redis I/O channel. On success C_OK is returned, otherwise C_ERR
910  * is returned and part of the output, or all the output, can be
911  * missing because of I/O errors.
912  *
913  * When the function returns C_ERR and if 'error' is not NULL, the
914  * integer pointed by 'error' is set to the value of errno just after the I/O
915  * error. */
916 int rdbSaveRio(rio *rdb, int *error, int flags, rdbSaveInfo *rsi) {
917  dictIterator *di = NULL;
918  dictEntry *de;
919  char magic[10];
920  int j;
921  long long now = mstime();
922  uint64_t cksum;
923  size_t processed = 0;
924 
925  if (server.rdb_checksum)
926  rdb->update_cksum = rioGenericUpdateChecksum;
927  snprintf(magic,sizeof(magic),"REDIS%04d",RDB_VERSION);
928  if (rdbWriteRaw(rdb,magic,9) == -1) goto werr;
929  if (rdbSaveInfoAuxFields(rdb,flags,rsi) == -1) goto werr;
930 
931  for (j = 0; j < server.dbnum; j++) {
932  redisDb *db = server.db+j;
933  dict *d = db->dict;
934  if (dictSize(d) == 0) continue;
935  di = dictGetSafeIterator(d);
936  if (!di) return C_ERR;
937 
938  /* Write the SELECT DB opcode */
939  if (rdbSaveType(rdb,RDB_OPCODE_SELECTDB) == -1) goto werr;
940  if (rdbSaveLen(rdb,j) == -1) goto werr;
941 
942  /* Write the RESIZE DB opcode. We trim the size to UINT32_MAX, which
943  * is currently the largest type we are able to represent in RDB sizes.
944  * However this does not limit the actual size of the DB to load since
945  * these sizes are just hints to resize the hash tables. */
946  uint32_t db_size, expires_size;
947  db_size = (dictSize(db->dict) <= UINT32_MAX) ?
948  dictSize(db->dict) :
949  UINT32_MAX;
950  expires_size = (dictSize(db->expires) <= UINT32_MAX) ?
951  dictSize(db->expires) :
952  UINT32_MAX;
953  if (rdbSaveType(rdb,RDB_OPCODE_RESIZEDB) == -1) goto werr;
954  if (rdbSaveLen(rdb,db_size) == -1) goto werr;
955  if (rdbSaveLen(rdb,expires_size) == -1) goto werr;
956 
957  /* Iterate this DB writing every entry */
958  while((de = dictNext(di)) != NULL) {
959  sds keystr = dictGetKey(de);
960  robj key, *o = dictGetVal(de);
961  long long expire;
962 
963  initStaticStringObject(key,keystr);
964  expire = getExpire(db,&key);
965  if (rdbSaveKeyValuePair(rdb,&key,o,expire,now) == -1) goto werr;
966 
967  /* When this RDB is produced as part of an AOF rewrite, move
968  * accumulated diff from parent to child while rewriting in
969  * order to have a smaller final write. */
970  if (flags & RDB_SAVE_AOF_PREAMBLE &&
971  rdb->processed_bytes > processed+AOF_READ_DIFF_INTERVAL_BYTES)
972  {
973  processed = rdb->processed_bytes;
974  aofReadDiffFromParent();
975  }
976  }
977  dictReleaseIterator(di);
978  }
979  di = NULL; /* So that we don't release it again on error. */
980 
981  /* If we are storing the replication information on disk, persist
982  * the script cache as well: on successful PSYNC after a restart, we need
983  * to be able to process any EVALSHA inside the replication backlog the
984  * master will send us. */
985  if (rsi && dictSize(server.lua_scripts)) {
986  di = dictGetIterator(server.lua_scripts);
987  while((de = dictNext(di)) != NULL) {
988  robj *body = dictGetVal(de);
989  if (rdbSaveAuxField(rdb,"lua",3,body->ptr,sdslen(body->ptr)) == -1)
990  goto werr;
991  }
992  dictReleaseIterator(di);
993  }
994 
995  /* EOF opcode */
996  if (rdbSaveType(rdb,RDB_OPCODE_EOF) == -1) goto werr;
997 
998  /* CRC64 checksum. It will be zero if checksum computation is disabled, the
999  * loading code skips the check in this case. */
1000  cksum = rdb->cksum;
1001  memrev64ifbe(&cksum);
1002  if (rioWrite(rdb,&cksum,8) == 0) goto werr;
1003  return C_OK;
1004 
1005 werr:
1006  if (error) *error = errno;
1007  if (di) dictReleaseIterator(di);
1008  return C_ERR;
1009 }

1011 /* This is just a wrapper to rdbSaveRio() that additionally adds a prefix
1012  * and a suffix to the generated RDB dump. The prefix is:
1013  *
1014  * $EOF:<40 bytes unguessable hex string>\r\n
1015  *
1016  * While the suffix is the 40 bytes hex string we announced in the prefix.
1017  * This way processes receiving the payload can understand when it ends
1018  * without doing any processing of the content. */
1019 int rdbSaveRioWithEOFMark(rio *rdb, int *error, rdbSaveInfo *rsi) {
1020  char eofmark[RDB_EOF_MARK_SIZE];
1021 
1022  getRandomHexChars(eofmark,RDB_EOF_MARK_SIZE);
1023  if (error) *error = 0;
1024  if (rioWrite(rdb,"$EOF:",5) == 0) goto werr;
1025  if (rioWrite(rdb,eofmark,RDB_EOF_MARK_SIZE) == 0) goto werr;
1026  if (rioWrite(rdb,"\r\n",2) == 0) goto werr;
1027  if (rdbSaveRio(rdb,error,RDB_SAVE_NONE,rsi) == C_ERR) goto werr;
1028  if (rioWrite(rdb,eofmark,RDB_EOF_MARK_SIZE) == 0) goto werr;
1029  return C_OK;
1030 
1031 werr: /* Write error. */
1032  /* Set 'error' only if not already set by rdbSaveRio() call. */
1033  if (error && *error == 0) *error = errno;
1034  return C_ERR;
1035 }

1037 /* Save the DB on disk. Return C_ERR on error, C_OK on success. */
1038 int rdbSave(char *filename, rdbSaveInfo *rsi) {
1039  char tmpfile[256];
1040  char cwd[MAXPATHLEN]; /* Current working dir path for error messages. */
1041  FILE *fp;
1042  rio rdb;
1043  int error = 0;
1044 
1045  snprintf(tmpfile,256,"temp-%d.rdb", (int) getpid());
1046  fp = fopen(tmpfile,"w");
1047  if (!fp) {
1048  char *cwdp = getcwd(cwd,MAXPATHLEN);
1049  serverLog(LL_WARNING,
1050  "Failed opening the RDB file %s (in server root dir %s) "
1051  "for saving: %s",
1052  filename,
1053  cwdp ? cwdp : "unknown",
1054  strerror(errno));
1055  return C_ERR;
1056  }
1057 
1058  rioInitWithFile(&rdb,fp);
1059  if (rdbSaveRio(&rdb,&error,RDB_SAVE_NONE,rsi) == C_ERR) {
1060  errno = error;
1061  goto werr;
1062  }
1063 
1064  /* Make sure data will not remain on the OS's output buffers */
1065  if (fflush(fp) == EOF) goto werr;
1066  if (fsync(fileno(fp)) == -1) goto werr;
1067  if (fclose(fp) == EOF) goto werr;
1068 
1069  /* Use RENAME to make sure the DB file is changed atomically only
1070  * if the generate DB file is ok. */
1071  if (rename(tmpfile,filename) == -1) {
1072  char *cwdp = getcwd(cwd,MAXPATHLEN);
1073  serverLog(LL_WARNING,
1074  "Error moving temp DB file %s on the final "
1075  "destination %s (in server root dir %s): %s",
1076  tmpfile,
1077  filename,
1078  cwdp ? cwdp : "unknown",
1079  strerror(errno));
1080  unlink(tmpfile);
1081  return C_ERR;
1082  }
1083 
1084  serverLog(LL_NOTICE,"DB saved on disk");
1085  server.dirty = 0;
1086  server.lastsave = time(NULL);
1087  server.lastbgsave_status = C_OK;
1088  return C_OK;
1089 
1090 werr:
1091  serverLog(LL_WARNING,"Write error saving DB on disk: %s", strerror(errno));
1092  fclose(fp);
1093  unlink(tmpfile);
1094  return C_ERR;
1095 }

1197  * On success a newly allocated object is returned, otherwise NULL. */

1200  uint64_t len;
1201  unsigned int i;
1202 
1203  if (rdbtype == RDB_TYPE_STRING) {
1204  /* Read string value */
1205  if ((o = rdbLoadEncodedStringObject(rdb)) == NULL) return NULL;
1206  o = tryObjectEncoding(o);
1207  } else if (rdbtype == RDB_TYPE_LIST) {
1208  /* Read list value */
1209  if ((len = rdbLoadLen(rdb,NULL)) == RDB_LENERR) return NULL;
1210

1213  server.list_compress_depth);
1214

1217  if ((ele = rdbLoadEncodedStringObject(rdb)) == NULL) return NULL;
1218  dec = getDecodedObject(ele);

1220  quicklistPushTail(o->ptr, dec->ptr, len);
1221  decrRefCount(dec);
1222  decrRefCount(ele);
1223  }
1224  } else if (rdbtype == RDB_TYPE_SET) {
1225  /* Read Set value */
1226  if ((len = rdbLoadLen(rdb,NULL)) == RDB_LENERR) return NULL;
1227

1229  if (len > server.set_max_intset_entries) {
1230  o = createSetObject();
1231  /* It's faster to expand the dict to the right size asap in order
1232  * to avoid rehashing */
1233  if (len > DICT_HT_INITIAL_SIZE)
1234  dictExpand(o->ptr,len);
1235  } else {
1236  o = createIntsetObject();
1237  }
1238 
1239  /* Load every single element of the set */
1240  for (i = 0; i < len; i++) {
1241  long long llval;
1242  sds sdsele;
1243 
1244  if ((sdsele = rdbGenericLoadStringObject(rdb,RDB_LOAD_SDS,NULL))
1245  == NULL) return NULL;

1248  /* Fetch integer value from element. */
1249  if (isSdsRepresentableAsLongLong(sdsele,&llval) == C_OK) {
1250  o->ptr = intsetAdd(o->ptr,llval,NULL);
1251  } else {
1252  setTypeConvert(o,OBJ_ENCODING_HT);
1253  dictExpand(o->ptr,len);
1254  }
1255  }
1256

1258  * to a regular hash table encoded set. */
1259  if (o->encoding == OBJ_ENCODING_HT) {
1260  dictAdd((dict*)o->ptr,sdsele,NULL);
1261  } else {
1262  sdsfree(sdsele);
1263  }
1264  }
1265  } else if (rdbtype == RDB_TYPE_ZSET_2 || rdbtype == RDB_TYPE_ZSET) {
1266  /* Read list/set value. */

1268  size_t maxelelen = 0;

1272  o = createZsetObject();

1274 
1275  /* Load every single element of the sorted set. */

1279  zskiplistNode *znode;

1281  if ((sdsele = rdbGenericLoadStringObject(rdb,RDB_LOAD_SDS,NULL))
1282  == NULL) return NULL;
1283

1285  if (rdbLoadBinaryDoubleValue(rdb,&score) == -1) return NULL;
1286  } else {
1287  if (rdbLoadDoubleValue(rdb,&score) == -1) return NULL;
1288  }
1289 
1290  /* Don't care about integer-encoded strings. */
1291  if (sdslen(sdsele) > maxelelen) maxelelen = sdslen(sdsele);
1292 
1293  znode = zslInsert(zs->zsl,score,sdsele);
1294  dictAdd(zs->dict,sdsele,&znode->score);
1295  }
1296

1298  if (zsetLength(o) <= server.zset_max_ziplist_entries &&
1299  maxelelen <= server.zset_max_ziplist_value)
1300  zsetConvert(o,OBJ_ENCODING_ZIPLIST);
1301  } else if (rdbtype == RDB_TYPE_HASH) {
1302  uint64_t len;
1303  int ret;
1304  sds field, value;
1305 
1306  len = rdbLoadLen(rdb, NULL);
1307  if (len == RDB_LENERR) return NULL;
1308 
1309  o = createHashObject();
1310 
1311  /* Too many entries? Use a hash table. */
1312  if (len > server.hash_max_ziplist_entries)
1313  hashTypeConvert(o, OBJ_ENCODING_HT);
1314 
1315  /* Load every field and value into the ziplist */
1316  while (o->encoding == OBJ_ENCODING_ZIPLIST && len > 0) {
1317  len--;
1318  /* Load raw strings */
1319  if ((field = rdbGenericLoadStringObject(rdb,RDB_LOAD_SDS,NULL))
1320  == NULL) return NULL;
1321  if ((value = rdbGenericLoadStringObject(rdb,RDB_LOAD_SDS,NULL))
1322  == NULL) return NULL;
1323

1326  sdslen(field), ZIPLIST_TAIL);
1327  o->ptr = ziplistPush(o->ptr, (unsigned char*)value,
1328  sdslen(value), ZIPLIST_TAIL);
1329 
1330  /* Convert to hash table if size threshold is exceeded */
1331  if (sdslen(field) > server.hash_max_ziplist_value ||
1332  sdslen(value) > server.hash_max_ziplist_value)
1333  {
1334  sdsfree(field);
1335  sdsfree(value);
1336  hashTypeConvert(o, OBJ_ENCODING_HT);
1337  break;
1338  }
1339  sdsfree(field);
1340  sdsfree(value);
1341  }
1342 
1343  /* Load remaining fields and values into the hash table */
1344  while (o->encoding == OBJ_ENCODING_HT && len > 0) {
1345  len--;
1346  /* Load encoded strings */
1347  if ((field = rdbGenericLoadStringObject(rdb,RDB_LOAD_SDS,NULL))
1348  == NULL) return NULL;
1349  if ((value = rdbGenericLoadStringObject(rdb,RDB_LOAD_SDS,NULL))
1350  == NULL) return NULL;
1351 
1352  /* Add pair to hash table */
1353  ret = dictAdd((dict*)o->ptr, field, value);
1354  if (ret == DICT_ERR) {
1355  rdbExitReportCorruptRDB("Duplicate keys detected");
1356  }
1357  }
1358 
1359  /* All pairs should be read by now */
1360  serverAssert(len == 0);
1361  } else if (rdbtype == RDB_TYPE_LIST_QUICKLIST) {
1362  if ((len = rdbLoadLen(rdb,NULL)) == RDB_LENERR) return NULL;
1363  o = createQuicklistObject();
1364  quicklistSetOptions(o->ptr, server.list_max_ziplist_size,
1365  server.list_compress_depth);
1366

1368  unsigned char *zl =
1369  rdbGenericLoadStringObject(rdb,RDB_LOAD_PLAIN,NULL);
1370  if (zl == NULL) return NULL;
1371  quicklistAppendZiplist(o->ptr, zl);
1372  }
1373  } else if (rdbtype == RDB_TYPE_HASH_ZIPMAP ||
1374  rdbtype == RDB_TYPE_LIST_ZIPLIST ||
1375  rdbtype == RDB_TYPE_SET_INTSET ||
1376  rdbtype == RDB_TYPE_ZSET_ZIPLIST ||
1377  rdbtype == RDB_TYPE_HASH_ZIPLIST)
1378  {

1380  rdbGenericLoadStringObject(rdb,RDB_LOAD_PLAIN,NULL);
1381  if (encoded == NULL) return NULL;
1382  o = createObject(OBJ_STRING,encoded); /* Obj type fixed below. */
1383

1385  * data type into the base type if accordingly to the current
1386  * configuration there are too many elements in the encoded data
1387  * type. Note that we only check the length and not max element
1388  * size as this is an O(N) scan. Eventually everything will get
1389  * converted. */
1390  switch(rdbtype) {
1391  case RDB_TYPE_HASH_ZIPMAP:
1392  /* Convert to ziplist encoded hash. This must be deprecated
1393  * when loading dumps created by Redis 2.4 gets deprecated. */

1400

1402  if (flen > maxlen) maxlen = flen;
1403  if (vlen > maxlen) maxlen = vlen;
1404  zl = ziplistPush(zl, fstr, flen, ZIPLIST_TAIL);
1405  zl = ziplistPush(zl, vstr, vlen, ZIPLIST_TAIL);
1406  }
1407 
1408  zfree(o->ptr);
1409  o->ptr = zl;
1410  o->type = OBJ_HASH;
1411  o->encoding = OBJ_ENCODING_ZIPLIST;
1412 
1413  if (hashTypeLength(o) > server.hash_max_ziplist_entries ||
1414  maxlen > server.hash_max_ziplist_value)
1415  {
1416  hashTypeConvert(o, OBJ_ENCODING_HT);
1417  }
1418  }
1419  break;
1420  case RDB_TYPE_LIST_ZIPLIST:
1421  o->type = OBJ_LIST;
1422  o->encoding = OBJ_ENCODING_ZIPLIST;
1423  listTypeConvert(o,OBJ_ENCODING_QUICKLIST);
1424  break;
1425  case RDB_TYPE_SET_INTSET:
1426  o->type = OBJ_SET;
1427  o->encoding = OBJ_ENCODING_INTSET;
1428  if (intsetLen(o->ptr) > server.set_max_intset_entries)
1429  setTypeConvert(o,OBJ_ENCODING_HT);
1430  break;
1431  case RDB_TYPE_ZSET_ZIPLIST:
1432  o->type = OBJ_ZSET;
1433  o->encoding = OBJ_ENCODING_ZIPLIST;
1434  if (zsetLength(o) > server.zset_max_ziplist_entries)
1435  zsetConvert(o,OBJ_ENCODING_SKIPLIST);
1436  break;
1437  case RDB_TYPE_HASH_ZIPLIST:
1438  o->type = OBJ_HASH;
1439  o->encoding = OBJ_ENCODING_ZIPLIST;
1440  if (hashTypeLength(o) > server.hash_max_ziplist_entries)
1441  hashTypeConvert(o, OBJ_ENCODING_HT);

1445  break;
1446  }
1447  } else if (rdbtype == RDB_TYPE_STREAM_LISTPACKS) {
1448  o = createStreamObject();
1449  stream *s = o->ptr;
1450  uint64_t listpacks = rdbLoadLen(rdb,NULL);

1455  * relatively to this ID. */
1456  sds nodekey = rdbGenericLoadStringObject(rdb,RDB_LOAD_SDS,NULL);
1457  if (sdslen(nodekey) != sizeof(streamID)) {
1458  rdbExitReportCorruptRDB("Stream node key entry is not the "
1459  "size of a stream ID");

1469  * deletion we should remove the radix tree key if the
1470  * resulting listpack is emtpy. */
1471  rdbExitReportCorruptRDB("Empty listpack inside stream");
1472  }
1473

1476  (unsigned char*)nodekey,sizeof(streamID),lp,NULL);
1477  sdsfree(nodekey);
1478  if (!retval)
1479  rdbExitReportCorruptRDB("Listpack re-added with existing key");
1480  }
1481  /* Load total number of items inside the stream. */
1482  s->length = rdbLoadLen(rdb,NULL);
1483  /* Load the last entry ID. */
1484  s->last_id.ms = rdbLoadLen(rdb,NULL);
1485  s->last_id.seq = rdbLoadLen(rdb,NULL);
1486  } else if (rdbtype == RDB_TYPE_MODULE || rdbtype == RDB_TYPE_MODULE_2) {
1487  uint64_t moduleid = rdbLoadLen(rdb,NULL);
1488  moduleType *mt = moduleTypeLookupModuleByID(moduleid);
1489  char name[10];
1490

1492  return rdbLoadCheckModuleValue(rdb,name);

1498  }

1500  moduleInitIOContext(io,mt,rdb);
1501  io.ver = (rdbtype == RDB_TYPE_MODULE) ? 1 : 2;
1502  /* Call the rdb_load method of the module providing the 10 bit
1503  * encoding version in the lower 10 bits of the module ID. */
1504  void *ptr = mt->rdb_load(&io,moduleid&1023);
1505  if (io.ctx) {
1506  moduleFreeContext(io.ctx);
1507  zfree(io.ctx);
1508  }
1509 
1510  /* Module v2 serialization has an EOF mark at the end. */
1511  if (io.ver == 2) {
1512  uint64_t eof = rdbLoadLen(rdb,NULL);
1513  if (eof != RDB_MODULE_OPCODE_EOF) {
1514  serverLog(LL_WARNING,"The RDB file contains module data for the module '%s' that is not terminated by the proper module value EOF marker", name);
1515  exit(1);
1516  }
1517  }
1518 
1519  if (ptr == NULL) {
1520  moduleTypeNameByID(name,moduleid);
1521  serverLog(LL_WARNING,"The RDB file contains module data for the module type '%s', that the responsible module is not able to load. Check for modules log above for additional clues.", name);
1522  exit(1);
1523  }
1524  o = createModuleObject(mt,ptr);
1525  } else {
1526  rdbExitReportCorruptRDB("Unknown RDB encoding type %d",rdbtype);
1527  }
1528  return o;
1529 }

1578 /* Load an RDB file from the rio stream 'rdb'. On success C_OK is returned,
1579  * otherwise C_ERR is returned and 'errno' is set accordingly. */
1580 int rdbLoadRio(rio *rdb, rdbSaveInfo *rsi) {
1581  uint64_t dbid;
1582  int type, rdbver;
1583  redisDb *db = server.db+0;
1584  char buf[1024];
1585  long long expiretime, now = mstime();
1586 
1587  rdb->update_cksum = rdbLoadProgressCallback;
1588  rdb->max_processing_chunk = server.loading_process_events_interval_bytes;
1589  if (rioRead(rdb,buf,9) == 0) goto eoferr;
1590  buf[9] = '\0';
1591  if (memcmp(buf,"REDIS",5) != 0) {
1592  serverLog(LL_WARNING,"Wrong signature trying to load DB from file");
1593  errno = EINVAL;
1594  return C_ERR;
1595  }
1596  rdbver = atoi(buf+5);
1597  if (rdbver < 1 || rdbver > RDB_VERSION) {
1598  serverLog(LL_WARNING,"Can't handle RDB format version %d",rdbver);
1599  errno = EINVAL;
1600  return C_ERR;
1601  }
1602 
1603  while(1) {
1604  robj *key, *val;
1605  expiretime = -1;
1606 
1607  /* Read type. */
1608  if ((type = rdbLoadType(rdb)) == -1) goto eoferr;
1609 
1610  /* Handle special types. */
1611  if (type == RDB_OPCODE_EXPIRETIME) {
1612  /* EXPIRETIME: load an expire associated with the next key
1613  * to load. Note that after loading an expire we need to
1614  * load the actual type, and continue. */
1615  if ((expiretime = rdbLoadTime(rdb)) == -1) goto eoferr;
1616  /* We read the time so we need to read the object type again. */
1617  if ((type = rdbLoadType(rdb)) == -1) goto eoferr;
1618  /* the EXPIRETIME opcode specifies time in seconds, so convert
1619  * into milliseconds. */
1620  expiretime *= 1000;
1621  } else if (type == RDB_OPCODE_EXPIRETIME_MS) {
1622  /* EXPIRETIME_MS: milliseconds precision expire times introduced
1623  * with RDB v3. Like EXPIRETIME but no with more precision. */
1624  if ((expiretime = rdbLoadMillisecondTime(rdb)) == -1) goto eoferr;
1625  /* We read the time so we need to read the object type again. */
1626  if ((type = rdbLoadType(rdb)) == -1) goto eoferr;
1627  } else if (type == RDB_OPCODE_EOF) {
1628  /* EOF: End of file, exit the main loop. */
1629  break;
1630  } else if (type == RDB_OPCODE_SELECTDB) {
1631  /* SELECTDB: Select the specified database. */
1632  if ((dbid = rdbLoadLen(rdb,NULL)) == RDB_LENERR)
1633  goto eoferr;
1634  if (dbid >= (unsigned)server.dbnum) {
1635  serverLog(LL_WARNING,
1636  "FATAL: Data file was created with a Redis "
1637  "server configured to handle more than %d "
1638  "databases. Exiting\n", server.dbnum);
1639  exit(1);
1640  }
1641  db = server.db+dbid;
1642  continue; /* Read type again. */
1643  } else if (type == RDB_OPCODE_RESIZEDB) {
1644  /* RESIZEDB: Hint about the size of the keys in the currently
1645  * selected data base, in order to avoid useless rehashing. */
1646  uint64_t db_size, expires_size;
1647  if ((db_size = rdbLoadLen(rdb,NULL)) == RDB_LENERR)
1648  goto eoferr;
1649  if ((expires_size = rdbLoadLen(rdb,NULL)) == RDB_LENERR)
1650  goto eoferr;
1651  dictExpand(db->dict,db_size);
1652  dictExpand(db->expires,expires_size);
1653  continue; /* Read type again. */
1654  } else if (type == RDB_OPCODE_AUX) {
1655  /* AUX: generic string-string fields. Use to add state to RDB
1656  * which is backward compatible. Implementations of RDB loading
1657  * are requierd to skip AUX fields they don't understand.
1658  *
1659  * An AUX field is composed of two strings: key and value. */
1660  robj *auxkey, *auxval;
1661  if ((auxkey = rdbLoadStringObject(rdb)) == NULL) goto eoferr;
1662  if ((auxval = rdbLoadStringObject(rdb)) == NULL) goto eoferr;
1663 
1664  if (((char*)auxkey->ptr)[0] == '%') {
1665  /* All the fields with a name staring with '%' are considered
1666  * information fields and are logged at startup with a log
1667  * level of NOTICE. */
1668  serverLog(LL_NOTICE,"RDB '%s': %s",
1669  (char*)auxkey->ptr,
1670  (char*)auxval->ptr);
1671  } else if (!strcasecmp(auxkey->ptr,"repl-stream-db")) {
1672  if (rsi) rsi->repl_stream_db = atoi(auxval->ptr);
1673  } else if (!strcasecmp(auxkey->ptr,"repl-id")) {
1674  if (rsi && sdslen(auxval->ptr) == CONFIG_RUN_ID_SIZE) {
1675  memcpy(rsi->repl_id,auxval->ptr,CONFIG_RUN_ID_SIZE+1);
